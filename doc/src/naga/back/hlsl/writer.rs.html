175    /// Generates statements to be inserted immediately before and at the very
176    /// start of the body of each loop, to defeat infinite loop reasoning.
177    /// The 0th item of the returned tuple should be inserted immediately prior
178    /// to the loop and the 1st item should be inserted at the very start of
179    /// the loop body.
180    ///
181    /// See [`back::msl::Writer::gen_force_bounded_loop_statements`] for details.
182    fn gen_force_bounded_loop_statements(
183        &mut self,
184        level: back::Level,
185    ) -> Option<(String, String)> {
186        if !self.options.force_loop_bounding {
187            return None;
188        }
189
190        let loop_bound_name = self.namer.call("loop_bound");
191        let max = u32::MAX;
192        // Count down from u32::MAX rather than up from 0 to avoid hang on
193        // certain Intel drivers. See <https://github.com/gfx-rs/wgpu/issues/7319>.
194        let decl = format!("{level}uint2 {loop_bound_name} = uint2({max}u, {max}u);");
195        let level = level.next();
196        let break_and_inc = format!(
197            "{level}if (all({loop_bound_name} == uint2(0u, 0u))) {{ break; }}
198{level}{loop_bound_name} -= uint2({loop_bound_name}.y == 0u, 1u);"
199        );
200
201        Some((decl, break_and_inc))
202    }

204    /// Helper method used to find which expressions of a given function require baking
205    ///
206    /// # Notes
207    /// Clears `need_bake_expressions` set before adding to it
208    fn update_expressions_to_bake(
209        &mut self,
210        module: &Module,
211        func: &crate::Function,
212        info: &valid::FunctionInfo,
213    ) {
214        use crate::Expression;
215        self.need_bake_expressions.clear();
216        for (exp_handle, expr) in func.expressions.iter() {
217            let expr_info = &info[exp_handle];
218            let min_ref_count = func.expressions[exp_handle].bake_ref_count();
219            if min_ref_count <= expr_info.ref_count {
220                self.need_bake_expressions.insert(exp_handle);
221            }
222
223            if let Expression::Math { fun, arg, arg1, .. } = *expr {
224                match fun {
225                    crate::MathFunction::Asinh
226                    | crate::MathFunction::Acosh
227                    | crate::MathFunction::Atanh
228                    | crate::MathFunction::Unpack2x16float
229                    | crate::MathFunction::Unpack2x16snorm
230                    | crate::MathFunction::Unpack2x16unorm
231                    | crate::MathFunction::Unpack4x8snorm
232                    | crate::MathFunction::Unpack4x8unorm
233                    | crate::MathFunction::Unpack4xI8
234                    | crate::MathFunction::Unpack4xU8
235                    | crate::MathFunction::Pack2x16float
236                    | crate::MathFunction::Pack2x16snorm
237                    | crate::MathFunction::Pack2x16unorm
238                    | crate::MathFunction::Pack4x8snorm
239                    | crate::MathFunction::Pack4x8unorm
240                    | crate::MathFunction::Pack4xI8
241                    | crate::MathFunction::Pack4xU8
242                    | crate::MathFunction::Pack4xI8Clamp
243                    | crate::MathFunction::Pack4xU8Clamp => {
244                        self.need_bake_expressions.insert(arg);
245                    }
246                    crate::MathFunction::CountLeadingZeros => {
247                        let inner = info[exp_handle].ty.inner_with(&module.types);
248                        if let Some(ScalarKind::Sint) = inner.scalar_kind() {
249                            self.need_bake_expressions.insert(arg);
250                        }
251                    }
252                    crate::MathFunction::Dot4U8Packed | crate::MathFunction::Dot4I8Packed => {
253                        self.need_bake_expressions.insert(arg);
254                        self.need_bake_expressions.insert(arg1.unwrap());
255                    }
256                    _ => {}
257                }
258            }
259
260            if let Expression::Derivative { axis, ctrl, expr } = *expr {
261                use crate::{DerivativeAxis as Axis, DerivativeControl as Ctrl};
262                if axis == Axis::Width && (ctrl == Ctrl::Coarse || ctrl == Ctrl::Fine) {
263                    self.need_bake_expressions.insert(expr);
264                }
265            }
266
267            if let Expression::GlobalVariable(_) = *expr {
268                let inner = info[exp_handle].ty.inner_with(&module.types);
269
270                if let TypeInner::Sampler { .. } = *inner {
271                    self.need_bake_expressions.insert(exp_handle);
272                }
273            }
274        }
275        for statement in func.body.iter() {
276            match *statement {
277                crate::Statement::SubgroupCollectiveOperation {
278                    op: _,
279                    collective_op: crate::CollectiveOperation::InclusiveScan,
280                    argument,
281                    result: _,
282                } => {
283                    self.need_bake_expressions.insert(argument);
284                }
285                crate::Statement::Atomic {
286                    fun: crate::AtomicFunction::Exchange { compare: Some(cmp) },
287                    ..
288                } => {
289                    self.need_bake_expressions.insert(cmp);
290                }
291                _ => {}
292            }
293        }
294    }

296    pub fn write(
297        &mut self,
298        module: &Module,
299        module_info: &valid::ModuleInfo,
300        fragment_entry_point: Option<&FragmentEntryPoint<'_>>,
301    ) -> Result<super::ReflectionInfo, Error> {
302        self.reset(module);
303
304        // Write special constants, if needed
305        if let Some(ref bt) = self.options.special_constants_binding {
306            writeln!(self.out, "struct {SPECIAL_CBUF_TYPE} {{")?;
307            writeln!(self.out, "{}int {};", back::INDENT, SPECIAL_FIRST_VERTEX)?;
308            writeln!(self.out, "{}int {};", back::INDENT, SPECIAL_FIRST_INSTANCE)?;
309            writeln!(self.out, "{}uint {};", back::INDENT, SPECIAL_OTHER)?;
310            writeln!(self.out, "}};")?;
311            write!(
312                self.out,
313                "ConstantBuffer<{}> {}: register(b{}",
314                SPECIAL_CBUF_TYPE, SPECIAL_CBUF_VAR, bt.register
315            )?;
316            if bt.space != 0 {
317                write!(self.out, ", space{}", bt.space)?;
318            }
319            writeln!(self.out, ");")?;
320
321            // Extra newline for readability
322            writeln!(self.out)?;
323        }
324
325        for (group, bt) in self.options.dynamic_storage_buffer_offsets_targets.iter() {
326            writeln!(self.out, "struct __dynamic_buffer_offsetsTy{group} {{")?;
327            for i in 0..bt.size {
328                writeln!(self.out, "{}uint _{};", back::INDENT, i)?;
329            }
330            writeln!(self.out, "}};")?;
331            writeln!(
332                self.out,
333                "ConstantBuffer<__dynamic_buffer_offsetsTy{}> __dynamic_buffer_offsets{}: register(b{}, space{});",
334                group, group, bt.register, bt.space
335            )?;
336
337            // Extra newline for readability
338            writeln!(self.out)?;
339        }
340
341        // Save all entry point output types
342        let ep_results = module
343            .entry_points
344            .iter()
345            .map(|ep| (ep.stage, ep.function.result.clone()))
346            .collect::<Vec<(ShaderStage, Option<crate::FunctionResult>)>>();
347
348        self.write_all_mat_cx2_typedefs_and_functions(module)?;
349
350        // Write all structs
351        for (handle, ty) in module.types.iter() {
352            if let TypeInner::Struct { ref members, span } = ty.inner {
353                if module.types[members.last().unwrap().ty]
354                    .inner
355                    .is_dynamically_sized(&module.types)
356                {
357                    // unsized arrays can only be in storage buffers,
358                    // for which we use `ByteAddressBuffer` anyway.
359                    continue;
360                }
361
362                let ep_result = ep_results.iter().find(|e| {
363                    if let Some(ref result) = e.1 {
364                        result.ty == handle
365                    } else {
366                        false
367                    }
368                });
369
370                self.write_struct(
371                    module,
372                    handle,
373                    members,
374                    span,
375                    ep_result.map(|r| (r.0, Io::Output)),
376                )?;
377                writeln!(self.out)?;
378            }
379        }
380
381        self.write_special_functions(module)?;
382
383        self.write_wrapped_expression_functions(module, &module.global_expressions, None)?;
384        self.write_wrapped_zero_value_functions(module, &module.global_expressions)?;
385
386        // Write all named constants
387        let mut constants = module
388            .constants
389            .iter()
390            .filter(|&(_, c)| c.name.is_some())
391            .peekable();
392        while let Some((handle, _)) = constants.next() {
393            self.write_global_constant(module, handle)?;
394            // Add extra newline for readability on last iteration
395            if constants.peek().is_none() {
396                writeln!(self.out)?;
397            }
398        }
399
400        // Write all globals
401        for (global, _) in module.global_variables.iter() {
402            self.write_global(module, global)?;
403        }
404
405        if !module.global_variables.is_empty() {
406            // Add extra newline for readability
407            writeln!(self.out)?;
408        }
409
410        let ep_range = get_entry_points(module, self.pipeline_options.entry_point.as_ref())
411            .map_err(|(stage, name)| Error::EntryPointNotFound(stage, name))?;
412
413        // Write all entry points wrapped structs
414        for index in ep_range.clone() {
415            let ep = &module.entry_points[index];
416            let ep_name = self.names[&NameKey::EntryPoint(index as u16)].clone();
417            let ep_io = self.write_ep_interface(
418                module,
419                &ep.function,
420                ep.stage,
421                &ep_name,
422                fragment_entry_point,
423            )?;
424            self.entry_point_io.insert(index, ep_io);
425        }
426
427        // Write all regular functions
428        for (handle, function) in module.functions.iter() {
429            let info = &module_info[handle];
430
431            // Check if all of the globals are accessible
432            if !self.options.fake_missing_bindings {
433                if let Some((var_handle, _)) =
434                    module
435                        .global_variables
436                        .iter()
437                        .find(|&(var_handle, var)| match var.binding {
438                            Some(ref binding) if !info[var_handle].is_empty() => {
439                                self.options.resolve_resource_binding(binding).is_err()
440                                    && self
441                                        .options
442                                        .resolve_external_texture_resource_binding(binding)
443                                        .is_err()
444                            }
445                            _ => false,
446                        })
447                {
448                    log::debug!(
449                        "Skipping function {:?} (name {:?}) because global {:?} is inaccessible",
450                        handle,
451                        function.name,
452                        var_handle
453                    );
454                    continue;
455                }
456            }
457
458            let ctx = back::FunctionCtx {
459                ty: back::FunctionType::Function(handle),
460                info,
461                expressions: &function.expressions,
462                named_expressions: &function.named_expressions,
463            };
464            let name = self.names[&NameKey::Function(handle)].clone();
465
466            self.write_wrapped_functions(module, &ctx)?;
467
468            self.write_function(module, name.as_str(), function, &ctx, info)?;
469
470            writeln!(self.out)?;
471        }
472
473        let mut translated_ep_names = Vec::with_capacity(ep_range.len());
474
475        // Write all entry points
476        for index in ep_range {
477            let ep = &module.entry_points[index];
478            let info = module_info.get_entry_point(index);
479
480            if !self.options.fake_missing_bindings {
481                let mut ep_error = None;
482                for (var_handle, var) in module.global_variables.iter() {
483                    match var.binding {
484                        Some(ref binding) if !info[var_handle].is_empty() => {
485                            if let Err(err) = self.options.resolve_resource_binding(binding) {
486                                if self
487                                    .options
488                                    .resolve_external_texture_resource_binding(binding)
489                                    .is_err()
490                                {
491                                    ep_error = Some(err);
492                                    break;
493                                }
494                            }
495                        }
496                        _ => {}
497                    }
498                }
499                if let Some(err) = ep_error {
500                    translated_ep_names.push(Err(err));
501                    continue;
502                }
503            }
504
505            let ctx = back::FunctionCtx {
506                ty: back::FunctionType::EntryPoint(index as u16),
507                info,
508                expressions: &ep.function.expressions,
509                named_expressions: &ep.function.named_expressions,
510            };
511
512            self.write_wrapped_functions(module, &ctx)?;
513
514            if ep.stage.compute_like() {
515                // HLSL is calling workgroup size "num threads"
516                let num_threads = ep.workgroup_size;
517                writeln!(
518                    self.out,
519                    "[numthreads({}, {}, {})]",
520                    num_threads[0], num_threads[1], num_threads[2]
521                )?;
522            }
523
524            let name = self.names[&NameKey::EntryPoint(index as u16)].clone();
525            self.write_function(module, &name, &ep.function, &ctx, info)?;
526
527            if index < module.entry_points.len() - 1 {
528                writeln!(self.out)?;
529            }
530
531            translated_ep_names.push(Ok(name));
532        }
533
534        Ok(super::ReflectionInfo {
535            entry_point_names: translated_ep_names,
536        })
537    }

539    fn write_modifier(&mut self, binding: &crate::Binding) -> BackendResult {
540        match *binding {
541            crate::Binding::BuiltIn(crate::BuiltIn::Position { invariant: true }) => {
542                write!(self.out, "precise ")?;
543            }
544            crate::Binding::BuiltIn(crate::BuiltIn::Barycentric { perspective: false }) => {
545                write!(self.out, "noperspective ")?;
546            }
547            crate::Binding::Location {
548                interpolation,
549                sampling,
550                ..
551            } => {
552                if let Some(interpolation) = interpolation {
553                    if let Some(string) = interpolation.to_hlsl_str() {
554                        write!(self.out, "{string} ")?
555                    }
556                }
557
558                if let Some(sampling) = sampling {
559                    if let Some(string) = sampling.to_hlsl_str() {
560                        write!(self.out, "{string} ")?
561                    }
562                }
563            }
564            crate::Binding::BuiltIn(_) => {}
565        }
566
567        Ok(())
568    }

570    //TODO: we could force fragment outputs to always go through `entry_point_io.output` path

572    fn write_semantic(
573        &mut self,
574        binding: &Option<crate::Binding>,
575        stage: Option<(ShaderStage, Io)>,
576    ) -> BackendResult {
577        match *binding {
578            Some(crate::Binding::BuiltIn(builtin)) if !is_subgroup_builtin_binding(binding) => {
579                if builtin == crate::BuiltIn::ViewIndex
580                    && self.options.shader_model < ShaderModel::V6_1
581                {
582                    return Err(Error::ShaderModelTooLow(
583                        "used @builtin(view_index) or SV_ViewID".to_string(),
584                        ShaderModel::V6_1,
585                    ));
586                }
587                let builtin_str = builtin.to_hlsl_str()?;
588                write!(self.out, " : {builtin_str}")?;
589            }
590            Some(crate::Binding::Location {
591                blend_src: Some(1), ..
592            }) => {
593                write!(self.out, " : SV_Target1")?;
594            }
595            Some(crate::Binding::Location { location, .. }) => {
596                if stage == Some((ShaderStage::Fragment, Io::Output)) {
597                    write!(self.out, " : SV_Target{location}")?;
598                } else {
599                    write!(self.out, " : {LOCATION_SEMANTIC}{location}")?;
600                }
601            }
602            _ => {}
603        }
604
605        Ok(())
606    }

608    fn write_interface_struct(
609        &mut self,
610        module: &Module,
611        shader_stage: (ShaderStage, Io),
612        struct_name: String,
613        mut members: Vec<EpStructMember>,
614    ) -> Result<EntryPointBinding, Error> {
615        // Sort the members so that first come the user-defined varyings
616        // in ascending locations, and then built-ins. This allows VS and FS
617        // interfaces to match with regards to order.
618        members.sort_by_key(|m| InterfaceKey::new(m.binding.as_ref()));
619
620        write!(self.out, "struct {struct_name}")?;
621        writeln!(self.out, " {{")?;
622        for m in members.iter() {
623            // Sanity check that each IO member is a built-in or is assigned a
624            // location. Also see note about nesting in `write_ep_input_struct`.
625            debug_assert!(m.binding.is_some());
626
627            if is_subgroup_builtin_binding(&m.binding) {
628                continue;
629            }
630            write!(self.out, "{}", back::INDENT)?;
631            if let Some(ref binding) = m.binding {
632                self.write_modifier(binding)?;
633            }
634            self.write_type(module, m.ty)?;
635            write!(self.out, " {}", &m.name)?;
636            self.write_semantic(&m.binding, Some(shader_stage))?;
637            writeln!(self.out, ";")?;
638        }
639        if members.iter().any(|arg| {
640            matches!(
641                arg.binding,
642                Some(crate::Binding::BuiltIn(crate::BuiltIn::SubgroupId))
643            )
644        }) {
645            writeln!(
646                self.out,
647                "{}uint __local_invocation_index : SV_GroupIndex;",
648                back::INDENT
649            )?;
650        }
651        writeln!(self.out, "}};")?;
652        writeln!(self.out)?;
653
654        // See ordering notes on EntryPointInterface fields
655        match shader_stage.1 {
656            Io::Input => {
657                // bring back the original order
658                members.sort_by_key(|m| m.index);
659            }
660            Io::Output => {
661                // keep it sorted by binding
662            }
663        }
664
665        Ok(EntryPointBinding {
666            arg_name: self.namer.call(struct_name.to_lowercase().as_str()),
667            ty_name: struct_name,
668            members,
669        })
670    }

672    /// Flatten all entry point arguments into a single struct.
673    /// This is needed since we need to re-order them: first placing user locations,
674    /// then built-ins.
675    fn write_ep_input_struct(
676        &mut self,
677        module: &Module,
678        func: &crate::Function,
679        stage: ShaderStage,
680        entry_point_name: &str,
681    ) -> Result<EntryPointBinding, Error> {
682        let struct_name = format!("{stage:?}Input_{entry_point_name}");
683
684        let mut fake_members = Vec::new();
685        for arg in func.arguments.iter() {
686            // NOTE: We don't need to handle nesting structs. All members must
687            // be either built-ins or assigned a location. I.E. `binding` is
688            // `Some`. This is checked in `VaryingContext::validate`. See:
689            // https://gpuweb.github.io/gpuweb/wgsl/#input-output-locations
690            match module.types[arg.ty].inner {
691                TypeInner::Struct { ref members, .. } => {
692                    for member in members.iter() {
693                        let name = self.namer.call_or(&member.name, "member");
694                        let index = fake_members.len() as u32;
695                        fake_members.push(EpStructMember {
696                            name,
697                            ty: member.ty,
698                            binding: member.binding.clone(),
699                            index,
700                        });
701                    }
702                }
703                _ => {
704                    let member_name = self.namer.call_or(&arg.name, "member");
705                    let index = fake_members.len() as u32;
706                    fake_members.push(EpStructMember {
707                        name: member_name,
708                        ty: arg.ty,
709                        binding: arg.binding.clone(),
710                        index,
711                    });
712                }
713            }
714        }
715
716        self.write_interface_struct(module, (stage, Io::Input), struct_name, fake_members)
717    }

719    /// Flatten all entry point results into a single struct.
720    /// This is needed since we need to re-order them: first placing user locations,
721    /// then built-ins.
722    fn write_ep_output_struct(
723        &mut self,
724        module: &Module,
725        result: &crate::FunctionResult,
726        stage: ShaderStage,
727        entry_point_name: &str,
728        frag_ep: Option<&FragmentEntryPoint<'_>>,
729    ) -> Result<EntryPointBinding, Error> {
730        let struct_name = format!("{stage:?}Output_{entry_point_name}");
731
732        let empty = [];
733        let members = match module.types[result.ty].inner {
734            TypeInner::Struct { ref members, .. } => members,
735            ref other => {
736                log::error!("Unexpected {other:?} output type without a binding");
737                &empty[..]
738            }
739        };
740
741        // Gather list of fragment input locations. We use this below to remove user-defined
742        // varyings from VS outputs that aren't in the FS inputs. This makes the VS interface match
743        // as long as the FS inputs are a subset of the VS outputs. This is only applied if the
744        // writer is supplied with information about the fragment entry point.
745        let fs_input_locs = if let (Some(frag_ep), ShaderStage::Vertex) = (frag_ep, stage) {
746            let mut fs_input_locs = Vec::new();
747            for arg in frag_ep.func.arguments.iter() {
748                let mut push_if_location = |binding: &Option<crate::Binding>| match *binding {
749                    Some(crate::Binding::Location { location, .. }) => fs_input_locs.push(location),
750                    Some(crate::Binding::BuiltIn(_)) | None => {}
751                };
752
753                // NOTE: We don't need to handle struct nesting. See note in
754                // `write_ep_input_struct`.
755                match frag_ep.module.types[arg.ty].inner {
756                    TypeInner::Struct { ref members, .. } => {
757                        for member in members.iter() {
758                            push_if_location(&member.binding);
759                        }
760                    }
761                    _ => push_if_location(&arg.binding),
762                }
763            }
764            fs_input_locs.sort();
765            Some(fs_input_locs)
766        } else {
767            None
768        };
769
770        let mut fake_members = Vec::new();
771        for (index, member) in members.iter().enumerate() {
772            if let Some(ref fs_input_locs) = fs_input_locs {
773                match member.binding {
774                    Some(crate::Binding::Location { location, .. }) => {
775                        if fs_input_locs.binary_search(&location).is_err() {
776                            continue;
777                        }
778                    }
779                    Some(crate::Binding::BuiltIn(_)) | None => {}
780                }
781            }
782
783            let member_name = self.namer.call_or(&member.name, "member");
784            fake_members.push(EpStructMember {
785                name: member_name,
786                ty: member.ty,
787                binding: member.binding.clone(),
788                index: index as u32,
789            });
790        }
791
792        self.write_interface_struct(module, (stage, Io::Output), struct_name, fake_members)
793    }

795    /// Writes special interface structures for an entry point. The special structures have
796    /// all the fields flattened into them and sorted by binding. They are needed to emulate
797    /// subgroup built-ins and to make the interfaces between VS outputs and FS inputs match.
798    fn write_ep_interface(
799        &mut self,
800        module: &Module,
801        func: &crate::Function,
802        stage: ShaderStage,
803        ep_name: &str,
804        frag_ep: Option<&FragmentEntryPoint<'_>>,
805    ) -> Result<EntryPointInterface, Error> {
806        Ok(EntryPointInterface {
807            input: if !func.arguments.is_empty()
808                && (stage == ShaderStage::Fragment
809                    || func
810                        .arguments
811                        .iter()
812                        .any(|arg| is_subgroup_builtin_binding(&arg.binding)))
813            {
814                Some(self.write_ep_input_struct(module, func, stage, ep_name)?)

818            output: match func.result {

820                    Some(self.write_ep_output_struct(module, fr, stage, ep_name, frag_ep)?)
821                }
822                _ => None,
823            },
824        })
825    }

859    /// Write an entry point preface that initializes the arguments as specified in IR.
860    fn write_ep_arguments_initialization(
861        &mut self,
862        module: &Module,
863        func: &crate::Function,
864        ep_index: u16,
865    ) -> BackendResult {
866        let ep = &module.entry_points[ep_index as usize];
867        let ep_input = match self
868            .entry_point_io
869            .get_mut(&(ep_index as usize))
870            .unwrap()
871            .input
872            .take()
873        {
874            Some(ep_input) => ep_input,
875            None => return Ok(()),
876        };
877        let mut fake_iter = ep_input.members.iter();
878        for (arg_index, arg) in func.arguments.iter().enumerate() {
879            write!(self.out, "{}", back::INDENT)?;
880            self.write_type(module, arg.ty)?;
881            let arg_name = &self.names[&NameKey::EntryPointArgument(ep_index, arg_index as u32)];
882            write!(self.out, " {arg_name}")?;
883            match module.types[arg.ty].inner {
884                TypeInner::Array { base, size, .. } => {
885                    self.write_array_size(module, base, size)?;
886                    write!(self.out, " = ")?;
887                    self.write_ep_argument_initialization(
888                        ep,
889                        &ep_input,
890                        fake_iter.next().unwrap(),
891                    )?;
892                    writeln!(self.out, ";")?;
893                }
894                TypeInner::Struct { ref members, .. } => {
895                    write!(self.out, " = {{ ")?;
896                    for index in 0..members.len() {
897                        if index != 0 {
898                            write!(self.out, ", ")?;
899                        }
900                        self.write_ep_argument_initialization(
901                            ep,
902                            &ep_input,
903                            fake_iter.next().unwrap(),
904                        )?;
905                    }
906                    writeln!(self.out, " }};")?;
907                }
908                _ => {
909                    write!(self.out, " = ")?;
910                    self.write_ep_argument_initialization(
911                        ep,
912                        &ep_input,
913                        fake_iter.next().unwrap(),
914                    )?;
915                    writeln!(self.out, ";")?;
916                }
917            }
918        }
919        assert!(fake_iter.next().is_none());
920        Ok(())
921    }

923    /// Helper method used to write global variables
924    /// # Notes
925    /// Always adds a newline
926    fn write_global(
927        &mut self,
928        module: &Module,
929        handle: Handle<crate::GlobalVariable>,
930    ) -> BackendResult {
931        let global = &module.global_variables[handle];
932        let inner = &module.types[global.ty].inner;
933
934        let handle_ty = match *inner {
935            TypeInner::BindingArray { ref base, .. } => &module.types[*base].inner,
936            _ => inner,
937        };
938
939        // External textures are handled entirely differently, so defer entirely to that method.
940        // We do so prior to calling resolve_resource_binding() below, as we even need to resolve
941        // their bindings separately.
942        let is_external_texture = matches!(
943            *handle_ty,
944            TypeInner::Image {
945                class: crate::ImageClass::External,
946                ..
947            }
948        );
949        if is_external_texture {
950            return self.write_global_external_texture(module, handle, global);
951        }
952
953        if let Some(ref binding) = global.binding {
954            if let Err(err) = self.options.resolve_resource_binding(binding) {
955                log::debug!(
956                    "Skipping global {:?} (name {:?}) for being inaccessible: {}",
957                    handle,
958                    global.name,
959                    err,
960                );
961                return Ok(());
962            }
963        }
964
965        // Samplers are handled entirely differently, so defer entirely to that method.
966        let is_sampler = matches!(*handle_ty, TypeInner::Sampler { .. });
967
968        if is_sampler {
969            return self.write_global_sampler(module, handle, global);
970        }
971
972        // https://docs.microsoft.com/en-us/windows/win32/direct3dhlsl/dx-graphics-hlsl-variable-register
973        let register_ty = match global.space {
974            crate::AddressSpace::Function => unreachable!("Function address space"),
975            crate::AddressSpace::Private => {
976                write!(self.out, "static ")?;
977                self.write_type(module, global.ty)?;
978                ""
979            }
980            crate::AddressSpace::WorkGroup => {
981                write!(self.out, "groupshared ")?;
982                self.write_type(module, global.ty)?;
983                ""
984            }
985            crate::AddressSpace::TaskPayload => unimplemented!(),
986            crate::AddressSpace::Uniform => {
987                // constant buffer declarations are expected to be inlined, e.g.
988                // `cbuffer foo: register(b0) { field1: type1; }`
989                write!(self.out, "cbuffer")?;
990                "b"
991            }
992            crate::AddressSpace::Storage { access } => {
993                let (prefix, register) = if access.contains(crate::StorageAccess::STORE) {
994                    ("RW", "u")
995                } else {
996                    ("", "t")
997                };
998                write!(self.out, "{prefix}ByteAddressBuffer")?;
999                register
1000            }
1001            crate::AddressSpace::Handle => {
1002                let register = match *handle_ty {
1003                    // all storage textures are UAV, unconditionally
1004                    TypeInner::Image {
1005                        class: crate::ImageClass::Storage { .. },
1006                        ..
1007                    } => "u",
1008                    _ => "t",
1009                };
1010                self.write_type(module, global.ty)?;
1011                register
1012            }
1013            crate::AddressSpace::Immediate => {
1014                // The type of the immediates will be wrapped in `ConstantBuffer`
1015                write!(self.out, "ConstantBuffer<")?;
1016                "b"
1017            }
1018        };
1019
1020        // If the global is a immediate data write the type now because it will be a
1021        // generic argument to `ConstantBuffer`
1022        if global.space == crate::AddressSpace::Immediate {
1023            self.write_global_type(module, global.ty)?;
1024
1025            // need to write the array size if the type was emitted with `write_type`
1026            if let TypeInner::Array { base, size, .. } = module.types[global.ty].inner {
1027                self.write_array_size(module, base, size)?;
1028            }
1029
1030            // Close the angled brackets for the generic argument
1031            write!(self.out, ">")?;
1032        }
1033
1034        let name = &self.names[&NameKey::GlobalVariable(handle)];
1035        write!(self.out, " {name}")?;
1036
1037        // Immediates need to be assigned a binding explicitly by the consumer
1038        // since naga has no way to know the binding from the shader alone
1039        if global.space == crate::AddressSpace::Immediate {
1040            match module.types[global.ty].inner {
1041                TypeInner::Struct { .. } => {}
1042                _ => {
1043                    return Err(Error::Unimplemented(format!(
1044                        "push-constant '{name}' has non-struct type; tracked by: https://github.com/gfx-rs/wgpu/issues/5683"
1045                    )));
1046                }
1047            }
1048
1049            let target = self
1050                .options
1051                .immediates_target
1052                .as_ref()
1053                .expect("No bind target was defined for the immediates block");
1054            write!(self.out, ": register(b{}", target.register)?;
1055            if target.space != 0 {
1056                write!(self.out, ", space{}", target.space)?;
1057            }
1058            write!(self.out, ")")?;
1059        }
1060
1061        if let Some(ref binding) = global.binding {
1062            // this was already resolved earlier when we started evaluating an entry point.
1063            let bt = self.options.resolve_resource_binding(binding).unwrap();
1064
1065            // need to write the binding array size if the type was emitted with `write_type`
1066            if let TypeInner::BindingArray { base, size, .. } = module.types[global.ty].inner {
1067                if let Some(overridden_size) = bt.binding_array_size {
1068                    write!(self.out, "[{overridden_size}]")?;
1069                } else {
1070                    self.write_array_size(module, base, size)?;
1071                }
1072            }
1073
1074            write!(self.out, " : register({}{}", register_ty, bt.register)?;
1075            if bt.space != 0 {
1076                write!(self.out, ", space{}", bt.space)?;
1077            }
1078            write!(self.out, ")")?;
1079        } else {
1080            // need to write the array size if the type was emitted with `write_type`
1081            if let TypeInner::Array { base, size, .. } = module.types[global.ty].inner {
1082                self.write_array_size(module, base, size)?;
1083            }
1084            if global.space == crate::AddressSpace::Private {
1085                write!(self.out, " = ")?;
1086                if let Some(init) = global.init {
1087                    self.write_const_expression(module, init, &module.global_expressions)?;
1088                } else {
1089                    self.write_default_init(module, global.ty)?;
1090                }
1091            }
1092        }
1093
1094        if global.space == crate::AddressSpace::Uniform {
1095            write!(self.out, " {{ ")?;
1096
1097            self.write_global_type(module, global.ty)?;
1098
1099            write!(
1100                self.out,
1101                " {}",
1102                &self.names[&NameKey::GlobalVariable(handle)]
1103            )?;
1104
1105            // need to write the array size if the type was emitted with `write_type`
1106            if let TypeInner::Array { base, size, .. } = module.types[global.ty].inner {
1107                self.write_array_size(module, base, size)?;
1108            }
1109
1110            writeln!(self.out, "; }}")?;
1111        } else {
1112            writeln!(self.out, ";")?;
1113        }
1114
1115        Ok(())
1116    }

1118    fn write_global_sampler(
1119        &mut self,
1120        module: &Module,
1121        handle: Handle<crate::GlobalVariable>,
1122        global: &crate::GlobalVariable,
1123    ) -> BackendResult {
1124        let binding = *global.binding.as_ref().unwrap();
1125
1126        let key = super::SamplerIndexBufferKey {
1127            group: binding.group,
1128        };
1129        self.write_wrapped_sampler_buffer(key)?;
1130
1131        // This was already validated, so we can confidently unwrap it.
1132        let bt = self.options.resolve_resource_binding(&binding).unwrap();
1133
1134        match module.types[global.ty].inner {
1135            TypeInner::Sampler { comparison } => {
1136                // If we are generating a static access, we create a variable for the sampler.
1137                //
1138                // This prevents the DXIL from containing multiple lookups for the sampler, which
1139                // the backend compiler will then have to eliminate. AMD does seem to be able to
1140                // eliminate these, but better safe than sorry.
1141
1142                write!(self.out, "static const ")?;
1143                self.write_type(module, global.ty)?;
1144
1145                let heap_var = if comparison {
1146                    COMPARISON_SAMPLER_HEAP_VAR
1147                } else {
1148                    SAMPLER_HEAP_VAR
1149                };
1150
1151                let index_buffer_name = &self.wrapped.sampler_index_buffers[&key];
1152                let name = &self.names[&NameKey::GlobalVariable(handle)];
1153                writeln!(
1154                    self.out,
1155                    " {name} = {heap_var}[{index_buffer_name}[{register}]];",
1156                    register = bt.register
1157                )?;
1158            }
1159            TypeInner::BindingArray { .. } => {
1160                // If we are generating a binding array, we cannot directly access the sampler as the index
1161                // into the sampler index buffer is unknown at compile time. Instead we generate a constant
1162                // that represents the "base" index into the sampler index buffer. This constant is added
1163                // to the user provided index to get the final index into the sampler index buffer.
1164
1165                let name = &self.names[&NameKey::GlobalVariable(handle)];
1166                writeln!(
1167                    self.out,
1168                    "static const uint {name} = {register};",
1169                    register = bt.register
1170                )?;
1171            }
1172            _ => unreachable!(),
1173        };
1174
1175        Ok(())
1176    }

1309        writeln!(self.out, "struct {struct_name} {{")?;
1310
1311        let mut last_offset = 0;
1312        for (index, member) in members.iter().enumerate() {
1313            if member.binding.is_none() && member.offset > last_offset {
1314                // using int as padding should work as long as the backend
1315                // doesn't support a type that's less than 4 bytes in size
1316                // (Error::UnsupportedScalar catches this)
1317                let padding = (member.offset - last_offset) / 4;
1318                for i in 0..padding {
1319                    writeln!(self.out, "{}int _pad{}_{};", back::INDENT, index, i)?;
1320                }
1321            }
1322            let ty_inner = &module.types[member.ty].inner;
1323            last_offset = member.offset + ty_inner.size_hlsl(module.to_ctx())?;
1324
1325            // The indentation is only for readability
1326            write!(self.out, "{}", back::INDENT)?;
1327
1328            match module.types[member.ty].inner {
1329                TypeInner::Array { base, size, .. } => {
1330                    // HLSL arrays are written as `type name[size]`
1331
1332                    self.write_global_type(module, member.ty)?;
1333
1334                    // Write `name`
1335                    write!(
1336                        self.out,
1337                        " {}",
1338                        &self.names[&NameKey::StructMember(handle, index as u32)]
1339                    )?;
1340                    // Write [size]
1341                    self.write_array_size(module, base, size)?;
1342                }
1343                // We treat matrices of the form `matCx2` as a sequence of C `vec2`s.
1344                // See the module-level block comment in mod.rs for details.
1345                TypeInner::Matrix {
1346                    rows,
1347                    columns,
1348                    scalar,
1349                } if member.binding.is_none() && rows == crate::VectorSize::Bi => {
1350                    let vec_ty = TypeInner::Vector { size: rows, scalar };
1351                    let field_name_key = NameKey::StructMember(handle, index as u32);
1352
1353                    for i in 0..columns as u8 {
1354                        if i != 0 {
1355                            write!(self.out, "; ")?;
1356                        }
1357                        self.write_value_type(module, &vec_ty)?;
1358                        write!(self.out, " {}_{}", &self.names[&field_name_key], i)?;
1359                    }
1360                }
1361                _ => {
1362                    // Write modifier before type
1363                    if let Some(ref binding) = member.binding {
1364                        self.write_modifier(binding)?;
1365                    }
1366
1367                    // Even though Naga IR matrices are column-major, we must describe
1368                    // matrices passed from the CPU as being in row-major order.
1369                    // See the module-level block comment in mod.rs for details.
1370                    if let TypeInner::Matrix { .. } = module.types[member.ty].inner {
1371                        write!(self.out, "row_major ")?;
1372                    }
1373
1374                    // Write the member type and name
1375                    self.write_type(module, member.ty)?;
1376                    write!(
1377                        self.out,
1378                        " {}",
1379                        &self.names[&NameKey::StructMember(handle, index as u32)]
1380                    )?;
1381                }
1382            }
1383
1384            self.write_semantic(&member.binding, shader_stage)?;
1385            writeln!(self.out, ";")?;
1386        }
1387
1388        // add padding at the end since sizes of types don't get rounded up to their alignment in HLSL
1389        if members.last().unwrap().binding.is_none() && span > last_offset {
1390            let padding = (span - last_offset) / 4;
1391            for i in 0..padding {
1392                writeln!(self.out, "{}int _end_pad_{};", back::INDENT, i)?;
1393            }
1394        }
1395
1396        writeln!(self.out, "}};")?;
1397        Ok(())
1398    }

1452    /// Helper method used to write value types
1453    ///
1454    /// # Notes
1455    /// Adds no trailing or leading whitespace
1456    pub(super) fn write_value_type(&mut self, module: &Module, inner: &TypeInner) -> BackendResult {
1457        match *inner {
1458            TypeInner::Scalar(scalar) | TypeInner::Atomic(scalar) => {
1459                write!(self.out, "{}", scalar.to_hlsl_str()?)?;
1460            }
1461            TypeInner::Vector { size, scalar } => {
1462                write!(
1463                    self.out,
1464                    "{}{}",
1465                    scalar.to_hlsl_str()?,
1466                    common::vector_size_str(size)
1467                )?;
1468            }
1469            TypeInner::Matrix {
1470                columns,
1471                rows,
1472                scalar,
1473            } => {
1474                // The IR supports only float matrix
1475                // https://docs.microsoft.com/en-us/windows/win32/direct3dhlsl/dx-graphics-hlsl-matrix
1476
1477                // Because of the implicit transpose all matrices have in HLSL, we need to transpose the size as well.
1478                write!(
1479                    self.out,
1480                    "{}{}x{}",
1481                    scalar.to_hlsl_str()?,
1482                    common::vector_size_str(columns),
1483                    common::vector_size_str(rows),
1484                )?;
1485            }
1486            TypeInner::Image {
1487                dim,
1488                arrayed,
1489                class,
1490            } => {
1491                self.write_image_type(dim, arrayed, class)?;
1492            }
1493            TypeInner::Sampler { comparison } => {
1494                let sampler = if comparison {
1495                    "SamplerComparisonState"
1496                } else {
1497                    "SamplerState"
1498                };
1499                write!(self.out, "{sampler}")?;
1500            }
1501            // HLSL arrays are written as `type name[size]`
1502            // Current code is written arrays only as `[size]`
1503            // Base `type` and `name` should be written outside
1504            TypeInner::Array { base, size, .. } | TypeInner::BindingArray { base, size } => {
1505                self.write_array_size(module, base, size)?;
1506            }
1507            TypeInner::AccelerationStructure { .. } => {
1508                write!(self.out, "RaytracingAccelerationStructure")?;
1509            }
1510            TypeInner::RayQuery { .. } => {
1511                // these are constant flags, there are dynamic flags also but constant flags are not supported by naga
1512                write!(self.out, "RayQuery<RAY_FLAG_NONE>")?;
1513            }
1514            _ => return Err(Error::Unimplemented(format!("write_value_type {inner:?}"))),
1515        }
1516
1517        Ok(())
1518    }

1520    /// Helper method used to write functions
1521    /// # Notes
1522    /// Ends in a newline
1523    fn write_function(
1524        &mut self,
1525        module: &Module,
1526        name: &str,
1527        func: &crate::Function,
1528        func_ctx: &back::FunctionCtx<'_>,
1529        info: &valid::FunctionInfo,
1530    ) -> BackendResult {
1531        // Function Declaration Syntax - https://docs.microsoft.com/en-us/windows/win32/direct3dhlsl/dx-graphics-hlsl-function-syntax
1532
1533        self.update_expressions_to_bake(module, func, info);
1534
1535        if let Some(ref result) = func.result {
1536            // Write typedef if return type is an array
1537            let array_return_type = match module.types[result.ty].inner {
1538                TypeInner::Array { base, size, .. } => {
1539                    let array_return_type = self.namer.call(&format!("ret_{name}"));
1540                    write!(self.out, "typedef ")?;
1541                    self.write_type(module, result.ty)?;
1542                    write!(self.out, " {array_return_type}")?;
1543                    self.write_array_size(module, base, size)?;
1544                    writeln!(self.out, ";")?;
1545                    Some(array_return_type)
1546                }
1547                _ => None,
1548            };
1549
1550            // Write modifier
1551            if let Some(
1552                ref binding @ crate::Binding::BuiltIn(crate::BuiltIn::Position { invariant: true }),
1553            ) = result.binding
1554            {
1555                self.write_modifier(binding)?;
1556            }
1557
1558            // Write return type
1559            match func_ctx.ty {
1560                back::FunctionType::Function(_) => {
1561                    if let Some(array_return_type) = array_return_type {
1562                        write!(self.out, "{array_return_type}")?;
1563                    } else {
1564                        self.write_type(module, result.ty)?;
1565                    }
1566                }
1567                back::FunctionType::EntryPoint(index) => {
1568                    if let Some(ref ep_output) =
1569                        self.entry_point_io.get(&(index as usize)).unwrap().output
1570                    {
1571                        write!(self.out, "{}", ep_output.ty_name)?;
1572                    } else {
1573                        self.write_type(module, result.ty)?;
1574                    }
1575                }
1576            }
1577        } else {
1578            write!(self.out, "void")?;
1579        }
1580
1581        // Write function name
1582        write!(self.out, " {name}(")?;
1583
1584        let need_workgroup_variables_initialization =
1585            self.need_workgroup_variables_initialization(func_ctx, module);
1586
1587        // Write function arguments for non entry point functions
1588        match func_ctx.ty {
1589            back::FunctionType::Function(handle) => {
1590                for (index, arg) in func.arguments.iter().enumerate() {
1591                    if index != 0 {
1592                        write!(self.out, ", ")?;
1593                    }
1594
1595                    self.write_function_argument(module, handle, arg, index)?;
1596                }
1597            }
1598            back::FunctionType::EntryPoint(ep_index) => {
1599                if let Some(ref ep_input) =
1600                    self.entry_point_io.get(&(ep_index as usize)).unwrap().input
1601                {
1602                    write!(self.out, "{} {}", ep_input.ty_name, ep_input.arg_name)?;
1603                } else {
1604                    let stage = module.entry_points[ep_index as usize].stage;
1605                    for (index, arg) in func.arguments.iter().enumerate() {
1606                        if index != 0 {
1607                            write!(self.out, ", ")?;
1608                        }
1609                        self.write_type(module, arg.ty)?;
1610
1611                        let argument_name =
1612                            &self.names[&NameKey::EntryPointArgument(ep_index, index as u32)];
1613
1614                        write!(self.out, " {argument_name}")?;
1615                        if let TypeInner::Array { base, size, .. } = module.types[arg.ty].inner {
1616                            self.write_array_size(module, base, size)?;
1617                        }
1618
1619                        self.write_semantic(&arg.binding, Some((stage, Io::Input)))?;
1620                    }
1621                }
1622                if need_workgroup_variables_initialization {
1623                    if self
1624                        .entry_point_io
1625                        .get(&(ep_index as usize))
1626                        .unwrap()
1627                        .input
1628                        .is_some()
1629                        || !func.arguments.is_empty()
1630                    {
1631                        write!(self.out, ", ")?;
1632                    }
1633                    write!(self.out, "uint3 __local_invocation_id : SV_GroupThreadID")?;
1634                }
1635            }
1636        }
1637        // Ends of arguments
1638        write!(self.out, ")")?;
1639
1640        // Write semantic if it present
1641        if let back::FunctionType::EntryPoint(index) = func_ctx.ty {
1642            let stage = module.entry_points[index as usize].stage;
1643            if let Some(crate::FunctionResult { ref binding, .. }) = func.result {
1644                self.write_semantic(binding, Some((stage, Io::Output)))?;
1645            }
1646        }
1647
1648        // Function body start
1649        writeln!(self.out)?;
1650        writeln!(self.out, "{{")?;
1651
1652        if need_workgroup_variables_initialization {
1653            self.write_workgroup_variables_initialization(func_ctx, module)?;
1654        }
1655
1656        if let back::FunctionType::EntryPoint(index) = func_ctx.ty {
1657            self.write_ep_arguments_initialization(module, func, index)?;
1658        }
1659
1660        // Write function local variables
1661        for (handle, local) in func.local_variables.iter() {
1662            // Write indentation (only for readability)
1663            write!(self.out, "{}", back::INDENT)?;
1664
1665            // Write the local name
1666            // The leading space is important
1667            self.write_type(module, local.ty)?;
1668            write!(self.out, " {}", self.names[&func_ctx.name_key(handle)])?;
1669            // Write size for array type
1670            if let TypeInner::Array { base, size, .. } = module.types[local.ty].inner {
1671                self.write_array_size(module, base, size)?;
1672            }
1673
1674            let is_ray_query = match module.types[local.ty].inner {
1675                // from https://microsoft.github.io/DirectX-Specs/d3d/Raytracing.html#tracerayinline-example-1 it seems that ray queries shouldn't be zeroed
1676                TypeInner::RayQuery { .. } => true,
1677                _ => {
1678                    write!(self.out, " = ")?;
1679                    // Write the local initializer if needed
1680                    if let Some(init) = local.init {
1681                        self.write_expr(module, init, func_ctx)?;
1682                    } else {
1683                        // Zero initialize local variables
1684                        self.write_default_init(module, local.ty)?;
1685                    }
1686                    false
1687                }
1688            };
1689            // Finish the local with `;` and add a newline (only for readability)
1690            writeln!(self.out, ";")?;
1691            // If it's a ray query, we also want a tracker variable
1692            if is_ray_query {
1693                write!(self.out, "{}", back::INDENT)?;
1694                self.write_value_type(module, &TypeInner::Scalar(Scalar::U32))?;
1695                writeln!(
1696                    self.out,
1697                    " {RAY_QUERY_TRACKER_VARIABLE_PREFIX}{} = 0;",
1698                    self.names[&func_ctx.name_key(handle)]
1699                )?;
1700            }
1701        }
1702
1703        if !func.local_variables.is_empty() {
1704            writeln!(self.out)?;
1705        }
1706
1707        // Write the function body (statement list)
1708        for sta in func.body.iter() {
1709            // The indentation should always be 1 when writing the function body
1710            self.write_stmt(module, sta, func_ctx, back::Level(1))?;
1711        }
1712
1713        writeln!(self.out, "}}")?;
1714
1715        self.named_expressions.clear();
1716
1717        Ok(())
1718    }

1827    /// Helper method used to write switches
1828    fn write_switch(
1829        &mut self,
1830        module: &Module,
1831        func_ctx: &back::FunctionCtx<'_>,
1832        level: back::Level,
1833        selector: Handle<crate::Expression>,
1834        cases: &[crate::SwitchCase],
1835    ) -> BackendResult {
1836        // Write all cases
1837        let indent_level_1 = level.next();
1838        let indent_level_2 = indent_level_1.next();
1839
1840        // See docs of `back::continue_forward` module.
1841        if let Some(variable) = self.continue_ctx.enter_switch(&mut self.namer) {
1842            writeln!(self.out, "{level}bool {variable} = false;",)?;
1843        };
1844
1845        // Check if there is only one body, by seeing if all except the last case are fall through
1846        // with empty bodies. FXC doesn't handle these switches correctly, so
1847        // we generate a `do {} while(false);` loop instead. There must be a default case, so there
1848        // is no need to check if one of the cases would have matched.
1849        let one_body = cases
1850            .iter()
1851            .rev()
1852            .skip(1)
1853            .all(|case| case.fall_through && case.body.is_empty());
1854        if one_body {
1855            // Start the do-while
1856            writeln!(self.out, "{level}do {{")?;
1857            // Note: Expressions have no side-effects so we don't need to emit selector expression.
1858
1859            // Body
1860            if let Some(case) = cases.last() {
1861                for sta in case.body.iter() {
1862                    self.write_stmt(module, sta, func_ctx, indent_level_1)?;
1863                }
1864            }
1865            // End do-while
1866            writeln!(self.out, "{level}}} while(false);")?;
1867        } else {
1868            // Start the switch
1869            write!(self.out, "{level}")?;
1870            write!(self.out, "switch(")?;
1871            self.write_expr(module, selector, func_ctx)?;
1872            writeln!(self.out, ") {{")?;
1873
1874            for (i, case) in cases.iter().enumerate() {
1875                match case.value {
1876                    crate::SwitchValue::I32(value) => {
1877                        write!(self.out, "{indent_level_1}case {value}:")?
1878                    }
1879                    crate::SwitchValue::U32(value) => {
1880                        write!(self.out, "{indent_level_1}case {value}u:")?
1881                    }
1882                    crate::SwitchValue::Default => write!(self.out, "{indent_level_1}default:")?,
1883                }
1884
1885                // The new block is not only stylistic, it plays a role here:
1886                // We might end up having to write the same case body
1887                // multiple times due to FXC not supporting fallthrough.
1888                // Therefore, some `Expression`s written by `Statement::Emit`
1889                // will end up having the same name (`_expr<handle_index>`).
1890                // So we need to put each case in its own scope.
1891                let write_block_braces = !(case.fall_through && case.body.is_empty());
1892                if write_block_braces {
1893                    writeln!(self.out, " {{")?;
1894                } else {
1895                    writeln!(self.out)?;
1896                }
1897
1898                // Although FXC does support a series of case clauses before
1899                // a block[^yes], it does not support fallthrough from a
1900                // non-empty case block to the next[^no]. If this case has a
1901                // non-empty body with a fallthrough, emulate that by
1902                // duplicating the bodies of all the cases it would fall
1903                // into as extensions of this case's own body. This makes
1904                // the HLSL output potentially quadratic in the size of the
1905                // Naga IR.
1906                //
1907                // [^yes]: ```hlsl
1908                // case 1:
1909                // case 2: do_stuff()
1910                // ```
1911                // [^no]: ```hlsl
1912                // case 1: do_this();
1913                // case 2: do_that();
1914                // ```
1915                if case.fall_through && !case.body.is_empty() {
1916                    let curr_len = i + 1;
1917                    let end_case_idx = curr_len
1918                        + cases
1919                            .iter()
1920                            .skip(curr_len)
1921                            .position(|case| !case.fall_through)
1922                            .unwrap();
1923                    let indent_level_3 = indent_level_2.next();
1924                    for case in &cases[i..=end_case_idx] {
1925                        writeln!(self.out, "{indent_level_2}{{")?;
1926                        let prev_len = self.named_expressions.len();
1927                        for sta in case.body.iter() {
1928                            self.write_stmt(module, sta, func_ctx, indent_level_3)?;
1929                        }
1930                        // Clear all named expressions that were previously inserted by the statements in the block
1931                        self.named_expressions.truncate(prev_len);
1932                        writeln!(self.out, "{indent_level_2}}}")?;
1933                    }
1934
1935                    let last_case = &cases[end_case_idx];
1936                    if last_case.body.last().is_none_or(|s| !s.is_terminator()) {
1937                        writeln!(self.out, "{indent_level_2}break;")?;
1938                    }
1939                } else {
1940                    for sta in case.body.iter() {
1941                        self.write_stmt(module, sta, func_ctx, indent_level_2)?;
1942                    }
1943                    if !case.fall_through && case.body.last().is_none_or(|s| !s.is_terminator()) {
1944                        writeln!(self.out, "{indent_level_2}break;")?;
1945                    }
1946                }
1947
1948                if write_block_braces {
1949                    writeln!(self.out, "{indent_level_1}}}")?;
1950                }
1951            }
1952
1953            writeln!(self.out, "{level}}}")?;
1954        }
1955
1956        // Handle any forwarded continue statements.
1957        use back::continue_forward::ExitControlFlow;
1958        let op = match self.continue_ctx.exit_switch() {
1959            ExitControlFlow::None => None,
1960            ExitControlFlow::Continue { variable } => Some(("continue", variable)),
1961            ExitControlFlow::Break { variable } => Some(("break", variable)),
1962        };
1963        if let Some((control_flow, variable)) = op {
1964            writeln!(self.out, "{level}if ({variable}) {{")?;
1965            writeln!(self.out, "{indent_level_1}{control_flow};")?;
1966            writeln!(self.out, "{level}}}")?;
1967        }
1968
1969        Ok(())
1970    }

2821    pub(super) fn write_literal(&mut self, literal: crate::Literal) -> BackendResult {
2822        match literal {
2823            crate::Literal::F64(value) => write!(self.out, "{value:?}L")?,
2824            crate::Literal::F32(value) => write!(self.out, "{value:?}")?,
2825            crate::Literal::F16(value) => write!(self.out, "{value:?}h")?,
2826            crate::Literal::U32(value) => write!(self.out, "{value}u")?,
2827            // `-2147483648` is parsed by some compilers as unary negation of
2828            // positive 2147483648, which is too large for an int, causing
2829            // issues for some compilers. Neither DXC nor FXC appear to have
2830            // this problem, but this is not specified and could change. We
2831            // therefore use `-2147483647 - 1` as a precaution.
2832            crate::Literal::I32(value) if value == i32::MIN => {
2833                write!(self.out, "int({} - 1)", value + 1)?
2834            }
2835            // HLSL has no suffix for explicit i32 literals, but not using any suffix
2836            // makes the type ambiguous which prevents overload resolution from
2837            // working. So we explicitly use the int() constructor syntax.
2838            crate::Literal::I32(value) => write!(self.out, "int({value})")?,
2839            crate::Literal::U64(value) => write!(self.out, "{value}uL")?,
2840            // I64 version of the minimum I32 value issue described above.
2841            crate::Literal::I64(value) if value == i64::MIN => {
2842                write!(self.out, "({}L - 1L)", value + 1)?;
2843            }
2844            crate::Literal::I64(value) => write!(self.out, "{value}L")?,
2845            crate::Literal::Bool(value) => write!(self.out, "{value}")?,
2846            crate::Literal::AbstractInt(_) | crate::Literal::AbstractFloat(_) => {
2847                return Err(Error::Custom(
2848                    "Abstract types should not appear in IR presented to backends".into(),
2849                ));
2850            }
2851        }
2852        Ok(())
2853    }

2855    fn write_possibly_const_expression<E>(
2856        &mut self,
2857        module: &Module,
2858        expr: Handle<crate::Expression>,
2859        expressions: &crate::Arena<crate::Expression>,
2860        write_expression: E,
2861    ) -> BackendResult
2862    where
2863        E: Fn(&mut Self, Handle<crate::Expression>) -> BackendResult,
2864    {
2865        use crate::Expression;
2866
2867        match expressions[expr] {
2868            Expression::Literal(literal) => {
2869                self.write_literal(literal)?;
2870            }
2871            Expression::Constant(handle) => {
2872                let constant = &module.constants[handle];
2873                if constant.name.is_some() {
2874                    write!(self.out, "{}", self.names[&NameKey::Constant(handle)])?;
2875                } else {
2876                    self.write_const_expression(module, constant.init, &module.global_expressions)?;
2877                }
2878            }
2879            Expression::ZeroValue(ty) => {
2880                self.write_wrapped_zero_value_function_name(module, WrappedZeroValue { ty })?;
2881                write!(self.out, "()")?;
2882            }
2883            Expression::Compose { ty, ref components } => {
2884                match module.types[ty].inner {
2885                    TypeInner::Struct { .. } | TypeInner::Array { .. } => {
2886                        self.write_wrapped_constructor_function_name(
2887                            module,
2888                            WrappedConstructor { ty },
2889                        )?;
2890                    }
2891                    _ => {
2892                        self.write_type(module, ty)?;
2893                    }
2894                };
2895                write!(self.out, "(")?;
2896                for (index, component) in components.iter().enumerate() {
2897                    if index != 0 {
2898                        write!(self.out, ", ")?;
2899                    }
2900                    write_expression(self, *component)?;
2901                }
2902                write!(self.out, ")")?;
2903            }
2904            Expression::Splat { size, value } => {
2905                // hlsl is not supported one value constructor
2906                // if we write, for example, int4(0), dxc returns error:
2907                // error: too few elements in vector initialization (expected 4 elements, have 1)
2908                let number_of_components = match size {
2909                    crate::VectorSize::Bi => "xx",
2910                    crate::VectorSize::Tri => "xxx",
2911                    crate::VectorSize::Quad => "xxxx",
2912                };
2913                write!(self.out, "(")?;
2914                write_expression(self, value)?;
2915                write!(self.out, ").{number_of_components}")?
2916            }
2917            _ => {
2918                return Err(Error::Override);
2919            }
2920        }
2921
2922        Ok(())
2923    }

2925    /// Helper method to write expressions
2926    ///
2927    /// # Notes
2928    /// Doesn't add any newlines or leading/trailing spaces
2929    pub(super) fn write_expr(
2930        &mut self,
2931        module: &Module,
2932        expr: Handle<crate::Expression>,
2933        func_ctx: &back::FunctionCtx<'_>,
2934    ) -> BackendResult {
2935        use crate::Expression;
2936
2937        // Handle the special semantics of vertex_index/instance_index
2938        let ff_input = if self.options.special_constants_binding.is_some() {
2939            func_ctx.is_fixed_function_input(expr, module)
2940        } else {
2941            None
2942        };
2943        let closing_bracket = match ff_input {
2944            Some(crate::BuiltIn::VertexIndex) => {
2945                write!(self.out, "({SPECIAL_CBUF_VAR}.{SPECIAL_FIRST_VERTEX} + ")?;
2946                ")"
2947            }
2948            Some(crate::BuiltIn::InstanceIndex) => {
2949                write!(self.out, "({SPECIAL_CBUF_VAR}.{SPECIAL_FIRST_INSTANCE} + ",)?;
2950                ")"
2951            }
2952            Some(crate::BuiltIn::NumWorkGroups) => {
2953                // Note: despite their names (`FIRST_VERTEX` and `FIRST_INSTANCE`),
2954                // in compute shaders the special constants contain the number
2955                // of workgroups, which we are using here.
2956                write!(
2957                    self.out,
2958                    "uint3({SPECIAL_CBUF_VAR}.{SPECIAL_FIRST_VERTEX}, {SPECIAL_CBUF_VAR}.{SPECIAL_FIRST_INSTANCE}, {SPECIAL_CBUF_VAR}.{SPECIAL_OTHER})",
2959                )?;
2960                return Ok(());
2961            }
2962            _ => "",
2963        };
2964
2965        if let Some(name) = self.named_expressions.get(&expr) {
2966            write!(self.out, "{name}{closing_bracket}")?;
2967            return Ok(());
2968        }
2969
2970        let expression = &func_ctx.expressions[expr];
2971
2972        match *expression {
2973            Expression::Literal(_)
2974            | Expression::Constant(_)
2975            | Expression::ZeroValue(_)
2976            | Expression::Compose { .. }
2977            | Expression::Splat { .. } => {
2978                self.write_possibly_const_expression(
2979                    module,
2980                    expr,
2981                    func_ctx.expressions,
2982                    |writer, expr| writer.write_expr(module, expr, func_ctx),
2983                )?;
2984            }
2985            Expression::Override(_) => return Err(Error::Override),
2986            // Avoid undefined behaviour for addition, subtraction, and
2987            // multiplication of signed integers by casting operands to
2988            // unsigned, performing the operation, then casting the result back
2989            // to signed.
2990            // TODO(#7109): This relies on the asint()/asuint() functions which only work
2991            // for 32-bit types, so we must find another solution for different bit widths.
2992            Expression::Binary {
2993                op:
2994                    op @ crate::BinaryOperator::Add
2995                    | op @ crate::BinaryOperator::Subtract
2996                    | op @ crate::BinaryOperator::Multiply,
2997                left,
2998                right,
2999            } if matches!(
3000                func_ctx.resolve_type(expr, &module.types).scalar(),
3001                Some(Scalar::I32)
3002            ) =>
3003            {
3004                write!(self.out, "asint(asuint(",)?;
3005                self.write_expr(module, left, func_ctx)?;
3006                write!(self.out, ") {} asuint(", back::binary_operation_str(op))?;
3007                self.write_expr(module, right, func_ctx)?;
3008                write!(self.out, "))")?;
3009            }
3010            // All of the multiplication can be expressed as `mul`,
3011            // except vector * vector, which needs to use the "*" operator.
3012            Expression::Binary {
3013                op: crate::BinaryOperator::Multiply,
3014                left,
3015                right,
3016            } if func_ctx.resolve_type(left, &module.types).is_matrix()
3017                || func_ctx.resolve_type(right, &module.types).is_matrix() =>
3018            {
3019                // We intentionally flip the order of multiplication as our matrices are implicitly transposed.
3020                write!(self.out, "mul(")?;
3021                self.write_expr(module, right, func_ctx)?;
3022                write!(self.out, ", ")?;
3023                self.write_expr(module, left, func_ctx)?;
3024                write!(self.out, ")")?;
3025            }
3026
3027            // WGSL says that floating-point division by zero should return
3028            // infinity. Microsoft's Direct3D 11 functional specification
3029            // (https://microsoft.github.io/DirectX-Specs/d3d/archive/D3D11_3_FunctionalSpec.htm)
3030            // says:
3031            //
3032            //     Divide by 0 produces +/- INF, except 0/0 which results in NaN.
3033            //
3034            // which is what we want. The DXIL specification for the FDiv
3035            // instruction corroborates this:
3036            //
3037            // https://github.com/microsoft/DirectXShaderCompiler/blob/main/docs/DXIL.rst#fdiv
3038            Expression::Binary {
3039                op: crate::BinaryOperator::Divide,
3040                left,
3041                right,
3042            } if matches!(
3043                func_ctx.resolve_type(expr, &module.types).scalar_kind(),
3044                Some(ScalarKind::Sint | ScalarKind::Uint)
3045            ) =>
3046            {
3047                write!(self.out, "{DIV_FUNCTION}(")?;
3048                self.write_expr(module, left, func_ctx)?;
3049                write!(self.out, ", ")?;
3050                self.write_expr(module, right, func_ctx)?;
3051                write!(self.out, ")")?;
3052            }
3053
3054            Expression::Binary {
3055                op: crate::BinaryOperator::Modulo,
3056                left,
3057                right,
3058            } if matches!(
3059                func_ctx.resolve_type(expr, &module.types).scalar_kind(),
3060                Some(ScalarKind::Sint | ScalarKind::Uint | ScalarKind::Float)
3061            ) =>
3062            {
3063                write!(self.out, "{MOD_FUNCTION}(")?;
3064                self.write_expr(module, left, func_ctx)?;
3065                write!(self.out, ", ")?;
3066                self.write_expr(module, right, func_ctx)?;
3067                write!(self.out, ")")?;
3068            }
3069
3070            Expression::Binary { op, left, right } => {
3071                write!(self.out, "(")?;
3072                self.write_expr(module, left, func_ctx)?;
3073                write!(self.out, " {} ", back::binary_operation_str(op))?;
3074                self.write_expr(module, right, func_ctx)?;
3075                write!(self.out, ")")?;
3076            }
3077            Expression::Access { base, index } => {
3078                if let Some(crate::AddressSpace::Storage { .. }) =
3079                    func_ctx.resolve_type(expr, &module.types).pointer_space()
3080                {
3081                    // do nothing, the chain is written on `Load`/`Store`
3082                } else {
3083                    // We use the function __get_col_of_matCx2 here in cases
3084                    // where `base`s type resolves to a matCx2 and is part of a
3085                    // struct member with type of (possibly nested) array of matCx2's.
3086                    //
3087                    // Note that this only works for `Load`s and we handle
3088                    // `Store`s differently in `Statement::Store`.
3089                    if let Some(MatrixType {
3090                        columns,
3091                        rows: crate::VectorSize::Bi,
3092                        width: 4,
3093                    }) = get_inner_matrix_of_struct_array_member(module, base, func_ctx, true)
3094                        .or_else(|| get_global_uniform_matrix(module, base, func_ctx))
3095                    {
3096                        write!(self.out, "__get_col_of_mat{}x2(", columns as u8)?;
3097                        self.write_expr(module, base, func_ctx)?;
3098                        write!(self.out, ", ")?;
3099                        self.write_expr(module, index, func_ctx)?;
3100                        write!(self.out, ")")?;
3101                        return Ok(());
3102                    }
3103
3104                    let resolved = func_ctx.resolve_type(base, &module.types);
3105
