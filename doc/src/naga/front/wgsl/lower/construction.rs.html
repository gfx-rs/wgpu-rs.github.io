10/// A [`constructor built-in function`].
11///
12/// WGSL has two types of such functions:
13///
14/// - Those that fully specify the type being constructed, like
15///   `vec3<f32>(x,y,z)`, which obviously constructs a `vec3<f32>`.
16///
17/// - Those that leave the component type of the composite being constructed
18///   implicit, to be inferred from the argument types, like `vec3(x,y,z)`,
19///   which constructs a `vec3<T>` where `T` is the type of `x`, `y`, and `z`.
20///
21/// This enum represents both cases. The `PartialFoo` variants
22/// represent the second case, where the component type is implicit.
23///
24/// [`constructor built-in function`]: https://gpuweb.github.io/gpuweb/wgsl/#constructor-builtin-function
25pub enum Constructor<T> {
26    /// A vector construction whose component type is inferred from the
27    /// argument: `vec3(1.0)`.
28    PartialVector { size: crate::VectorSize },
29
30    /// A matrix construction whose component type is inferred from the
31    /// argument: `mat2x2(1,2,3,4)`.
32    PartialMatrix {
33        columns: crate::VectorSize,
34        rows: crate::VectorSize,
35    },
36
37    /// An array whose component type and size are inferred from the arguments:
38    /// `array(3,4,5)`.
39    PartialArray,
40
41    /// A known Naga type.
42    ///
43    /// When we match on this type, we need to see the `TypeInner` here, but at
44    /// the point that we build this value we'll still need mutable access to
45    /// the module later. To avoid borrowing from the module, the type parameter
46    /// `T` is `Handle<Type>` initially. Then we use `borrow_inner` to produce a
47    /// version holding a tuple `(Handle<Type>, &TypeInner)`.
48    Type(T),
49}
50
51impl Constructor<Handle<crate::Type>> {
52    /// Return an equivalent `Constructor` value that includes borrowed
53    /// `TypeInner` values alongside any type handles.
54    ///
55    /// The returned form is more convenient to match on, since the patterns
56    /// can actually see what the handle refers to.
57    fn borrow_inner(
58        self,
59        module: &crate::Module,
60    ) -> Constructor<(Handle<crate::Type>, &crate::TypeInner)> {
61        match self {
62            Constructor::PartialVector { size } => Constructor::PartialVector { size },
63            Constructor::PartialMatrix { columns, rows } => {
64                Constructor::PartialMatrix { columns, rows }
65            }
66            Constructor::PartialArray => Constructor::PartialArray,
67            Constructor::Type(handle) => Constructor::Type((handle, &module.types[handle].inner)),
68        }
69    }
70}

72enum Components<'a> {
73    None,
74    One {
75        component: Handle<crate::Expression>,
76        span: Span,
77        ty_inner: &'a crate::TypeInner,
78    },
79    Many {
80        components: Vec<Handle<crate::Expression>>,
81        spans: Vec<Span>,
82    },
83}
84
85impl Components<'_> {
86    fn into_components_vec(self) -> Vec<Handle<crate::Expression>> {
87        match self {
88            Self::None => vec![],
89            Self::One { component, .. } => vec![component],
90            Self::Many { components, .. } => components,
91        }
92    }
93}

95impl<'source> Lowerer<'source, '_> {
96    /// Generate Naga IR for a type constructor expression.
97    ///
98    /// The `constructor` value represents the head of the constructor
99    /// expression, which is at least a hint of which type is being built; if
100    /// it's one of the `Partial` variants, we need to consider the argument
101    /// types as well.
102    ///
103    /// This is used for [`Call`] expressions, once we've determined that
104    /// the "callable" (in WGSL spec terms) is actually a type.
105    ///
106    /// [`Call`]: ast::Expression::Call
107    pub fn construct(
108        &mut self,
109        span: Span,
110        constructor: Constructor<Handle<crate::Type>>,
111        ty_span: Span,
112        components: &[Handle<ast::Expression<'source>>],
113        ctx: &mut ExpressionContext<'source, '_, '_>,
114    ) -> Result<'source, Handle<crate::Expression>> {
115        use crate::proc::TypeResolution as Tr;
116
117        let components = match *components {
118            [] => Components::None,
119            [component] => {
120                let span = ctx.ast_expressions.get_span(component);
121                let component = self.expression_for_abstract(component, ctx)?;
122                let ty_inner = super::resolve_inner!(ctx, component);
123
124                Components::One {
125                    component,
126                    span,
127                    ty_inner,
128                }
129            }
130            ref ast_components @ [_, _, ..] => {
131                let components = ast_components
132                    .iter()
133                    .map(|&expr| self.expression_for_abstract(expr, ctx))
134                    .collect::<Result<_>>()?;
135                let spans = ast_components
136                    .iter()
137                    .map(|&expr| ctx.ast_expressions.get_span(expr))
138                    .collect();
139
140                for &component in &components {
141                    ctx.grow_types(component)?;
142                }
143
144                Components::Many { components, spans }
145            }
146        };
147
148        // Even though we computed `constructor` above, wait until now to borrow
149        // a reference to the `TypeInner`, so that the component-handling code
150        // above can have mutable access to the type arena.
151        let constructor = constructor.borrow_inner(ctx.module);
152
153        let expr;
154        match (components, constructor) {
155            // Empty constructor
156            (Components::None, dst_ty) => match dst_ty {
157                Constructor::Type((result_ty, _)) => {
158                    expr = crate::Expression::ZeroValue(result_ty);
159                }
160                Constructor::PartialVector { size } => {
161                    // vec2(), vec3(), vec4() return vectors of abstractInts; the same
162                    // is not true of the similar constructors for matrices or arrays.
163                    // See https://www.w3.org/TR/WGSL/#vec2-builtin et seq.
164                    let result_ty = ctx.module.types.insert(
165                        crate::Type {
166                            name: None,
167                            inner: crate::TypeInner::Vector {
168                                size,
169                                scalar: crate::Scalar::ABSTRACT_INT,
170                            },
171                        },
172                        span,
173                    );
174                    expr = crate::Expression::ZeroValue(result_ty);
175                }
176                Constructor::PartialMatrix { .. } | Constructor::PartialArray => {
177                    // We have no arguments from which to infer the result type, so
178                    // partial constructors aren't acceptable here.
179                    return Err(Box::new(Error::TypeNotInferable(ty_span)));
180                }
181            },
182
183            // Scalar constructor & conversion (scalar -> scalar)
184            (
185                Components::One {
186                    component,
187                    ty_inner: &crate::TypeInner::Scalar { .. },
188                    ..
189                },
190                Constructor::Type((_, &crate::TypeInner::Scalar(scalar))),
191            ) => {
192                expr = crate::Expression::As {
193                    expr: component,
194                    kind: scalar.kind,
195                    convert: Some(scalar.width),
196                };
197            }
198
199            // Vector conversion (vector -> vector)
200            (
201                Components::One {
202                    component,
203                    ty_inner: &crate::TypeInner::Vector { size: src_size, .. },
204                    ..
205                },
206                Constructor::Type((
207                    _,
208                    &crate::TypeInner::Vector {
209                        size: dst_size,
210                        scalar: dst_scalar,
211                    },
212                )),
213            ) if dst_size == src_size => {
214                expr = crate::Expression::As {
215                    expr: component,
216                    kind: dst_scalar.kind,
217                    convert: Some(dst_scalar.width),
218                };
219            }
220
221            // Vector conversion (vector -> vector) - partial
222            (
223                Components::One {
224                    component,
225                    ty_inner: &crate::TypeInner::Vector { size: src_size, .. },
226                    ..
227                },
228                Constructor::PartialVector { size: dst_size },
229            ) if dst_size == src_size => {
230                // This is a trivial conversion: the sizes match, and a Partial
231                // constructor doesn't specify a scalar type, so nothing can
232                // possibly happen.
233                return Ok(component);
234            }
235
236            // Matrix conversion (matrix -> matrix)
237            (
238                Components::One {
239                    component,
240                    ty_inner:
241                        &crate::TypeInner::Matrix {
242                            columns: src_columns,
243                            rows: src_rows,
244                            ..
245                        },
246                    ..
247                },
248                Constructor::Type((
249                    _,
250                    &crate::TypeInner::Matrix {
251                        columns: dst_columns,
252                        rows: dst_rows,
253                        scalar: dst_scalar,
254                    },
255                )),
256            ) if dst_columns == src_columns && dst_rows == src_rows => {
257                expr = crate::Expression::As {
258                    expr: component,
259                    kind: dst_scalar.kind,
260                    convert: Some(dst_scalar.width),
261                };
262            }
263
264            // Matrix conversion (matrix -> matrix) - partial
265            (
266                Components::One {
267                    component,
268                    ty_inner:
269                        &crate::TypeInner::Matrix {
270                            columns: src_columns,
271                            rows: src_rows,
272                            ..
273                        },
274                    ..
275                },
276                Constructor::PartialMatrix {
277                    columns: dst_columns,
278                    rows: dst_rows,
279                },
280            ) if dst_columns == src_columns && dst_rows == src_rows => {
281                // This is a trivial conversion: the sizes match, and a Partial
282                // constructor doesn't specify a scalar type, so nothing can
283                // possibly happen.
284                return Ok(component);
285            }
286
287            // Vector constructor (splat) - infer type
288            (
289                Components::One {
290                    component,
291                    ty_inner: &crate::TypeInner::Scalar { .. },
292                    ..
293                },
294                Constructor::PartialVector { size },
295            ) => {
296                expr = crate::Expression::Splat {
297                    size,
298                    value: component,
299                };
300            }
301
302            // Vector constructor (splat)
303            (
304                Components::One {
305                    mut component,
306                    ty_inner: &crate::TypeInner::Scalar(component_scalar),
307                    span,
308                },
309                Constructor::Type((
310                    type_handle,
311                    &crate::TypeInner::Vector {
312                        size,
313                        scalar: vec_scalar,
314                    },
315                )),
316            ) => {
317                // Splat only allows automatic conversions of the component's scalar.
318                if !component_scalar.automatically_converts_to(vec_scalar) {
319                    let component_ty = &ctx.typifier()[component];
320                    let arg_ty = ctx.type_resolution_to_string(component_ty);
321                    return Err(Box::new(Error::WrongArgumentType {
322                        function: ctx.type_to_string(type_handle),
323                        call_span: ty_span,
324                        arg_span: span,
325                        arg_index: 0,
326                        arg_ty,
327                        allowed: vec![vec_scalar.to_wgsl_for_diagnostics()],
328                    }));
329                }
330                ctx.convert_slice_to_common_leaf_scalar(
331                    core::slice::from_mut(&mut component),
332                    vec_scalar,
333                )?;
334                expr = crate::Expression::Splat {
335                    size,
336                    value: component,
337                };
338            }
339
340            // Vector constructor (by elements), partial
341            (
342                Components::Many {
343                    mut components,
344                    spans,
345                },
346                Constructor::PartialVector { size },
347            ) => {
348                let consensus_scalar =
349                    ctx.automatic_conversion_consensus(&components)
350                        .map_err(|index| {
351                            Error::InvalidConstructorComponentType(spans[index], index as i32)
352                        })?;
353                ctx.convert_slice_to_common_leaf_scalar(&mut components, consensus_scalar)?;
354                let inner = consensus_scalar.to_inner_vector(size);
355                let ty = ctx.ensure_type_exists(inner);
356                expr = crate::Expression::Compose { ty, components };
357            }
358
359            // Vector constructor (by elements), full type given
360            (
361                Components::Many { mut components, .. },
362                Constructor::Type((ty, &crate::TypeInner::Vector { scalar, .. })),
363            ) => {
364                ctx.try_automatic_conversions_for_vector(&mut components, scalar, ty_span)?;
365                expr = crate::Expression::Compose { ty, components };
366            }
367
368            // Matrix constructor (by elements), partial
369            (
370                Components::Many {
371                    mut components,
372                    spans,
373                },
374                Constructor::PartialMatrix { columns, rows },
375            ) if components.len() == columns as usize * rows as usize => {
376                let consensus_scalar =
377                    ctx.automatic_conversion_consensus(&components)
378                        .map_err(|index| {
379                            Error::InvalidConstructorComponentType(spans[index], index as i32)
380                        })?;
381                // We actually only accept floating-point elements.
382                let consensus_scalar = consensus_scalar
383                    .automatic_conversion_combine(crate::Scalar::ABSTRACT_FLOAT)
384                    .unwrap_or(consensus_scalar);
385                ctx.convert_slice_to_common_leaf_scalar(&mut components, consensus_scalar)?;
386                let vec_ty = ctx.ensure_type_exists(consensus_scalar.to_inner_vector(rows));
387
388                let components = components
389                    .chunks(rows as usize)
390                    .map(|vec_components| {
391                        ctx.append_expression(
392                            crate::Expression::Compose {
393                                ty: vec_ty,
394                                components: Vec::from(vec_components),
395                            },
396                            Default::default(),
397                        )
398                    })
399                    .collect::<Result<Vec<_>>>()?;
400
401                let ty = ctx.ensure_type_exists(crate::TypeInner::Matrix {
402                    columns,
403                    rows,
404                    scalar: consensus_scalar,
405                });
406                expr = crate::Expression::Compose { ty, components };
407            }
408
409            // Matrix constructor (by elements), type given
410            (
411                Components::Many { mut components, .. },
412                Constructor::Type((
413                    _,
414                    &crate::TypeInner::Matrix {
415                        columns,
416                        rows,
417                        scalar,
418                    },
419                )),
420            ) if components.len() == columns as usize * rows as usize => {
421                let element = Tr::Value(crate::TypeInner::Scalar(scalar));
422                ctx.try_automatic_conversions_slice(&mut components, &element, ty_span)?;
423                let vec_ty = ctx.ensure_type_exists(scalar.to_inner_vector(rows));
424
425                let components = components
426                    .chunks(rows as usize)
427                    .map(|vec_components| {
428                        ctx.append_expression(
429                            crate::Expression::Compose {
430                                ty: vec_ty,
431                                components: Vec::from(vec_components),
432                            },
433                            Default::default(),
434                        )
435                    })
436                    .collect::<Result<Vec<_>>>()?;
437
438                let ty = ctx.ensure_type_exists(crate::TypeInner::Matrix {
439                    columns,
440                    rows,
441                    scalar,
442                });
443                expr = crate::Expression::Compose { ty, components };
444            }
445
446            // Matrix constructor (by columns), partial
447            (
448                Components::Many {
449                    mut components,
450                    spans,
451                },
452                Constructor::PartialMatrix { columns, rows },
453            ) => {
454                let consensus_scalar =
455                    ctx.automatic_conversion_consensus(&components)
456                        .map_err(|index| {
457                            Error::InvalidConstructorComponentType(spans[index], index as i32)
458                        })?;
459                ctx.convert_slice_to_common_leaf_scalar(&mut components, consensus_scalar)?;
460                let ty = ctx.ensure_type_exists(crate::TypeInner::Matrix {
461                    columns,
462                    rows,
463                    scalar: consensus_scalar,
464                });
465                expr = crate::Expression::Compose { ty, components };
466            }
467
468            // Matrix constructor (by columns), type given
469            (
470                Components::Many { mut components, .. },
471                Constructor::Type((
472                    ty,
473                    &crate::TypeInner::Matrix {
474                        columns: _,
475                        rows,
476                        scalar,
477                    },
478                )),
479            ) => {
480                let component_ty = crate::TypeInner::Vector { size: rows, scalar };
481                ctx.try_automatic_conversions_slice(
482                    &mut components,
483                    &Tr::Value(component_ty),
484                    ty_span,
485                )?;
486                expr = crate::Expression::Compose { ty, components };
487            }
488
489            // Array constructor - infer type
490            (components, Constructor::PartialArray) => {
491                let mut components = components.into_components_vec();
492                if let Ok(consensus_scalar) = ctx.automatic_conversion_consensus(&components) {
493                    // Note that this will *not* necessarily convert all the
494                    // components to the same type! The `automatic_conversion_consensus`
495                    // method only considers the parameters' leaf scalar
496                    // types; the parameters themselves could be any mix of
497                    // vectors, matrices, and scalars.
498                    //
499                    // But *if* it is possible for this array construction
500                    // expression to be well-typed at all, then all the
501                    // parameters must have the same type constructors (vec,
502                    // matrix, scalar) applied to their leaf scalars, so
503                    // reconciling their scalars is always the right thing to
504                    // do. And if this array construction is not well-typed,
505                    // these conversions will not make it so, and we can let
506                    // validation catch the error.
507                    ctx.convert_slice_to_common_leaf_scalar(&mut components, consensus_scalar)?;
508                } else {
509                    // There's no consensus scalar. Emit the `Compose`
510                    // expression anyway, and let validation catch the problem.
511                }
512
513                let base = ctx.register_type(components[0])?;
514
515                let inner = crate::TypeInner::Array {
516                    base,
517                    size: crate::ArraySize::Constant(
518                        NonZeroU32::new(u32::try_from(components.len()).unwrap()).unwrap(),
519                    ),
520                    stride: {
521                        ctx.layouter.update(ctx.module.to_ctx()).unwrap();
522                        ctx.layouter[base].to_stride()
523                    },
524                };
525                let ty = ctx.ensure_type_exists(inner);
526
527                expr = crate::Expression::Compose { ty, components };
528            }
529
530            // Array constructor, explicit type
531            (components, Constructor::Type((ty, &crate::TypeInner::Array { base, .. }))) => {
532                let mut components = components.into_components_vec();
533                ctx.try_automatic_conversions_slice(&mut components, &Tr::Handle(base), ty_span)?;
534                expr = crate::Expression::Compose { ty, components };
535            }
536
537            // Struct constructor
538            (
539                components,
540                Constructor::Type((ty, &crate::TypeInner::Struct { ref members, .. })),
541            ) => {
542                let mut components = components.into_components_vec();
543                let struct_ty_span = ctx.module.types.get_span(ty);
544
545                // Make a vector of the members' type handles in advance, to
546                // avoid borrowing `members` from `ctx` while we generate
547                // new code.
548                let members: Vec<Handle<crate::Type>> = members.iter().map(|m| m.ty).collect();
549
550                for (component, &ty) in components.iter_mut().zip(&members) {
551                    *component =
552                        ctx.try_automatic_conversions(*component, &Tr::Handle(ty), struct_ty_span)?;
553                }
554                expr = crate::Expression::Compose { ty, components };
555            }
556
557            // ERRORS
558
559            // Bad conversion (type cast)
560            (
561                Components::One {
562                    span, component, ..
563                },
564                Constructor::Type((
565                    ty,
566                    &(crate::TypeInner::Scalar { .. }
567                    | crate::TypeInner::Vector { .. }
568                    | crate::TypeInner::Matrix { .. }),
569                )),
570            ) => {
571                let component_ty = &ctx.typifier()[component];
572                let from_type = ctx.type_resolution_to_string(component_ty);
573                return Err(Box::new(Error::BadTypeCast {
574                    span,
575                    from_type,
576                    to_type: ctx.type_to_string(ty),
577                }));
578            }
579
580            // Too many parameters for scalar constructor
581            (
582                Components::Many { spans, .. },
583                Constructor::Type((_, &crate::TypeInner::Scalar { .. })),
584            ) => {
585                let span = spans[1].until(spans.last().unwrap());
586                return Err(Box::new(Error::UnexpectedComponents(span)));
587            }
588
589            // Other types can't be constructed
590            _ => return Err(Box::new(Error::TypeNotConstructible(ty_span))),
591        }
592
593        let expr = ctx.append_expression(expr, span)?;
594        Ok(expr)
595    }
