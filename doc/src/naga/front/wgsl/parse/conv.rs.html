9pub fn map_address_space<'a>(
10    word: &str,
11    span: Span,
12    enable_extensions: &EnableExtensions,
13) -> Result<'a, crate::AddressSpace> {
14    match word {
15        "private" => Ok(crate::AddressSpace::Private),
16        "workgroup" => Ok(crate::AddressSpace::WorkGroup),
17        "uniform" => Ok(crate::AddressSpace::Uniform),
18        "storage" => Ok(crate::AddressSpace::Storage {
19            access: crate::StorageAccess::default(),
20        }),
21        "immediate" => Ok(crate::AddressSpace::Immediate),
22        "function" => Ok(crate::AddressSpace::Function),
23        "task_payload" => {
24            if enable_extensions.contains(ImplementedEnableExtension::WgpuMeshShader) {
25                Ok(crate::AddressSpace::TaskPayload)
26            } else {
27                Err(Box::new(Error::EnableExtensionNotEnabled {
28                    span,
29                    kind: ImplementedEnableExtension::WgpuMeshShader.into(),
30                }))
31            }
32        }
33        _ => Err(Box::new(Error::UnknownAddressSpace(span))),
34    }
35}

111        "line_indices" => crate::BuiltIn::LineIndices,
112        "triangle_indices" => crate::BuiltIn::TriangleIndices,
113        "mesh_task_size" => crate::BuiltIn::MeshTaskSize,

124                return Err(Box::new(Error::EnableExtensionNotEnabled {
125                    span,
126                    kind: ImplementedEnableExtension::ClipDistances.into(),

137        | crate::BuiltIn::Primitives => {
138            if !enable_extensions.contains(ImplementedEnableExtension::WgpuMeshShader) {
139                return Err(Box::new(Error::EnableExtensionNotEnabled {
140                    span,
141                    kind: ImplementedEnableExtension::WgpuMeshShader.into(),
142                }));
143            }
144        }
145        _ => {}
146    }
147    Ok(built_in)
148}

502        "array" => TypeGenerator::Array.into(),
503        // atomic
504        "atomic" => TypeGenerator::Atomic.into(),
505        // pointer
506        "ptr" => TypeGenerator::Pointer.into(),
507        // sampled textures
508        "texture_1d" =>               TypeGenerator::SampledTexture { dim: ImageDimension::D1,   arrayed: false, multi: false }.into(),
509        "texture_2d" =>               TypeGenerator::SampledTexture { dim: ImageDimension::D2,   arrayed: false, multi: false }.into(),
510        "texture_2d_array" =>         TypeGenerator::SampledTexture { dim: ImageDimension::D2,   arrayed: true,  multi: false }.into(),
511        "texture_3d" =>               TypeGenerator::SampledTexture { dim: ImageDimension::D3,   arrayed: false, multi: false }.into(),
512        "texture_cube" =>             TypeGenerator::SampledTexture { dim: ImageDimension::Cube, arrayed: false, multi: false }.into(),
513        "texture_cube_array" =>       TypeGenerator::SampledTexture { dim: ImageDimension::Cube, arrayed: true,  multi: false }.into(),
514        "texture_multisampled_2d" =>  TypeGenerator::SampledTexture { dim: ImageDimension::D2,   arrayed: false, multi: true  }.into(),
515        // storage textures
516        "texture_storage_1d" =>       TypeGenerator::StorageTexture { dim: ImageDimension::D1,   arrayed: false }.into(),
517        "texture_storage_2d" =>       TypeGenerator::StorageTexture { dim: ImageDimension::D2,   arrayed: false }.into(),
518        "texture_storage_2d_array" => TypeGenerator::StorageTexture { dim: ImageDimension::D2,   arrayed: true  }.into(),
519        "texture_storage_3d" =>       TypeGenerator::StorageTexture { dim: ImageDimension::D3,   arrayed: false }.into(),
520        // binding array
521        "binding_array" => TypeGenerator::BindingArray.into(),
522        // acceleration structure
523        "acceleration_structure" => TypeGenerator::AccelerationStructure.into(),
524        // ray query
525        "ray_query" => TypeGenerator::RayQuery.into(),
526        // cooperative matrix
527        "coop_mat8x8" => TypeGenerator::CooperativeMatrix {
528            columns: crate::CooperativeSize::Eight,
529            rows: crate::CooperativeSize::Eight,
530        }.into(),
531        "coop_mat16x16" => TypeGenerator::CooperativeMatrix {
532            columns: crate::CooperativeSize::Sixteen,
533            rows: crate::CooperativeSize::Sixteen,
534        }.into(),
535        _ => return Ok(None),
536    };
537
538    // Check for the enable extension required to use this type, if any.
539    let extension_needed = match ty {
540        PredeclaredType::TypeInner(ref ty) if ty.scalar() == Some(Sc::F16) => {
541            Some(ImplementedEnableExtension::F16)
542        }
543        PredeclaredType::RayDesc
544        | PredeclaredType::RayIntersection
545        | PredeclaredType::TypeGenerator(TypeGenerator::AccelerationStructure)
546        | PredeclaredType::TypeGenerator(TypeGenerator::RayQuery) => {
547            Some(ImplementedEnableExtension::WgpuRayQuery)
548        }
549        PredeclaredType::TypeGenerator(TypeGenerator::CooperativeMatrix { .. }) => {
550            Some(ImplementedEnableExtension::WgpuCooperativeMatrix)
551        }
552        _ => None,
553    };
554    if let Some(extension_needed) = extension_needed {
555        if !enable_extensions.contains(extension_needed) {
556            return Err(Box::new(Error::EnableExtensionNotEnabled {
557                span,
558                kind: extension_needed.into(),
559            }));
560        }
561    }
562
563    Ok(Some(ty))
564}
