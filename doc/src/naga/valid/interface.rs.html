128fn storage_usage(access: crate::StorageAccess) -> GlobalUse {
129    let mut storage_usage = GlobalUse::QUERY;
130    if access.contains(crate::StorageAccess::LOAD) {
131        storage_usage |= GlobalUse::READ;
132    }
133    if access.contains(crate::StorageAccess::STORE) {
134        storage_usage |= GlobalUse::WRITE;
135    }
136    if access.contains(crate::StorageAccess::ATOMIC) {
137        storage_usage |= GlobalUse::ATOMIC;
138    }
139    storage_usage
140}

142struct VaryingContext<'a> {
143    stage: crate::ShaderStage,
144    output: bool,
145    types: &'a UniqueArena<crate::Type>,
146    type_info: &'a Vec<super::r#type::TypeInfo>,
147    location_mask: &'a mut BitSet,
148    blend_src_mask: &'a mut BitSet,
149    built_ins: &'a mut crate::FastHashSet<crate::BuiltIn>,
150    capabilities: Capabilities,
151    flags: super::ValidationFlags,
152}

154impl VaryingContext<'_> {
155    fn validate_impl(
156        &mut self,
157        ep: &crate::EntryPoint,
158        ty: Handle<crate::Type>,
159        binding: &crate::Binding,
160    ) -> Result<(), VaryingError> {
161        use crate::{BuiltIn as Bi, ShaderStage as St, TypeInner as Ti, VectorSize as Vs};
162
163        let ty_inner = &self.types[ty].inner;
164        match *binding {
165            crate::Binding::BuiltIn(built_in) => {
166                // Ignore the `invariant` field for the sake of duplicate checks,
167                // but use the original in error messages.
168                let canonical = if let crate::BuiltIn::Position { .. } = built_in {
169                    crate::BuiltIn::Position { invariant: false }
170                } else {
171                    built_in
172                };
173
174                if self.built_ins.contains(&canonical) {
175                    return Err(VaryingError::DuplicateBuiltIn(built_in));
176                }
177                self.built_ins.insert(canonical);
178
179                let required = match built_in {
180                    Bi::ClipDistance => Capabilities::CLIP_DISTANCE,
181                    Bi::CullDistance => Capabilities::CULL_DISTANCE,
182                    Bi::PrimitiveIndex => Capabilities::PRIMITIVE_INDEX,
183                    Bi::Barycentric => Capabilities::SHADER_BARYCENTRICS,
184                    Bi::ViewIndex => Capabilities::MULTIVIEW,
185                    Bi::SampleIndex => Capabilities::MULTISAMPLED_SHADING,
186                    Bi::NumSubgroups
187                    | Bi::SubgroupId
188                    | Bi::SubgroupSize
189                    | Bi::SubgroupInvocationId => Capabilities::SUBGROUP,
190                    _ => Capabilities::empty(),
191                };
192                if !self.capabilities.contains(required) {
193                    return Err(VaryingError::UnsupportedCapability(required));
194                }
195
196                if matches!(
197                    built_in,
198                    crate::BuiltIn::SubgroupId | crate::BuiltIn::SubgroupInvocationId
199                ) && ep.workgroup_size[1..].iter().any(|&s| s > 1)
200                {
201                    return Err(VaryingError::InvalidMultiDimensionalSubgroupBuiltIn);
202                }
203
204                let (visible, type_good) = match built_in {
205                    Bi::BaseInstance
206                    | Bi::BaseVertex
207                    | Bi::InstanceIndex
208                    | Bi::VertexIndex
209                    | Bi::DrawID => (
210                        self.stage == St::Vertex && !self.output,
211                        *ty_inner == Ti::Scalar(crate::Scalar::U32),
212                    ),
213                    Bi::ClipDistance | Bi::CullDistance => (
214                        self.stage == St::Vertex && self.output,
215                        match *ty_inner {
216                            Ti::Array { base, size, .. } => {
217                                self.types[base].inner == Ti::Scalar(crate::Scalar::F32)
218                                    && match size {
219                                        crate::ArraySize::Constant(non_zero) => non_zero.get() <= 8,
220                                        _ => false,
221                                    }
222                            }
223                            _ => false,
224                        },
225                    ),
226                    Bi::PointSize => (
227                        self.stage == St::Vertex && self.output,
228                        *ty_inner == Ti::Scalar(crate::Scalar::F32),
229                    ),
230                    Bi::PointCoord => (
231                        self.stage == St::Fragment && !self.output,
232                        *ty_inner
233                            == Ti::Vector {
234                                size: Vs::Bi,
235                                scalar: crate::Scalar::F32,
236                            },
237                    ),
238                    Bi::Position { .. } => (
239                        match self.stage {
240                            St::Vertex => self.output,
241                            St::Fragment => !self.output,
242                            St::Compute => false,
243                            St::Task | St::Mesh => unreachable!(),
244                        },
245                        *ty_inner
246                            == Ti::Vector {
247                                size: Vs::Quad,
248                                scalar: crate::Scalar::F32,
249                            },
250                    ),
251                    Bi::ViewIndex => (
252                        match self.stage {
253                            St::Vertex | St::Fragment => !self.output,
254                            St::Compute => false,
255                            St::Task | St::Mesh => unreachable!(),
256                        },
257                        *ty_inner == Ti::Scalar(crate::Scalar::I32),
258                    ),
259                    Bi::FragDepth => (
260                        self.stage == St::Fragment && self.output,
261                        *ty_inner == Ti::Scalar(crate::Scalar::F32),
262                    ),
263                    Bi::FrontFacing => (
264                        self.stage == St::Fragment && !self.output,
265                        *ty_inner == Ti::Scalar(crate::Scalar::BOOL),
266                    ),
267                    Bi::PrimitiveIndex => (
268                        self.stage == St::Fragment && !self.output,
269                        *ty_inner == Ti::Scalar(crate::Scalar::U32),
270                    ),
271                    Bi::Barycentric => (
272                        self.stage == St::Fragment && !self.output,
273                        *ty_inner
274                            == Ti::Vector {
275                                size: Vs::Tri,
276                                scalar: crate::Scalar::F32,
277                            },
278                    ),
279                    Bi::SampleIndex => (
280                        self.stage == St::Fragment && !self.output,
281                        *ty_inner == Ti::Scalar(crate::Scalar::U32),
282                    ),
283                    Bi::SampleMask => (
284                        self.stage == St::Fragment,
285                        *ty_inner == Ti::Scalar(crate::Scalar::U32),
286                    ),
287                    Bi::LocalInvocationIndex => (
288                        self.stage == St::Compute && !self.output,
289                        *ty_inner == Ti::Scalar(crate::Scalar::U32),
290                    ),
291                    Bi::GlobalInvocationId
292                    | Bi::LocalInvocationId
293                    | Bi::WorkGroupId
294                    | Bi::WorkGroupSize
295                    | Bi::NumWorkGroups => (
296                        self.stage == St::Compute && !self.output,
297                        *ty_inner
298                            == Ti::Vector {
299                                size: Vs::Tri,
300                                scalar: crate::Scalar::U32,
301                            },
302                    ),
303                    Bi::NumSubgroups | Bi::SubgroupId => (
304                        self.stage == St::Compute && !self.output,
305                        *ty_inner == Ti::Scalar(crate::Scalar::U32),
306                    ),
307                    Bi::SubgroupSize | Bi::SubgroupInvocationId => (
308                        match self.stage {
309                            St::Compute | St::Fragment => !self.output,
310                            St::Vertex => false,
311                            St::Task | St::Mesh => unreachable!(),
312                        },
313                        *ty_inner == Ti::Scalar(crate::Scalar::U32),
314                    ),
315                };
316
317                if !visible {
318                    return Err(VaryingError::InvalidBuiltInStage(built_in));
319                }
320                if !type_good {
321                    log::warn!("Wrong builtin type: {ty_inner:?}");
322                    return Err(VaryingError::InvalidBuiltInType(built_in));
323                }
324            }
325            crate::Binding::Location {
326                location,
327                interpolation,
328                sampling,
329                blend_src,
330            } => {
331                // Only IO-shareable types may be stored in locations.
332                if !self.type_info[ty.index()]
333                    .flags
334                    .contains(super::TypeFlags::IO_SHAREABLE)
335                {
336                    return Err(VaryingError::NotIOShareableType(ty));
337                }
338
339                if let Some(blend_src) = blend_src {
340                    // `blend_src` is only valid if dual source blending was explicitly enabled,
341                    // see https://www.w3.org/TR/WGSL/#extension-dual_source_blending
342                    if !self
343                        .capabilities
344                        .contains(Capabilities::DUAL_SOURCE_BLENDING)
345                    {
346                        return Err(VaryingError::UnsupportedCapability(
347                            Capabilities::DUAL_SOURCE_BLENDING,
348                        ));
349                    }
350                    if self.stage != crate::ShaderStage::Fragment {
351                        return Err(VaryingError::InvalidAttributeInStage(
352                            "blend_src",
353                            self.stage,
354                        ));
355                    }
356                    if !self.output {
357                        return Err(VaryingError::InvalidInputAttributeInStage(
358                            "blend_src",
359                            self.stage,
360                        ));
361                    }
362                    if (blend_src != 0 && blend_src != 1) || location != 0 {
363                        return Err(VaryingError::InvalidBlendSrcIndex {
364                            location,
365                            blend_src,
366                        });
367                    }
368                    if !self.blend_src_mask.insert(blend_src as usize) {
369                        return Err(VaryingError::BindingCollisionBlendSrc { blend_src });
370                    }
371                } else if !self.location_mask.insert(location as usize)
372                    && self.flags.contains(super::ValidationFlags::BINDINGS)
373                {
374                    return Err(VaryingError::BindingCollision { location });
375                }
376
377                if let Some(interpolation) = interpolation {
378                    let invalid_sampling = match (interpolation, sampling) {
379                        (_, None)
380                        | (
381                            crate::Interpolation::Perspective | crate::Interpolation::Linear,
382                            Some(
383                                crate::Sampling::Center
384                                | crate::Sampling::Centroid
385                                | crate::Sampling::Sample,
386                            ),
387                        )
388                        | (
389                            crate::Interpolation::Flat,
390                            Some(crate::Sampling::First | crate::Sampling::Either),
391                        ) => None,
392                        (_, Some(invalid_sampling)) => Some(invalid_sampling),
393                    };
394                    if let Some(sampling) = invalid_sampling {
395                        return Err(VaryingError::InvalidInterpolationSamplingCombination {
396                            interpolation,
397                            sampling,
398                        });
399                    }
400                }
401
402                let needs_interpolation = match self.stage {
403                    crate::ShaderStage::Vertex => self.output,
404                    crate::ShaderStage::Fragment => !self.output,
405                    crate::ShaderStage::Compute => false,
406                    crate::ShaderStage::Task | crate::ShaderStage::Mesh => unreachable!(),
407                };
408
409                // It doesn't make sense to specify a sampling when `interpolation` is `Flat`, but
410                // SPIR-V and GLSL both explicitly tolerate such combinations of decorators /
411                // qualifiers, so we won't complain about that here.
412                let _ = sampling;
413
414                let required = match sampling {
415                    Some(crate::Sampling::Sample) => Capabilities::MULTISAMPLED_SHADING,
416                    _ => Capabilities::empty(),
417                };
418                if !self.capabilities.contains(required) {
419                    return Err(VaryingError::UnsupportedCapability(required));
420                }
421
422                match ty_inner.scalar_kind() {
423                    Some(crate::ScalarKind::Float) => {
424                        if needs_interpolation && interpolation.is_none() {
425                            return Err(VaryingError::MissingInterpolation);
426                        }
427                    }
428                    Some(_) => {
429                        if needs_interpolation && interpolation != Some(crate::Interpolation::Flat)
430                        {
431                            return Err(VaryingError::InvalidInterpolation);
432                        }
433                    }
434                    None => return Err(VaryingError::InvalidType(ty)),
435                }
436            }
437        }
438
439        Ok(())
440    }
441
442    fn validate(
443        &mut self,
444        ep: &crate::EntryPoint,
445        ty: Handle<crate::Type>,
446        binding: Option<&crate::Binding>,
447    ) -> Result<(), WithSpan<VaryingError>> {
448        let span_context = self.types.get_span_context(ty);
449        match binding {
450            Some(binding) => self
451                .validate_impl(ep, ty, binding)
452                .map_err(|e| e.with_span_context(span_context)),
453            None => {
454                match self.types[ty].inner {
455                    crate::TypeInner::Struct { ref members, .. } => {
456                        for (index, member) in members.iter().enumerate() {
457                            let span_context = self.types.get_span_context(ty);
458                            match member.binding {
459                                None => {
460                                    if self.flags.contains(super::ValidationFlags::BINDINGS) {
461                                        return Err(VaryingError::MemberMissingBinding(
462                                            index as u32,
463                                        )
464                                        .with_span_context(span_context));
465                                    }
466                                }
467                                Some(ref binding) => self
468                                    .validate_impl(ep, member.ty, binding)
469                                    .map_err(|e| e.with_span_context(span_context))?,
470                            }
471                        }
472
473                        if !self.blend_src_mask.is_empty() {
474                            let span_context = self.types.get_span_context(ty);
475
476                            // If there's any blend_src usage, it must apply to all members of which there must be exactly 2.
477                            if members.len() != 2 || self.blend_src_mask.len() != 2 {
478                                return Err(VaryingError::IncompleteBlendSrcUsage
479                                    .with_span_context(span_context));
480                            }
481                            // Also, all members must have the same type.
482                            if members[0].ty != members[1].ty {
483                                return Err(VaryingError::BlendSrcOutputTypeMismatch {
484                                    blend_src_0_type: members[0].ty,
485                                    blend_src_1_type: members[1].ty,
486                                }
487                                .with_span_context(span_context));
488                            }
489                        }
490                    }
491                    _ => {
492                        if self.flags.contains(super::ValidationFlags::BINDINGS) {
493                            return Err(VaryingError::MissingBinding.with_span());
494                        }
495                    }
496                }
497                Ok(())
498            }
499        }
500    }

503impl super::Validator {
504    pub(super) fn validate_global_var(
505        &self,
506        var: &crate::GlobalVariable,
507        gctx: crate::proc::GlobalCtx,
508        mod_info: &ModuleInfo,
509        global_expr_kind: &crate::proc::ExpressionKindTracker,
510    ) -> Result<(), GlobalVariableError> {
511        use super::TypeFlags;
512
513        log::debug!("var {var:?}");
514        let inner_ty = match gctx.types[var.ty].inner {
515            // A binding array is (mostly) supposed to behave the same as a
516            // series of individually bound resources, so we can (mostly)
517            // validate a `binding_array<T>` as if it were just a plain `T`.
518            crate::TypeInner::BindingArray { base, .. } => match var.space {
519                crate::AddressSpace::Storage { .. }
520                | crate::AddressSpace::Uniform
521                | crate::AddressSpace::Handle => base,
522                _ => return Err(GlobalVariableError::InvalidUsage(var.space)),
523            },
524            _ => var.ty,
525        };
526        let type_info = &self.types[inner_ty.index()];
527
528        let (required_type_flags, is_resource) = match var.space {
529            crate::AddressSpace::Function => {
530                return Err(GlobalVariableError::InvalidUsage(var.space))
531            }
532            crate::AddressSpace::Storage { access } => {
533                if let Err((ty_handle, disalignment)) = type_info.storage_layout {
534                    if self.flags.contains(super::ValidationFlags::STRUCT_LAYOUTS) {
535                        return Err(GlobalVariableError::Alignment(
536                            var.space,
537                            ty_handle,
538                            disalignment,
539                        ));
540                    }
541                }
542                if access == crate::StorageAccess::STORE {
543                    return Err(GlobalVariableError::StorageAddressSpaceWriteOnlyNotSupported);
544                }
545                (
546                    TypeFlags::DATA | TypeFlags::HOST_SHAREABLE | TypeFlags::CREATION_RESOLVED,
547                    true,
548                )
549            }
550            crate::AddressSpace::Uniform => {
551                if let Err((ty_handle, disalignment)) = type_info.uniform_layout {
552                    if self.flags.contains(super::ValidationFlags::STRUCT_LAYOUTS) {
553                        return Err(GlobalVariableError::Alignment(
554                            var.space,
555                            ty_handle,
556                            disalignment,
557                        ));
558                    }
559                }
560                (
561                    TypeFlags::DATA
562                        | TypeFlags::COPY
563                        | TypeFlags::SIZED
564                        | TypeFlags::HOST_SHAREABLE
565                        | TypeFlags::CREATION_RESOLVED,
566                    true,
567                )
568            }
569            crate::AddressSpace::Handle => {
570                match gctx.types[inner_ty].inner {
571                    crate::TypeInner::Image { class, .. } => match class {
572                        crate::ImageClass::Storage {
573                            format:
574                                crate::StorageFormat::R16Unorm
575                                | crate::StorageFormat::R16Snorm
576                                | crate::StorageFormat::Rg16Unorm
577                                | crate::StorageFormat::Rg16Snorm
578                                | crate::StorageFormat::Rgba16Unorm
579                                | crate::StorageFormat::Rgba16Snorm,
580                            ..
581                        } => {
582                            if !self
583                                .capabilities
584                                .contains(Capabilities::STORAGE_TEXTURE_16BIT_NORM_FORMATS)
585                            {
586                                return Err(GlobalVariableError::UnsupportedCapability(
587                                    Capabilities::STORAGE_TEXTURE_16BIT_NORM_FORMATS,
588                                ));
589                            }
590                        }
591                        _ => {}
592                    },
593                    crate::TypeInner::Sampler { .. }
594                    | crate::TypeInner::AccelerationStructure { .. }
595                    | crate::TypeInner::RayQuery { .. } => {}
596                    _ => {
597                        return Err(GlobalVariableError::InvalidType(var.space));
598                    }
599                }
600
601                (TypeFlags::empty(), true)
602            }
603            crate::AddressSpace::Private => (
604                TypeFlags::CONSTRUCTIBLE | TypeFlags::CREATION_RESOLVED,
605                false,
606            ),
607            crate::AddressSpace::WorkGroup => (TypeFlags::DATA | TypeFlags::SIZED, false),
608            crate::AddressSpace::PushConstant => {
609                if !self.capabilities.contains(Capabilities::PUSH_CONSTANT) {
610                    return Err(GlobalVariableError::UnsupportedCapability(
611                        Capabilities::PUSH_CONSTANT,
612                    ));
613                }
614                if let Err(ref err) = type_info.push_constant_compatibility {
615                    return Err(GlobalVariableError::InvalidPushConstantType(err.clone()));
616                }
617                (
618                    TypeFlags::DATA
619                        | TypeFlags::COPY
620                        | TypeFlags::HOST_SHAREABLE
621                        | TypeFlags::SIZED,
622                    false,
623                )
624            }
625        };
626
627        if !type_info.flags.contains(required_type_flags) {
628            return Err(GlobalVariableError::MissingTypeFlags {
629                seen: type_info.flags,
630                required: required_type_flags,
631            });
632        }
633
634        if is_resource != var.binding.is_some() {
635            if self.flags.contains(super::ValidationFlags::BINDINGS) {
636                return Err(GlobalVariableError::InvalidBinding);
637            }
638        }
639
640        if let Some(init) = var.init {
641            match var.space {
642                crate::AddressSpace::Private | crate::AddressSpace::Function => {}
643                _ => {
644                    return Err(GlobalVariableError::InitializerNotAllowed(var.space));
645                }
646            }
647
648            if !global_expr_kind.is_const_or_override(init) {
649                return Err(GlobalVariableError::InitializerExprType);
650            }
651
652            if !gctx.compare_types(
653                &crate::proc::TypeResolution::Handle(var.ty),
654                &mod_info[init],
655            ) {
656                return Err(GlobalVariableError::InitializerType);
657            }
658        }
659
660        Ok(())
661    }
662
663    pub(super) fn validate_entry_point(
664        &mut self,
665        ep: &crate::EntryPoint,
666        module: &crate::Module,
667        mod_info: &ModuleInfo,
668    ) -> Result<FunctionInfo, WithSpan<EntryPointError>> {
669        if ep.early_depth_test.is_some() {
670            let required = Capabilities::EARLY_DEPTH_TEST;
671            if !self.capabilities.contains(required) {
672                return Err(
673                    EntryPointError::Result(VaryingError::UnsupportedCapability(required))
674                        .with_span(),
675                );
676            }
677
678            if ep.stage != crate::ShaderStage::Fragment {
679                return Err(EntryPointError::UnexpectedEarlyDepthTest.with_span());
680            }
681        }
682
683        if ep.stage == crate::ShaderStage::Compute {
684            if ep
685                .workgroup_size
686                .iter()
687                .any(|&s| s == 0 || s > MAX_WORKGROUP_SIZE)
688            {
689                return Err(EntryPointError::OutOfRangeWorkgroupSize.with_span());
690            }
691        } else if ep.workgroup_size != [0; 3] {
692            return Err(EntryPointError::UnexpectedWorkgroupSize.with_span());
693        }
694
695        let mut info = self
696            .validate_function(&ep.function, module, mod_info, true)
697            .map_err(WithSpan::into_other)?;
698
699        {
700            use super::ShaderStages;
701
702            let stage_bit = match ep.stage {
703                crate::ShaderStage::Vertex => ShaderStages::VERTEX,
704                crate::ShaderStage::Fragment => ShaderStages::FRAGMENT,
705                crate::ShaderStage::Compute => ShaderStages::COMPUTE,
706                crate::ShaderStage::Task | crate::ShaderStage::Mesh => unreachable!(),
707            };
708
709            if !info.available_stages.contains(stage_bit) {
710                return Err(EntryPointError::ForbiddenStageOperations.with_span());
711            }
712        }
713
714        self.location_mask.clear();
715        let mut argument_built_ins = crate::FastHashSet::default();
716        // TODO: add span info to function arguments
717        for (index, fa) in ep.function.arguments.iter().enumerate() {
718            let mut ctx = VaryingContext {
719                stage: ep.stage,
720                output: false,
721                types: &module.types,
722                type_info: &self.types,
723                location_mask: &mut self.location_mask,
724                blend_src_mask: &mut self.blend_src_mask,
725                built_ins: &mut argument_built_ins,
726                capabilities: self.capabilities,
727                flags: self.flags,
728            };
729            ctx.validate(ep, fa.ty, fa.binding.as_ref())
730                .map_err_inner(|e| EntryPointError::Argument(index as u32, e).with_span())?;
731        }
732
733        self.location_mask.clear();
734        if let Some(ref fr) = ep.function.result {
735            let mut result_built_ins = crate::FastHashSet::default();
736            let mut ctx = VaryingContext {
737                stage: ep.stage,
738                output: true,
739                types: &module.types,
740                type_info: &self.types,
741                location_mask: &mut self.location_mask,
742                blend_src_mask: &mut self.blend_src_mask,
743                built_ins: &mut result_built_ins,
744                capabilities: self.capabilities,
745                flags: self.flags,
746            };
747            ctx.validate(ep, fr.ty, fr.binding.as_ref())
748                .map_err_inner(|e| EntryPointError::Result(e).with_span())?;
749            if ep.stage == crate::ShaderStage::Vertex
750                && !result_built_ins.contains(&crate::BuiltIn::Position { invariant: false })
751            {
752                return Err(EntryPointError::MissingVertexOutputPosition.with_span());
753            }
754            if !self.blend_src_mask.is_empty() {
755                info.dual_source_blending = true;
756            }
757        } else if ep.stage == crate::ShaderStage::Vertex {
758            return Err(EntryPointError::MissingVertexOutputPosition.with_span());
759        }
760
761        {
762            let mut used_push_constants = module
763                .global_variables
764                .iter()
765                .filter(|&(_, var)| var.space == crate::AddressSpace::PushConstant)
766                .map(|(handle, _)| handle)
767                .filter(|&handle| !info[handle].is_empty());
768            // Check if there is more than one push constant, and error if so.
769            // Use a loop for when returning multiple errors is supported.
770            if let Some(handle) = used_push_constants.nth(1) {
771                return Err(EntryPointError::MoreThanOnePushConstantUsed
772                    .with_span_handle(handle, &module.global_variables));
773            }
774        }
775
776        self.ep_resource_bindings.clear();
777        for (var_handle, var) in module.global_variables.iter() {
778            let usage = info[var_handle];
779            if usage.is_empty() {
780                continue;
781            }
782
783            let allowed_usage = match var.space {
784                crate::AddressSpace::Function => unreachable!(),
785                crate::AddressSpace::Uniform => GlobalUse::READ | GlobalUse::QUERY,
786                crate::AddressSpace::Storage { access } => storage_usage(access),
787                crate::AddressSpace::Handle => match module.types[var.ty].inner {
788                    crate::TypeInner::BindingArray { base, .. } => match module.types[base].inner {
789                        crate::TypeInner::Image {
790                            class: crate::ImageClass::Storage { access, .. },
791                            ..
792                        } => storage_usage(access),
793                        _ => GlobalUse::READ | GlobalUse::QUERY,
794                    },
795                    crate::TypeInner::Image {
796                        class: crate::ImageClass::Storage { access, .. },
797                        ..
798                    } => storage_usage(access),
799                    _ => GlobalUse::READ | GlobalUse::QUERY,
800                },
801                crate::AddressSpace::Private | crate::AddressSpace::WorkGroup => {
802                    GlobalUse::READ | GlobalUse::WRITE | GlobalUse::QUERY
803                }
804                crate::AddressSpace::PushConstant => GlobalUse::READ,
805            };
806            if !allowed_usage.contains(usage) {
807                log::warn!("\tUsage error for: {var:?}");
808                log::warn!("\tAllowed usage: {allowed_usage:?}, requested: {usage:?}");
809                return Err(EntryPointError::InvalidGlobalUsage(var_handle, usage)
810                    .with_span_handle(var_handle, &module.global_variables));
811            }
812
813            if let Some(ref bind) = var.binding {
814                if !self.ep_resource_bindings.insert(*bind) {
815                    if self.flags.contains(super::ValidationFlags::BINDINGS) {
816                        return Err(EntryPointError::BindingCollision(var_handle)
817                            .with_span_handle(var_handle, &module.global_variables));
818                    }
819                }
820            }
821        }
822
823        Ok(info)
824    }
