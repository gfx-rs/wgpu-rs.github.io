41bitflags::bitflags! {
42    /// Validation flags.
43    ///
44    /// If you are working with trusted shaders, then you may be able
45    /// to save some time by skipping validation.
46    ///
47    /// If you do not perform full validation, invalid shaders may
48    /// cause Naga to panic. If you do perform full validation and
49    /// [`Validator::validate`] returns `Ok`, then Naga promises that
50    /// code generation will either succeed or return an error; it
51    /// should never panic.
52    ///
53    /// The default value for `ValidationFlags` is
54    /// `ValidationFlags::all()`.
55    #[cfg_attr(feature = "serialize", derive(serde::Serialize))]
56    #[cfg_attr(feature = "deserialize", derive(serde::Deserialize))]
57    #[derive(Clone, Copy, Debug, Eq, PartialEq)]
58    pub struct ValidationFlags: u8 {
59        /// Expressions.
60        const EXPRESSIONS = 0x1;
61        /// Statements and blocks of them.
62        const BLOCKS = 0x2;
63        /// Uniformity of control flow for operations that require it.
64        const CONTROL_FLOW_UNIFORMITY = 0x4;
65        /// Host-shareable structure layouts.
66        const STRUCT_LAYOUTS = 0x8;
67        /// Constants.
68        const CONSTANTS = 0x10;
69        /// Group, binding, and location attributes.
70        const BINDINGS = 0x20;
71    }
72}
73
74impl Default for ValidationFlags {
75    fn default() -> Self {
76        Self::all()
77    }
78}

208impl super::SubgroupOperation {
209    const fn required_operations(&self) -> SubgroupOperationSet {
210        use SubgroupOperationSet as S;
211        match *self {
212            Self::All | Self::Any => S::VOTE,
213            Self::Add | Self::Mul | Self::Min | Self::Max | Self::And | Self::Or | Self::Xor => {
214                S::ARITHMETIC

220impl super::GatherMode {
221    const fn required_operations(&self) -> SubgroupOperationSet {

227            Self::QuadBroadcast(_) | Self::QuadSwap(_) => S::QUAD_FRAGMENT_COMPUTE,
228        }

460    /// Construct a new validator instance.
461    pub fn new(flags: ValidationFlags, capabilities: Capabilities) -> Self {
462        let subgroup_operations = if capabilities.contains(Capabilities::SUBGROUP) {
463            use SubgroupOperationSet as S;
464            S::BASIC
465                | S::VOTE
466                | S::ARITHMETIC
467                | S::BALLOT
468                | S::SHUFFLE
469                | S::SHUFFLE_RELATIVE
470                | S::QUAD_FRAGMENT_COMPUTE
471        } else {
472            SubgroupOperationSet::empty()
473        };
474        let subgroup_stages = {
475            let mut stages = ShaderStages::empty();
476            if capabilities.contains(Capabilities::SUBGROUP_VERTEX_STAGE) {
477                stages |= ShaderStages::VERTEX;
478            }
479            if capabilities.contains(Capabilities::SUBGROUP) {
480                stages |= ShaderStages::FRAGMENT | ShaderStages::COMPUTE;
481            }
482            stages
483        };
484
485        Validator {
486            flags,
487            capabilities,
488            subgroup_stages,
489            subgroup_operations,
490            types: Vec::new(),
491            layouter: Layouter::default(),
492            location_mask: BitSet::new(),
493            blend_src_mask: BitSet::new(),
494            ep_resource_bindings: FastHashSet::default(),
495            switch_values: FastHashSet::default(),
496            valid_expression_list: Vec::new(),
497            valid_expression_set: HandleSet::new(),
498            override_ids: FastHashSet::default(),
499            overrides_resolved: false,
500            needs_visit: HandleSet::new(),
501        }
502    }

526
527    fn validate_constant(
528        &self,
529        handle: Handle<crate::Constant>,
530        gctx: crate::proc::GlobalCtx,
531        mod_info: &ModuleInfo,
532        global_expr_kind: &ExpressionKindTracker,
533    ) -> Result<(), ConstantError> {
534        let con = &gctx.constants[handle];
535
536        let type_info = &self.types[con.ty.index()];
537        if !type_info.flags.contains(TypeFlags::CONSTRUCTIBLE) {
538            return Err(ConstantError::NonConstructibleType);
539        }
540
541        if !global_expr_kind.is_const(con.init) {
542            return Err(ConstantError::InitializerExprType);
543        }
544
545        if !gctx.compare_types(&TypeResolution::Handle(con.ty), &mod_info[con.init]) {
546            return Err(ConstantError::InvalidType);
547        }
548
549        Ok(())
550    }
551
552    fn validate_override(
553        &mut self,
554        handle: Handle<crate::Override>,
555        gctx: crate::proc::GlobalCtx,
556        mod_info: &ModuleInfo,
557    ) -> Result<(), OverrideError> {
558        let o = &gctx.overrides[handle];
559
560        if let Some(id) = o.id {
561            if !self.override_ids.insert(id) {
562                return Err(OverrideError::DuplicateID);
563            }
564        }
565
566        let type_info = &self.types[o.ty.index()];
567        if !type_info.flags.contains(TypeFlags::CONSTRUCTIBLE) {
568            return Err(OverrideError::NonConstructibleType);
569        }
570
571        match gctx.types[o.ty].inner {
572            crate::TypeInner::Scalar(
573                crate::Scalar::BOOL
574                | crate::Scalar::I32
575                | crate::Scalar::U32
576                | crate::Scalar::F16
577                | crate::Scalar::F32
578                | crate::Scalar::F64,
579            ) => {}
580            _ => return Err(OverrideError::TypeNotScalar),
581        }
582
583        if let Some(init) = o.init {
584            if !gctx.compare_types(&TypeResolution::Handle(o.ty), &mod_info[init]) {
585                return Err(OverrideError::InvalidType);
586            }
587        } else if self.overrides_resolved {
588            return Err(OverrideError::UninitializedOverride);
589        }
590
591        Ok(())
592    }

603    /// Check the given module to be valid, requiring overrides to be resolved.
604    ///

609    /// [`validate`]: Validator::validate

617
618    fn validate_impl(
619        &mut self,
620        module: &crate::Module,
621    ) -> Result<ModuleInfo, WithSpan<ValidationError>> {
622        self.reset();
623        self.reset_types(module.types.len());
624
625        Self::validate_module_handles(module).map_err(|e| e.with_span())?;
626
627        self.layouter.update(module.to_ctx()).map_err(|e| {
628            let handle = e.ty;
629            ValidationError::from(e).with_span_handle(handle, &module.types)
630        })?;
631
632        // These should all get overwritten.
633        let placeholder = TypeResolution::Value(crate::TypeInner::Scalar(crate::Scalar {
634            kind: crate::ScalarKind::Bool,
635            width: 0,
636        }));
637
638        let mut mod_info = ModuleInfo {
639            type_flags: Vec::with_capacity(module.types.len()),
640            functions: Vec::with_capacity(module.functions.len()),
641            entry_points: Vec::with_capacity(module.entry_points.len()),
642            const_expression_types: vec![placeholder; module.global_expressions.len()]
643                .into_boxed_slice(),
644        };
645
646        for (handle, ty) in module.types.iter() {
647            let ty_info = self
648                .validate_type(handle, module.to_ctx())
649                .map_err(|source| {
650                    ValidationError::Type {
651                        handle,
652                        name: ty.name.clone().unwrap_or_default(),
653                        source,
654                    }
655                    .with_span_handle(handle, &module.types)
656                })?;
657            mod_info.type_flags.push(ty_info.flags);
658            self.types[handle.index()] = ty_info;
659        }
660
661        {
662            let t = crate::Arena::new();
663            let resolve_context = crate::proc::ResolveContext::with_locals(module, &t, &[]);
664            for (handle, _) in module.global_expressions.iter() {
665                mod_info
666                    .process_const_expression(handle, &resolve_context, module.to_ctx())
667                    .map_err(|source| {
668                        ValidationError::ConstExpression { handle, source }
669                            .with_span_handle(handle, &module.global_expressions)
670                    })?
671            }
672        }
673
674        let global_expr_kind = ExpressionKindTracker::from_arena(&module.global_expressions);
675
676        if self.flags.contains(ValidationFlags::CONSTANTS) {
677            for (handle, _) in module.global_expressions.iter() {
678                self.validate_const_expression(
679                    handle,
680                    module.to_ctx(),
681                    &mod_info,
682                    &global_expr_kind,
683                )
684                .map_err(|source| {
685                    ValidationError::ConstExpression { handle, source }
686                        .with_span_handle(handle, &module.global_expressions)
687                })?
688            }
689
690            for (handle, constant) in module.constants.iter() {
691                self.validate_constant(handle, module.to_ctx(), &mod_info, &global_expr_kind)
692                    .map_err(|source| {
693                        ValidationError::Constant {
694                            handle,
695                            name: constant.name.clone().unwrap_or_default(),
696                            source,
697                        }
698                        .with_span_handle(handle, &module.constants)
699                    })?
700            }
701
702            for (handle, r#override) in module.overrides.iter() {
703                self.validate_override(handle, module.to_ctx(), &mod_info)
704                    .map_err(|source| {
705                        ValidationError::Override {
706                            handle,
707                            name: r#override.name.clone().unwrap_or_default(),
708                            source,
709                        }
710                        .with_span_handle(handle, &module.overrides)
711                    })?;
712            }
713        }
714
715        for (var_handle, var) in module.global_variables.iter() {
716            self.validate_global_var(var, module.to_ctx(), &mod_info, &global_expr_kind)
717                .map_err(|source| {
718                    ValidationError::GlobalVariable {
719                        handle: var_handle,
720                        name: var.name.clone().unwrap_or_default(),
721                        source,
722                    }
723                    .with_span_handle(var_handle, &module.global_variables)
724                })?;
725        }
726
727        for (handle, fun) in module.functions.iter() {
728            match self.validate_function(fun, module, &mod_info, false) {
729                Ok(info) => mod_info.functions.push(info),
730                Err(error) => {
731                    return Err(error.and_then(|source| {
732                        ValidationError::Function {
733                            handle,
734                            name: fun.name.clone().unwrap_or_default(),
735                            source,
736                        }
737                        .with_span_handle(handle, &module.functions)
738                    }))
739                }
740            }
741        }
742
743        let mut ep_map = FastHashSet::default();
744        for ep in module.entry_points.iter() {
745            if !ep_map.insert((ep.stage, &ep.name)) {
746                return Err(ValidationError::EntryPoint {
747                    stage: ep.stage,
748                    name: ep.name.clone(),
749                    source: EntryPointError::Conflict,
750                }
751                .with_span()); // TODO: keep some EP span information?
752            }
753
754            match self.validate_entry_point(ep, module, &mod_info) {
755                Ok(info) => mod_info.entry_points.push(info),
756                Err(error) => {
757                    return Err(error.and_then(|source| {
758                        ValidationError::EntryPoint {
759                            stage: ep.stage,
760                            name: ep.name.clone(),
761                            source,
762                        }
763                        .with_span()
764                    }));
765                }
766            }
767        }
768
769        Ok(mod_info)
770    }
771}

773fn validate_atomic_compare_exchange_struct(
774    types: &crate::UniqueArena<crate::Type>,
775    members: &[crate::StructMember],
776    scalar_predicate: impl FnOnce(&crate::TypeInner) -> bool,
777) -> bool {
778    members.len() == 2
779        && members[0].name.as_deref() == Some("old_value")
780        && scalar_predicate(&types[members[0].ty].inner)
781        && members[1].name.as_deref() == Some("exchanged")
782        && types[members[1].ty].inner == crate::TypeInner::Scalar(crate::Scalar::BOOL)
783}
