78/// Represents the sets of limits an adapter/device supports.
79///
80/// We provide three different defaults.
81/// - [`Limits::downlevel_defaults()`]. This is a set of limits that is guaranteed to work on almost
82///   all backends, including "downlevel" backends such as OpenGL and D3D11, other than WebGL. For
83///   most applications we recommend using these limits, assuming they are high enough for your
84///   application, and you do not intent to support WebGL.
85/// - [`Limits::downlevel_webgl2_defaults()`] This is a set of limits that is lower even than the
86///   [`downlevel_defaults()`], configured to be low enough to support running in the browser using
87///   WebGL2.
88/// - [`Limits::default()`]. This is the set of limits that is guaranteed to work on all modern
89///   backends and is guaranteed to be supported by WebGPU. Applications needing more modern
90///   features can use this as a reasonable set of limits if they are targeting only desktop and
91///   modern mobile devices.
92///
93/// We recommend starting with the most restrictive limits you can and manually increasing the
94/// limits you need boosted. This will let you stay running on all hardware that supports the limits
95/// you need.
96///
97/// Limits "better" than the default must be supported by the adapter and requested when requesting
98/// a device. If limits "better" than the adapter supports are requested, requesting a device will
99/// panic. Once a device is requested, you may only use resources up to the limits requested _even_
100/// if the adapter supports "better" limits.
101///
102/// Requesting limits that are "better" than you need may cause performance to decrease because the
103/// implementation needs to support more than is needed. You should ideally only request exactly
104/// what you need.
105///
106/// Corresponds to [WebGPU `GPUSupportedLimits`](
107/// https://gpuweb.github.io/gpuweb/#gpusupportedlimits).
108///
109/// [`downlevel_defaults()`]: Limits::downlevel_defaults
110#[repr(C)]
111#[derive(Clone, Debug, PartialEq, Eq, Hash)]
112#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
113#[cfg_attr(feature = "serde", serde(rename_all = "camelCase", default))]
114pub struct Limits {
115    /// Maximum allowed value for the `size.width` of a texture created with `TextureDimension::D1`.
116    /// Defaults to 8192. Higher is "better".
117    #[cfg_attr(feature = "serde", serde(rename = "maxTextureDimension1D"))]
118    pub max_texture_dimension_1d: u32,
119    /// Maximum allowed value for the `size.width` and `size.height` of a texture created with `TextureDimension::D2`.
120    /// Defaults to 8192. Higher is "better".
121    #[cfg_attr(feature = "serde", serde(rename = "maxTextureDimension2D"))]
122    pub max_texture_dimension_2d: u32,
123    /// Maximum allowed value for the `size.width`, `size.height`, and `size.depth_or_array_layers`
124    /// of a texture created with `TextureDimension::D3`.
125    /// Defaults to 2048. Higher is "better".
126    #[cfg_attr(feature = "serde", serde(rename = "maxTextureDimension3D"))]
127    pub max_texture_dimension_3d: u32,
128    /// Maximum allowed value for the `size.depth_or_array_layers` of a texture created with `TextureDimension::D2`.
129    /// Defaults to 256. Higher is "better".
130    pub max_texture_array_layers: u32,
131    /// Amount of bind groups that can be attached to a pipeline at the same time. Defaults to 4. Higher is "better".
132    pub max_bind_groups: u32,
133    /// Maximum binding index allowed in `create_bind_group_layout`. Defaults to 1000. Higher is "better".
134    pub max_bindings_per_bind_group: u32,
135    /// Amount of uniform buffer bindings that can be dynamic in a single pipeline. Defaults to 8. Higher is "better".
136    pub max_dynamic_uniform_buffers_per_pipeline_layout: u32,
137    /// Amount of storage buffer bindings that can be dynamic in a single pipeline. Defaults to 4. Higher is "better".
138    pub max_dynamic_storage_buffers_per_pipeline_layout: u32,
139    /// Amount of sampled textures visible in a single shader stage. Defaults to 16. Higher is "better".
140    pub max_sampled_textures_per_shader_stage: u32,
141    /// Amount of samplers visible in a single shader stage. Defaults to 16. Higher is "better".
142    pub max_samplers_per_shader_stage: u32,
143    /// Amount of storage buffers visible in a single shader stage. Defaults to 8. Higher is "better".
144    pub max_storage_buffers_per_shader_stage: u32,
145    /// Amount of storage textures visible in a single shader stage. Defaults to 4. Higher is "better".
146    pub max_storage_textures_per_shader_stage: u32,
147    /// Amount of uniform buffers visible in a single shader stage. Defaults to 12. Higher is "better".
148    pub max_uniform_buffers_per_shader_stage: u32,
149    /// Amount of individual resources within binding arrays that can be accessed in a single shader stage. Applies
150    /// to all types of bindings except samplers.
151    ///
152    /// This "defaults" to 0. However if binding arrays are supported, all devices can support 500,000. Higher is "better".
153    pub max_binding_array_elements_per_shader_stage: u32,
154    /// Amount of individual samplers within binding arrays that can be accessed in a single shader stage.
155    ///
156    /// This "defaults" to 0. However if binding arrays are supported, all devices can support 1,000. Higher is "better".
157    pub max_binding_array_sampler_elements_per_shader_stage: u32,
158    /// Maximum size in bytes of a binding to a uniform buffer. Defaults to 64 KiB. Higher is "better".
159    pub max_uniform_buffer_binding_size: u32,
160    /// Maximum size in bytes of a binding to a storage buffer. Defaults to 128 MiB. Higher is "better".
161    pub max_storage_buffer_binding_size: u32,
162    /// Maximum length of `VertexState::buffers` when creating a `RenderPipeline`.
163    /// Defaults to 8. Higher is "better".
164    pub max_vertex_buffers: u32,
165    /// A limit above which buffer allocations are guaranteed to fail.
166    /// Defaults to 256 MiB. Higher is "better".
167    ///
168    /// Buffer allocations below the maximum buffer size may not succeed depending on available memory,
169    /// fragmentation and other factors.
170    pub max_buffer_size: u64,
171    /// Maximum length of `VertexBufferLayout::attributes`, summed over all `VertexState::buffers`,
172    /// when creating a `RenderPipeline`.
173    /// Defaults to 16. Higher is "better".
174    pub max_vertex_attributes: u32,
175    /// Maximum value for `VertexBufferLayout::array_stride` when creating a `RenderPipeline`.
176    /// Defaults to 2048. Higher is "better".
177    pub max_vertex_buffer_array_stride: u32,
178    /// Required `BufferBindingType::Uniform` alignment for `BufferBinding::offset`
179    /// when creating a `BindGroup`, or for `set_bind_group` `dynamicOffsets`.
180    /// Defaults to 256. Lower is "better".
181    pub min_uniform_buffer_offset_alignment: u32,
182    /// Required `BufferBindingType::Storage` alignment for `BufferBinding::offset`
183    /// when creating a `BindGroup`, or for `set_bind_group` `dynamicOffsets`.
184    /// Defaults to 256. Lower is "better".
185    pub min_storage_buffer_offset_alignment: u32,
186    /// Maximum allowed number of components (scalars) of input or output locations for
187    /// inter-stage communication (vertex outputs to fragment inputs). Defaults to 60.
188    /// Higher is "better".
189    pub max_inter_stage_shader_components: u32,
190    /// The maximum allowed number of color attachments.
191    pub max_color_attachments: u32,
192    /// The maximum number of bytes necessary to hold one sample (pixel or subpixel) of render
193    /// pipeline output data, across all color attachments as described by [`TextureFormat::target_pixel_byte_cost`]
194    /// and [`TextureFormat::target_component_alignment`]. Defaults to 32. Higher is "better".
195    ///
196    /// ⚠️ `Rgba8Unorm`/`Rgba8Snorm`/`Bgra8Unorm`/`Bgra8Snorm` are deceptively 8 bytes per sample. ⚠️
197    pub max_color_attachment_bytes_per_sample: u32,
198    /// Maximum number of bytes used for workgroup memory in a compute entry point. Defaults to
199    /// 16384. Higher is "better".
200    pub max_compute_workgroup_storage_size: u32,
201    /// Maximum value of the product of the `workgroup_size` dimensions for a compute entry-point.
202    /// Defaults to 256. Higher is "better".
203    pub max_compute_invocations_per_workgroup: u32,
204    /// The maximum value of the `workgroup_size` X dimension for a compute stage `ShaderModule` entry-point.
205    /// Defaults to 256. Higher is "better".
206    pub max_compute_workgroup_size_x: u32,
207    /// The maximum value of the `workgroup_size` Y dimension for a compute stage `ShaderModule` entry-point.
208    /// Defaults to 256. Higher is "better".
209    pub max_compute_workgroup_size_y: u32,
210    /// The maximum value of the `workgroup_size` Z dimension for a compute stage `ShaderModule` entry-point.
211    /// Defaults to 64. Higher is "better".
212    pub max_compute_workgroup_size_z: u32,
213    /// The maximum value for each dimension of a `ComputePass::dispatch(x, y, z)` operation.
214    /// Defaults to 65535. Higher is "better".
215    pub max_compute_workgroups_per_dimension: u32,
216
217    /// Minimal number of invocations in a subgroup. Lower is "better".
218    pub min_subgroup_size: u32,
219    /// Maximal number of invocations in a subgroup. Higher is "better".
220    pub max_subgroup_size: u32,
221    /// Amount of storage available for immediates in bytes. Defaults to 0. Higher is "better".
222    /// Requesting more than 0 during device creation requires [`Features::IMMEDIATES`] to be enabled.
223    ///
224    /// Expect the size to be:
225    /// - Vulkan: 128-256 bytes
226    /// - DX12: 256 bytes
227    /// - Metal: 4096 bytes
228    /// - OpenGL doesn't natively support immediates, and are emulated with uniforms,
229    ///   so this number is less useful but likely 256.
230    pub max_immediate_size: u32,
231    /// Maximum number of live non-sampler bindings.
232    ///
233    /// <div class="warning">
234    /// The default value is **1_000_000**, On systems with integrated GPUs (iGPUs)—particularly on Windows using the D3D12
235    /// backend—this can lead to significant system RAM consumption since iGPUs share system memory directly with the CPU.
236    /// </div>
237    ///
238    /// This limit only affects the d3d12 backend. Using a large number will allow the device
239    /// to create many bind groups at the cost of a large up-front allocation at device creation.
240    pub max_non_sampler_bindings: u32,
241
242    /// The maximum total value of x*y*z for a given `draw_mesh_tasks` command
243    pub max_task_workgroup_total_count: u32,
244    /// The maximum value for each dimension of a `RenderPass::draw_mesh_tasks(x, y, z)` operation.
245    /// Defaults to 65535. Higher is "better".
246    pub max_task_workgroups_per_dimension: u32,
247    /// The maximum number of layers that can be output from a mesh shader
248    pub max_mesh_output_layers: u32,
249    /// The maximum number of views that can be used by a mesh shader in multiview rendering
250    pub max_mesh_multiview_view_count: u32,
251
252    /// The maximum number of primitive (ex: triangles, aabbs) a BLAS is allowed to have. Requesting
253    /// more than 0 during device creation only makes sense if [`Features::EXPERIMENTAL_RAY_QUERY`]
254    /// is enabled.
255    pub max_blas_primitive_count: u32,
256    /// The maximum number of geometry descriptors a BLAS is allowed to have. Requesting
257    /// more than 0 during device creation only makes sense if [`Features::EXPERIMENTAL_RAY_QUERY`]
258    /// is enabled.
259    pub max_blas_geometry_count: u32,
260    /// The maximum number of instances a TLAS is allowed to have. Requesting more than 0 during
261    /// device creation only makes sense if [`Features::EXPERIMENTAL_RAY_QUERY`]
262    /// is enabled.
263    pub max_tlas_instance_count: u32,
264    /// The maximum number of acceleration structures allowed to be used in a shader stage.
265    /// Requesting more than 0 during device creation only makes sense if [`Features::EXPERIMENTAL_RAY_QUERY`]
266    /// is enabled.
267    pub max_acceleration_structures_per_shader_stage: u32,
268
269    /// The maximum number of views that can be used in multiview rendering
270    pub max_multiview_view_count: u32,
271}

621    /// Compares every limits within self is within the limits given in `allowed`.
622    ///
623    /// If you need detailed information on failures, look at [`Limits::check_limits_with_fail_fn`].
624    #[must_use]
625    pub fn check_limits(&self, allowed: &Self) -> bool {
626        let mut within = true;
627        self.check_limits_with_fail_fn(allowed, true, |_, _, _| within = false);
628        within
629    }

631    /// Compares every limits within self is within the limits given in `allowed`.
632    /// For an easy to use binary choice, use [`Limits::check_limits`].
633    ///
634    /// If a value is not within the allowed limit, this function calls the `fail_fn`
635    /// with the:
636    ///  - limit name
637    ///  - self's limit
638    ///  - allowed's limit.
639    ///
640    /// If fatal is true, a single failure bails out the comparison after a single failure.
641    pub fn check_limits_with_fail_fn(
642        &self,
643        allowed: &Self,
644        fatal: bool,
645        mut fail_fn: impl FnMut(&'static str, u64, u64),
646    ) {
647        macro_rules! check_with_fail_fn {
648            ($name:ident, $ordering:expr) => {
649                let invalid_ord = $ordering.reverse();
650                // In the case of `min_subgroup_size`, requesting a value of
651                // zero means "I'm not going to use subgroups", so we have to
652                // special case that. If any of our minimum limits could
653                // meaningfully go all the way to zero, that would conflict with
654                // this.
655                if self.$name != 0 && self.$name.cmp(&allowed.$name) == invalid_ord {
656                    fail_fn(stringify!($name), self.$name as u64, allowed.$name as u64);
657                    if fatal {
658                        return;
659                    }
660                }
661            };
662        }
663
664        if self.min_subgroup_size > self.max_subgroup_size {
665            fail_fn(
666                "max_subgroup_size",
667                self.min_subgroup_size as u64,
668                allowed.min_subgroup_size as u64,
669            );
670        }
671        with_limits!(check_with_fail_fn);
672    }
673
674    /// For each limit in `other` that is better than the value in `self`,
675    /// replace the value in `self` with the value from `other`.
676    ///
677    /// A request for a limit value less than the WebGPU-specified default must
678    /// be ignored. This function is used to clamp such requests to the default
679    /// value.
680    ///
681    /// This function is not for clamping requests for values beyond the
682    /// supported limits. For that purpose the desired function would be
683    /// `or_worse_values_from` (which doesn't exist, but could be added if
684    /// needed).
685    #[must_use]
686    pub fn or_better_values_from(mut self, other: &Self) -> Self {
687        macro_rules! or_better_value_from {
688            ($name:ident, $ordering:expr) => {
689                match $ordering {
690                    // Limits that are maximum values (most of them)
691                    Ordering::Less => self.$name = self.$name.max(other.$name),
692                    // Limits that are minimum values
693                    Ordering::Greater => self.$name = self.$name.min(other.$name),
694                    Ordering::Equal => unreachable!(),
695                }
696            };
697        }
698
699        with_limits!(or_better_value_from);
700
701        self
702    }
703}

742    ///

753bitflags::bitflags! {
754    /// Binary flags listing features that may or may not be present on downlevel adapters.
755    ///
756    /// A downlevel adapter is a GPU adapter that wgpu supports, but with potentially limited
757    /// features, due to the lack of hardware feature support.
758    ///
759    /// Flags that are **not** present for a downlevel adapter or device usually indicates
760    /// non-compliance with the WebGPU specification, but not always.
761    ///
762    /// You can check whether a set of flags is compliant through the
763    /// [`DownlevelCapabilities::is_webgpu_compliant()`] function.
764    #[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
765    #[cfg_attr(feature = "serde", serde(transparent))]
766    #[derive(Debug, Copy, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
767    pub struct DownlevelFlags: u32 {
768        /// The device supports compiling and using compute shaders.
769        ///
770        /// WebGL2, and GLES3.0 devices do not support compute.
771        const COMPUTE_SHADERS = 1 << 0;
772        /// Supports binding storage buffers and textures to fragment shaders.
773        const FRAGMENT_WRITABLE_STORAGE = 1 << 1;
774        /// Supports indirect drawing and dispatching.
775        ///
776        /// [`Self::COMPUTE_SHADERS`] must be present for this flag.
777        ///
778        /// WebGL2, GLES 3.0, and Metal on Apple1/Apple2 GPUs do not support indirect.
779        const INDIRECT_EXECUTION = 1 << 2;
780        /// Supports non-zero `base_vertex` parameter to direct indexed draw calls.
781        ///
782        /// Indirect calls, if supported, always support non-zero `base_vertex`.
783        ///
784        /// Supported by:
785        /// - Vulkan
786        /// - DX12
787        /// - Metal on Apple3+ or Mac1+
788        /// - OpenGL 3.2+
789        /// - OpenGL ES 3.2
790        const BASE_VERTEX = 1 << 3;
791        /// Supports reading from a depth/stencil texture while using it as a read-only
792        /// depth/stencil attachment.
793        ///
794        /// The WebGL2 and GLES backends do not support RODS.
795        const READ_ONLY_DEPTH_STENCIL = 1 << 4;
796        /// Supports textures with mipmaps which have a non power of two size.
797        const NON_POWER_OF_TWO_MIPMAPPED_TEXTURES = 1 << 5;
798        /// Supports textures that are cube arrays.
799        const CUBE_ARRAY_TEXTURES = 1 << 6;
800        /// Supports comparison samplers.
801        const COMPARISON_SAMPLERS = 1 << 7;
802        /// Supports different blend operations per color attachment.
803        const INDEPENDENT_BLEND = 1 << 8;
804        /// Supports storage buffers in vertex shaders.
805        const VERTEX_STORAGE = 1 << 9;
806
807        /// Supports samplers with anisotropic filtering. Note this isn't actually required by
808        /// WebGPU, the implementation is allowed to completely ignore aniso clamp. This flag is
809        /// here for native backends so they can communicate to the user of aniso is enabled.
810        ///
811        /// All backends and all devices support anisotropic filtering.
812        const ANISOTROPIC_FILTERING = 1 << 10;
813
814        /// Supports storage buffers in fragment shaders.
815        const FRAGMENT_STORAGE = 1 << 11;
816
817        /// Supports sample-rate shading.
818        const MULTISAMPLED_SHADING = 1 << 12;
819
820        /// Supports copies between depth textures and buffers.
821        ///
822        /// GLES/WebGL don't support this.
823        const DEPTH_TEXTURE_AND_BUFFER_COPIES = 1 << 13;
824
825        /// Supports all the texture usages described in WebGPU. If this isn't supported, you
826        /// should call `get_texture_format_features` to get how you can use textures of a given format
827        const WEBGPU_TEXTURE_FORMAT_SUPPORT = 1 << 14;
828
829        /// Supports buffer bindings with sizes that aren't a multiple of 16.
830        ///
831        /// WebGL doesn't support this.
832        const BUFFER_BINDINGS_NOT_16_BYTE_ALIGNED = 1 << 15;
833
834        /// Supports buffers to combine [`BufferUsages::INDEX`] with usages other than [`BufferUsages::COPY_DST`] and [`BufferUsages::COPY_SRC`].
835        /// Furthermore, in absence of this feature it is not allowed to copy index buffers from/to buffers with a set of usage flags containing
836        /// [`BufferUsages::VERTEX`]/[`BufferUsages::UNIFORM`]/[`BufferUsages::STORAGE`] or [`BufferUsages::INDIRECT`].
837        ///
838        /// WebGL doesn't support this.
839        const UNRESTRICTED_INDEX_BUFFER = 1 << 16;
840
841        /// Supports full 32-bit range indices (2^32-1 as opposed to 2^24-1 without this flag)
842        ///
843        /// Corresponds to Vulkan's `VkPhysicalDeviceFeatures.fullDrawIndexUint32`
844        const FULL_DRAW_INDEX_UINT32 = 1 << 17;
845
846        /// Supports depth bias clamping
847        ///
848        /// Corresponds to Vulkan's `VkPhysicalDeviceFeatures.depthBiasClamp`
849        const DEPTH_BIAS_CLAMP = 1 << 18;
850
851        /// Supports specifying which view format values are allowed when create_view() is called on a texture.
852        ///
853        /// The WebGL and GLES backends doesn't support this.
854        const VIEW_FORMATS = 1 << 19;
855
856        /// With this feature not present, there are the following restrictions on `Queue::copy_external_image_to_texture`:
857        /// - The source must not be [`web_sys::OffscreenCanvas`]
858        /// - [`CopyExternalImageSourceInfo::origin`] must be zero.
859        /// - [`CopyExternalImageDestInfo::color_space`] must be srgb.
860        /// - If the source is an [`web_sys::ImageBitmap`]:
861        ///   - [`CopyExternalImageSourceInfo::flip_y`] must be false.
862        ///   - [`CopyExternalImageDestInfo::premultiplied_alpha`] must be false.
863        ///
864        /// WebGL doesn't support this. WebGPU does.
865        const UNRESTRICTED_EXTERNAL_TEXTURE_COPIES = 1 << 20;
866
867        /// Supports specifying which view formats are allowed when calling create_view on the texture returned by
868        /// `Surface::get_current_texture`.
869        ///
870        /// The GLES/WebGL and Vulkan on Android doesn't support this.
871        const SURFACE_VIEW_FORMATS = 1 << 21;
872
873        /// If this is true, calls to `CommandEncoder::resolve_query_set` will be performed on the queue timeline.
874        ///
875        /// If this is false, calls to `CommandEncoder::resolve_query_set` will be performed on the device (i.e. cpu) timeline
876        /// and will block that timeline until the query has data. You may work around this limitation by waiting until the submit
877        /// whose queries you are resolving is fully finished (through use of `queue.on_submitted_work_done`) and only
878        /// then submitting the resolve_query_set command. The queries will be guaranteed finished, so will not block.
879        ///
880        /// Supported by:
881        /// - Vulkan,
882        /// - DX12
883        /// - Metal
884        /// - OpenGL 4.4+
885        ///
886        /// Not Supported by:
887        /// - GL ES / WebGL
888        const NONBLOCKING_QUERY_RESOLVE = 1 << 22;
889
890        /// Allows shaders to use `quantizeToF16`, `pack2x16float`, and `unpack2x16float`, which
891        /// operate on `f16`-precision values stored in `f32`s.
892        ///
893        /// Not supported by Vulkan on Mesa when [`Features::SHADER_F16`] is absent.
894        const SHADER_F16_IN_F32 = 1 << 23;
895    }
896}
