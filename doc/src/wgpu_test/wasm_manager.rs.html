7// Called when tests are run for WASM in a browser. Kicks off each
8// test by calling the test server which runs the test using playwright.
9pub fn run_wasm_browser_tests(tests: Vec<GpuTestInitializer>) -> MainResult {
10    let report: GpuReport = serde_json::from_str(
11        &std::fs::read_to_string(concat!(env!("CARGO_MANIFEST_DIR"), "/../.wasmgpuconfig"))
12            .expect("Failed to read .wasmgpuconfig"),
13    )
14    .expect("Failed to parse .wasmgpuconfig");
15
16    let trials = tests
17        .into_iter()
18        .map(|test| {
19            let test = test();
20            let test_info = TestInfo::from_configuration(&test, &report.devices[0]);
21            let backend = report.devices[0].info.backend;
22
23            let full_name = format!(
24                "[wasm] [{running_msg}] [{backend:?}] {base_name}",
25                running_msg = test_info.running_msg,
26                base_name = test.name,
27            );
28
29            libtest_mimic::Trial::test(&full_name, move || {
30                let agent: Agent = Agent::config_builder()
31                    .http_status_as_error(false)
32                    .build()
33                    .into();
34
35                let response = agent
36                    .get("http://127.0.0.1:3000/run_test")
37                    .query("wasm", std::env::var("CARGO_PKG_NAME")?)
38                    .query("name", &test.name)
39                    .call();
40
41                handle_response(response)
42            })
43        })
44        .collect::<Vec<_>>();
45
46    let args = libtest_mimic::Arguments::from_args();
47    libtest_mimic::run(&args, trials).exit_if_failed();
48
49    Ok(())
50}
51
52fn handle_response(
53    response: Result<Response<Body>, ureq::Error>,
54) -> Result<(), libtest_mimic::Failed> {
55    let mut response = response?;
56    if response.status() != 200 {
57        Err(response.body_mut().read_to_string()?.into())
58    } else {
59        Ok(())
60    }
61}
