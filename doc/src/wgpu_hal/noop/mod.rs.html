262impl crate::Queue for Context {
263    type A = Api;
264
265    unsafe fn submit(
266        &self,
267        command_buffers: &[&CommandBuffer],
268        surface_textures: &[&Resource],
269        (fence, fence_value): (&mut Fence, crate::FenceValue),
270    ) -> DeviceResult<()> {
271        // All commands are executed synchronously.
272        for cb in command_buffers {
273            // SAFETY: Caller is responsible for ensuring synchronization between commands and
274            // other mutations.
275            unsafe {
276                cb.execute();
277            }
278        }
279        fence.value.store(fence_value, Ordering::Release);
280        Ok(())
281    }
282    unsafe fn present(
283        &self,
284        surface: &Context,
285        texture: Resource,
286    ) -> Result<(), crate::SurfaceError> {
287        Ok(())
288    }

290    unsafe fn get_timestamp_period(&self) -> f32 {
291        1.0
