72impl crate::CopyExtent {
73    pub fn map_extent_to_copy_size(extent: &wgt::Extent3d, dim: wgt::TextureDimension) -> Self {
74        Self {
75            width: extent.width,
76            height: extent.height,
77            depth: match dim {
78                wgt::TextureDimension::D1 | wgt::TextureDimension::D2 => 1,
79                wgt::TextureDimension::D3 => extent.depth_or_array_layers,
80            },
81        }
82    }
83
84    pub fn min(&self, other: &Self) -> Self {
85        Self {
86            width: self.width.min(other.width),
87            height: self.height.min(other.height),
88            depth: self.depth.min(other.depth),
89        }
90    }
91
92    // Get the copy size at a specific mipmap level. This doesn't make most sense,
93    // since the copy extents are provided *for* a mipmap level to start with.
94    // But backends use `CopyExtent` more sparingly, and this piece is shared.
95    pub fn at_mip_level(&self, level: u32) -> Self {
96        Self {
97            width: (self.width >> level).max(1),
98            height: (self.height >> level).max(1),
99            depth: (self.depth >> level).max(1),
100        }
101    }
102}

104impl crate::TextureCopyBase {
105    pub fn max_copy_size(&self, full_size: &crate::CopyExtent) -> crate::CopyExtent {
106        let mip = full_size.at_mip_level(self.mip_level);
107        crate::CopyExtent {
108            width: mip.width - self.origin.x,
109            height: mip.height - self.origin.y,
110            depth: mip.depth - self.origin.z,
111        }
112    }

124        &mut self,

127    ) {
128        let max_src_size = self.src_base.max_copy_size(full_src_size);

134/// Clamp the limits in `limits` to honor HAL-imposed maximums and WebGPU
135/// alignment requirements.
136///
137/// Other limits are left unchanged.
138#[cfg_attr(not(any_backend), allow(dead_code))]
139pub(crate) fn apply_hal_limits(mut limits: wgt::Limits) -> wgt::Limits {
140    // The Metal backend maintains two copies of many limit values (one as
141    // `wgt::Limits` and one as `metal::PrivateCapabilities`). In order to avoid
142    // confusing discrepancies between the two, some of the logic here is
143    // duplicated in the initialization of `metal::PrivateCapabilities`.
144    // See <https://github.com/gfx-rs/wgpu/issues/8715>.
145
146    limits.max_bind_groups = limits.max_bind_groups.min(crate::MAX_BIND_GROUPS as u32);
147    limits.max_vertex_buffers = limits
148        .max_vertex_buffers
149        .min(crate::MAX_VERTEX_BUFFERS as u32);
150    limits.max_color_attachments = limits
151        .max_color_attachments
152        .min(crate::MAX_COLOR_ATTACHMENTS as u32);
153
154    // Vulkan and DX12 sometimes report large values for the "max resources per stage"
155    // limits. WebGPU requires that consuming the maximum number of resources of every type
156    // in every stage cannot exceed `max_bindings_per_bind_group`. To ensure this, we clamp
157    // per-stage limit at an equal division of `max_bindings_per_bind_group`.
158    //
159    // It is not expected that this clamping adjusts limits that are in a range where an
160    // application might reach them. (The main dilemma here being how WebGPU will eventually
161    // apply these or other limits to bindless. wgpu has separate `binding_array` limits for
162    // its native-only extension; those are not touched here.)
163    //
164    // See <https://gpuweb.github.io/gpuweb/#adapter-capability-guarantees>.
165    const BINDING_TYPE_AND_STAGE_COUNT: u32 = 10;
166    let max_bindings = limits.max_bindings_per_bind_group / BINDING_TYPE_AND_STAGE_COUNT;
167    let mut did_clamp = false;
168    let mut clamp = |limit: &mut u32| {
169        if *limit > max_bindings {
170            *limit = max_bindings;
171            did_clamp = true;
172        }
173    };
174    clamp(&mut limits.max_dynamic_uniform_buffers_per_pipeline_layout);
175    clamp(&mut limits.max_dynamic_storage_buffers_per_pipeline_layout);
176    clamp(&mut limits.max_sampled_textures_per_shader_stage);
177    clamp(&mut limits.max_samplers_per_shader_stage);
178    clamp(&mut limits.max_storage_buffers_per_shader_stage);
179    clamp(&mut limits.max_storage_textures_per_shader_stage);
180    clamp(&mut limits.max_uniform_buffers_per_shader_stage);
181    if did_clamp
182        && limits.max_bindings_per_bind_group < wgt::Limits::defaults().max_bindings_per_bind_group
183    {
184        // This is the "adjusted limits that are in a range where an application might reach
185        // them" case. (In reality, even something quite a bit less than the default
186        // `max_bindings_per_bind_group` of 1000 ought to be sufficient.)
187        log::warn!(
188            "Unexpected adjustment of per-stage resource limits to fit within max_bindings_per_bind_group."
189        );
190    }
191
192    // Round some limits down to the WebGPU alignment requirement, to avoid
193    // suggesting values that won't work. (In particular, the CTS queries limits
194    // and then tests the exact limit value.)
195    limits.max_storage_buffer_binding_size &= !(wgt::STORAGE_BINDING_SIZE_ALIGNMENT - 1);
196    limits.max_vertex_buffer_array_stride &= !(wgt::VERTEX_ALIGNMENT as u32 - 1);
197
198    limits
