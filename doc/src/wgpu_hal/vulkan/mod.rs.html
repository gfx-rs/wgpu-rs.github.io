1/*!
2# Vulkan API internals.
3
4## Stack memory
5
6Ash expects slices, which we don't generally have available.
7We cope with this requirement by the combination of the following ways:
8  - temporarily allocating `Vec` on heap, where overhead is permitted
9  - growing temporary local storage
10
11## Framebuffers and Render passes
12
13Render passes are cached on the device and kept forever.
14
15Framebuffers are also cached on the device, but they are removed when
16any of the image views (they have) gets removed.
17If Vulkan supports image-less framebuffers,
18then the actual views are excluded from the framebuffer key.
19
20## Fences
21
22If timeline semaphores are available, they are used 1:1 with wgpu-hal fences.
23Otherwise, we manage a pool of `VkFence` objects behind each `hal::Fence`.
24
25!*/

183/// Semaphore used to acquire a swapchain image.
184#[derive(Debug)]
185struct SwapchainAcquireSemaphore {
186    /// A semaphore that is signaled when this image is safe for us to modify.
187    ///
188    /// When [`vkAcquireNextImageKHR`] returns the index of the next swapchain
189    /// image that we should use, that image may actually still be in use by the
190    /// presentation engine, and is not yet safe to modify. However, that
191    /// function does accept a semaphore that it will signal when the image is
192    /// indeed safe to begin messing with.
193    ///
194    /// This semaphore is:
195    ///
196    /// - waited for by the first queue submission to operate on this image
197    ///   since it was acquired, and
198    ///
199    /// - signaled by [`vkAcquireNextImageKHR`] when the acquired image is ready
200    ///   for us to use.
201    ///
202    /// [`vkAcquireNextImageKHR`]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#vkAcquireNextImageKHR
203    acquire: vk::Semaphore,
204
205    /// True if the next command submission operating on this image should wait
206    /// for [`acquire`].
207    ///
208    /// We must wait for `acquire` before drawing to this swapchain image, but
209    /// because `wgpu-hal` queue submissions are always strongly ordered, only
210    /// the first submission that works with a swapchain image actually needs to
211    /// wait. We set this flag when this image is acquired, and clear it the
212    /// first time it's passed to [`Queue::submit`] as a surface texture.
213    ///
214    /// Additionally, semaphores can only be waited on once, so we need to ensure
215    /// that we only actually pass this semaphore to the first submission that
216    /// uses that image.
217    ///
218    /// [`acquire`]: SwapchainAcquireSemaphore::acquire
219    /// [`Queue::submit`]: crate::Queue::submit
220    should_wait_for_acquire: bool,
221
222    /// The fence value of the last command submission that wrote to this image.
223    ///
224    /// The next time we try to acquire this image, we'll block until
225    /// this submission finishes, proving that [`acquire`] is ready to
226    /// pass to `vkAcquireNextImageKHR` again.
227    ///
228    /// [`acquire`]: SwapchainAcquireSemaphore::acquire
229    previously_used_submission_index: crate::FenceValue,
230}
231
232impl SwapchainAcquireSemaphore {
233    fn new(device: &DeviceShared, index: usize) -> Result<Self, crate::DeviceError> {
234        Ok(Self {
235            acquire: device
236                .new_binary_semaphore(&format!("SwapchainImageSemaphore: Index {index} acquire"))?,
237            should_wait_for_acquire: true,
238            previously_used_submission_index: 0,
239        })
240    }
241
242    /// Sets the fence value which the next acquire will wait for. This prevents
243    /// the semaphore from being used while the previous submission is still in flight.
244    fn set_used_fence_value(&mut self, value: crate::FenceValue) {
245        self.previously_used_submission_index = value;
246    }
247
248    /// Return the semaphore that commands drawing to this image should wait for, if any.
249    ///
250    /// This only returns `Some` once per acquisition; see
251    /// [`SwapchainAcquireSemaphore::should_wait_for_acquire`] for details.
252    fn get_acquire_wait_semaphore(&mut self) -> Option<vk::Semaphore> {
253        if self.should_wait_for_acquire {
254            self.should_wait_for_acquire = false;
255            Some(self.acquire)
256        } else {
257            None
258        }
259    }
260
261    /// Indicates the cpu-side usage of this semaphore has finished for the frame,
262    /// so reset internal state to be ready for the next frame.
263    fn end_semaphore_usage(&mut self) {
264        // Reset the acquire semaphore, so that the next time we acquire this
265        // image, we can wait for it again.
266        self.should_wait_for_acquire = true;
267    }
268
269    unsafe fn destroy(&self, device: &ash::Device) {
270        unsafe {
271            device.destroy_semaphore(self.acquire, None);
272        }
273    }
274}

276#[derive(Debug)]
277struct SwapchainPresentSemaphores {
278    /// A pool of semaphores for ordering presentation after drawing.
279    ///
280    /// The first [`present_index`] semaphores in this vector are:
281    ///
282    /// - all waited on by the call to [`vkQueuePresentKHR`] that presents this
283    ///   image, and
284    ///
285    /// - each signaled by some [`vkQueueSubmit`] queue submission that draws to
286    ///   this image, when the submission finishes execution.
287    ///
288    /// This vector accumulates one semaphore per submission that writes to this
289    /// image. This is awkward, but hard to avoid: [`vkQueuePresentKHR`]
290    /// requires a semaphore to order it with respect to drawing commands, and
291    /// we can't attach new completion semaphores to a command submission after
292    /// it's been submitted. This means that, at submission time, we must create
293    /// the semaphore we might need if the caller's next action is to enqueue a
294    /// presentation of this image.
295    ///
296    /// An alternative strategy would be for presentation to enqueue an empty
297    /// submit, ordered relative to other submits in the usual way, and
298    /// signaling a single presentation semaphore. But we suspect that submits
299    /// are usually expensive enough, and semaphores usually cheap enough, that
300    /// performance-sensitive users will avoid making many submits, so that the
301    /// cost of accumulated semaphores will usually be less than the cost of an
302    /// additional submit.
303    ///
304    /// Only the first [`present_index`] semaphores in the vector are actually
305    /// going to be signalled by submitted commands, and need to be waited for
306    /// by the next present call. Any semaphores beyond that index were created
307    /// for prior presents and are simply being retained for recycling.
308    ///
309    /// [`present_index`]: SwapchainPresentSemaphores::present_index
310    /// [`vkQueuePresentKHR`]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#vkQueuePresentKHR
311    /// [`vkQueueSubmit`]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#vkQueueSubmit
312    present: Vec<vk::Semaphore>,
313
314    /// The number of semaphores in [`present`] to be signalled for this submission.
315    ///
316    /// [`present`]: SwapchainPresentSemaphores::present
317    present_index: usize,
318
319    /// Which image this semaphore set is used for.
320    frame_index: usize,
321}
322
323impl SwapchainPresentSemaphores {
324    pub fn new(frame_index: usize) -> Self {
325        Self {
326            present: Vec::new(),
327            present_index: 0,
328            frame_index,
329        }
330    }
331
332    /// Return the semaphore that the next submission that writes to this image should
333    /// signal when it's done.
334    ///
335    /// See [`SwapchainPresentSemaphores::present`] for details.
336    fn get_submit_signal_semaphore(
337        &mut self,
338        device: &DeviceShared,
339    ) -> Result<vk::Semaphore, crate::DeviceError> {
340        // Try to recycle a semaphore we created for a previous presentation.
341        let sem = match self.present.get(self.present_index) {
342            Some(sem) => *sem,
343            None => {
344                let sem = device.new_binary_semaphore(&format!(
345                    "SwapchainImageSemaphore: Image {} present semaphore {}",
346                    self.frame_index, self.present_index
347                ))?;
348                self.present.push(sem);
349                sem
350            }
351        };
352
353        self.present_index += 1;
354
355        Ok(sem)
356    }
357
358    /// Indicates the cpu-side usage of this semaphore has finished for the frame,
359    /// so reset internal state to be ready for the next frame.
360    fn end_semaphore_usage(&mut self) {
361        // Reset the index to 0, so that the next time we get a semaphore, we
362        // start from the beginning of the list.
363        self.present_index = 0;
364    }
365
366    /// Return the semaphores that a presentation of this image should wait on.
367    ///
368    /// Return a slice of semaphores that the call to [`vkQueueSubmit`] that
369    /// ends this image's acquisition should wait for. See
370    /// [`SwapchainPresentSemaphores::present`] for details.
371    ///
372    /// Reset `self` to be ready for the next acquisition cycle.
373    ///
374    /// [`vkQueueSubmit`]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#vkQueueSubmit
375    fn get_present_wait_semaphores(&mut self) -> Vec<vk::Semaphore> {
376        self.present[0..self.present_index].to_vec()
377    }
378
379    unsafe fn destroy(&self, device: &ash::Device) {
380        unsafe {
381            for sem in &self.present {
382                device.destroy_semaphore(*sem, None);
383            }
384        }
385    }
386}

388struct Swapchain {
389    raw: vk::SwapchainKHR,
390    functor: khr::swapchain::Device,
391    device: Arc<DeviceShared>,
392    images: Vec<vk::Image>,
393    config: crate::SurfaceConfiguration,
394
395    /// Semaphores used between image acquisition and the first submission
396    /// that uses that image. This is indexed using [`next_acquire_index`].
397    ///
398    /// Because we need to provide this to [`vkAcquireNextImageKHR`], we haven't
399    /// received the swapchain image index for the frame yet, so we cannot use
400    /// that to index it.
401    ///
402    /// Before we pass this to [`vkAcquireNextImageKHR`], we ensure that we wait on
403    /// the submission indicated by [`previously_used_submission_index`]. This enusres
404    /// the semaphore is no longer in use before we use it.
405    ///
406    /// [`next_acquire_index`]: Swapchain::next_acquire_index
407    /// [`vkAcquireNextImageKHR`]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#vkAcquireNextImageKHR
408    /// [`previously_used_submission_index`]: SwapchainAcquireSemaphore::previously_used_submission_index
409    acquire_semaphores: Vec<Arc<Mutex<SwapchainAcquireSemaphore>>>,
410    /// The index of the next acquire semaphore to use.
411    ///
412    /// This is incremented each time we acquire a new image, and wraps around
413    /// to 0 when it reaches the end of [`acquire_semaphores`].
414    ///
415    /// [`acquire_semaphores`]: Swapchain::acquire_semaphores
416    next_acquire_index: usize,
417
418    /// Semaphore sets used between all submissions that write to an image and
419    /// the presentation of that image.
420    ///
421    /// This is indexed by the swapchain image index returned by
422    /// [`vkAcquireNextImageKHR`].
423    ///
424    /// We know it is safe to use these semaphores because use them
425    /// _after_ the acquire semaphore. Because the acquire semaphore
426    /// has been signaled, the previous presentation using that image
427    /// is known-finished, so this semaphore is no longer in use.
428    ///
429    /// [`vkAcquireNextImageKHR`]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#vkAcquireNextImageKHR
430    present_semaphores: Vec<Arc<Mutex<SwapchainPresentSemaphores>>>,
431
432    /// The present timing information which will be set in the next call to [`present()`](crate::Queue::present()).
433    ///
434    /// # Safety
435    ///
436    /// This must only be set if [`wgt::Features::VULKAN_GOOGLE_DISPLAY_TIMING`] is enabled, and
437    /// so the VK_GOOGLE_display_timing extension is present.
438    next_present_time: Option<vk::PresentTimeGOOGLE>,
439}

441impl Swapchain {
442    /// Mark the current frame finished, advancing to the next acquire semaphore.
443    fn advance_acquire_semaphore(&mut self) {
444        let semaphore_count = self.acquire_semaphores.len();
445        self.next_acquire_index = (self.next_acquire_index + 1) % semaphore_count;
446    }
447
448    /// Get the next acquire semaphore that should be used with this swapchain.
449    fn get_acquire_semaphore(&self) -> Arc<Mutex<SwapchainAcquireSemaphore>> {
450        self.acquire_semaphores[self.next_acquire_index].clone()
451    }
452
453    /// Get the set of present semaphores that should be used with the given image index.
454    fn get_present_semaphores(&self, index: u32) -> Arc<Mutex<SwapchainPresentSemaphores>> {
455        self.present_semaphores[index as usize].clone()
456    }
457}

482    /// This can be used to give an id to presentations, for future use of [`vk::PastPresentationTimingGOOGLE`].
483    /// Note that `wgpu-hal` does *not* provide a way to use that API - you should manually access this through [`ash`].
484    ///
485    /// This can also be used to add a "not before" timestamp to the presentation.
486    ///
487    /// The exact semantics of the fields are also documented in the [specification](https://registry.khronos.org/vulkan/specs/1.3-extensions/man/html/VkPresentTimeGOOGLE.html) for the extension.
488    ///
489    /// # Panics
490    ///
491    /// - If the surface hasn't been configured.
492    /// - If the device doesn't [support present timing](wgt::Features::VULKAN_GOOGLE_DISPLAY_TIMING).
493    ///
494    /// [VK_GOOGLE_display_timing]: https://registry.khronos.org/vulkan/specs/1.3-extensions/man/html/VK_GOOGLE_display_timing.html
495    #[track_caller]
496    pub fn set_next_present_time(&self, present_timing: vk::PresentTimeGOOGLE) {
497        let mut swapchain = self.swapchain.write();
498        let swapchain = swapchain
499            .as_mut()
500            .expect("Surface should have been configured");
501        let features = wgt::Features::VULKAN_GOOGLE_DISPLAY_TIMING;
502        if swapchain.device.features.contains(features) {
503            swapchain.next_present_time = Some(present_timing);
504        } else {
505            // Ideally we'd use something like `device.required_features` here, but that's in `wgpu-core`, which we are a dependency of
506            panic!(
507                concat!(
508                    "Tried to set display timing properties ",
509                    "without the corresponding feature ({:?}) enabled."
510                ),
511                features
512            );
513        }
514    }

778/// Semaphores for forcing queue submissions to run in order.
779///
780/// The [`wgpu_hal::Queue`] trait promises that if two calls to [`submit`] are
781/// ordered, then the first submission will finish on the GPU before the second
782/// submission begins. To get this behavior on Vulkan we need to pass semaphores
783/// to [`vkQueueSubmit`] for the commands to wait on before beginning execution,
784/// and to signal when their execution is done.
785///
786/// Normally this can be done with a single semaphore, waited on and then
787/// signalled for each submission. At any given time there's exactly one
788/// submission that would signal the semaphore, and exactly one waiting on it,
789/// as Vulkan requires.
790///
791/// However, as of Oct 2021, bug [#5508] in the Mesa ANV drivers caused them to
792/// hang if we use a single semaphore. The workaround is to alternate between
793/// two semaphores. The bug has been fixed in Mesa, but we should probably keep
794/// the workaround until, say, Oct 2026.
795///
796/// [`wgpu_hal::Queue`]: crate::Queue
797/// [`submit`]: crate::Queue::submit
798/// [`vkQueueSubmit`]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#vkQueueSubmit
799/// [#5508]: https://gitlab.freedesktop.org/mesa/mesa/-/issues/5508
800#[derive(Clone)]
801struct RelaySemaphores {
802    /// The semaphore the next submission should wait on before beginning
803    /// execution on the GPU. This is `None` for the first submission, which
804    /// should not wait on anything at all.
805    wait: Option<vk::Semaphore>,
806
807    /// The semaphore the next submission should signal when it has finished
808    /// execution on the GPU.
809    signal: vk::Semaphore,
810}
811
812impl RelaySemaphores {
813    fn new(device: &DeviceShared) -> Result<Self, crate::DeviceError> {
814        Ok(Self {
815            wait: None,
816            signal: device.new_binary_semaphore("RelaySemaphores: 1")?,
817        })
818    }
819
820    /// Advances the semaphores, returning the semaphores that should be used for a submission.
821    fn advance(&mut self, device: &DeviceShared) -> Result<Self, crate::DeviceError> {
822        let old = self.clone();
823
824        // Build the state for the next submission.
825        match self.wait {
826            None => {
827                // The `old` values describe the first submission to this queue.
828                // The second submission should wait on `old.signal`, and then
829                // signal a new semaphore which we'll create now.
830                self.wait = Some(old.signal);
831                self.signal = device.new_binary_semaphore("RelaySemaphores: 2")?;
832            }
833            Some(ref mut wait) => {
834                // What this submission signals, the next should wait.
835                mem::swap(wait, &mut self.signal);
836            }
837        };
838
839        Ok(old)
840    }
841
842    /// Destroys the semaphores.
843    unsafe fn destroy(&self, device: &ash::Device) {
844        unsafe {
845            if let Some(wait) = self.wait {
846                device.destroy_semaphore(wait, None);
847            }
848            device.destroy_semaphore(self.signal, None);
849        }
850    }
851}

1085///

1087/// provides a way to generate unique IDs for each resource.
1088struct ResourceIdentityFactory<T> {
1089    #[cfg(not(target_has_atomic = "64"))]
1090    next_id: Mutex<u64>,
1091    #[cfg(target_has_atomic = "64")]
1092    next_id: core::sync::atomic::AtomicU64,
1093    _phantom: PhantomData<T>,
1094}

1096impl<T> ResourceIdentityFactory<T> {
1097    fn new() -> Self {
1098        Self {
1099            #[cfg(not(target_has_atomic = "64"))]
1100            next_id: Mutex::new(0),
1101            #[cfg(target_has_atomic = "64")]
1102            next_id: core::sync::atomic::AtomicU64::new(0),
1103            _phantom: PhantomData,
1104        }
1105    }

1107    /// Returns a new unique ID for a resource of type `T`.
1108    fn next(&self) -> ResourceIdentity<T> {
1109        #[cfg(not(target_has_atomic = "64"))]
1110        {
1111            let mut next_id = self.next_id.lock();
1112            let id = *next_id;
1113            *next_id += 1;
1114            ResourceIdentity {
1115                id,
1116                _phantom: PhantomData,
1117            }
1118        }
1119
1120        #[cfg(target_has_atomic = "64")]
1121        ResourceIdentity {
1122            id: self
1123                .next_id
1124                .fetch_add(1, core::sync::atomic::Ordering::Relaxed),
1125            _phantom: PhantomData,
1126        }
1127    }

1317/// The [`Api::Fence`] type for [`vulkan::Api`].
1318///
1319/// This is an `enum` because there are two possible implementations of
1320/// `wgpu-hal` fences on Vulkan: Vulkan fences, which work on any version of
1321/// Vulkan, and Vulkan timeline semaphores, which are easier and cheaper but
1322/// require non-1.0 features.
1323///
1324/// [`Device::create_fence`] returns a [`TimelineSemaphore`] if
1325/// [`VK_KHR_timeline_semaphore`] is available and enabled, and a [`FencePool`]
1326/// otherwise.
1327///
1328/// [`Api::Fence`]: crate::Api::Fence
1329/// [`vulkan::Api`]: Api
1330/// [`Device::create_fence`]: crate::Device::create_fence
1331/// [`TimelineSemaphore`]: Fence::TimelineSemaphore
1332/// [`VK_KHR_timeline_semaphore`]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#VK_KHR_timeline_semaphore
1333/// [`FencePool`]: Fence::FencePool
1334#[derive(Debug)]
1335pub enum Fence {
1336    /// A Vulkan [timeline semaphore].
1337    ///
1338    /// These are simpler to use than Vulkan fences, since timeline semaphores
1339    /// work exactly the way [`wpgu_hal::Api::Fence`] is specified to work.
1340    ///
1341    /// [timeline semaphore]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#synchronization-semaphores
1342    /// [`wpgu_hal::Api::Fence`]: crate::Api::Fence
1343    TimelineSemaphore(vk::Semaphore),
1344
1345    /// A collection of Vulkan [fence]s, each associated with a [`FenceValue`].
1346    ///
1347    /// The effective [`FenceValue`] of this variant is the greater of
1348    /// `last_completed` and the maximum value associated with a signalled fence
1349    /// in `active`.
1350    ///
1351    /// Fences are available in all versions of Vulkan, but since they only have
1352    /// two states, "signaled" and "unsignaled", we need to use a separate fence
1353    /// for each queue submission we might want to wait for, and remember which
1354    /// [`FenceValue`] each one represents.
1355    ///
1356    /// [fence]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#synchronization-fences
1357    /// [`FenceValue`]: crate::FenceValue
1358    FencePool {
1359        last_completed: crate::FenceValue,
1360        /// The pending fence values have to be ascending.
1361        active: Vec<(crate::FenceValue, vk::Fence)>,
1362        free: Vec<vk::Fence>,
1363    },
1364}

1368impl Fence {
1369    /// Return the highest [`FenceValue`] among the signalled fences in `active`.
1370    ///
1371    /// As an optimization, assume that we already know that the fence has
1372    /// reached `last_completed`, and don't bother checking fences whose values
1373    /// are less than that: those fences remain in the `active` array only
1374    /// because we haven't called `maintain` yet to clean them up.
1375    ///
1376    /// [`FenceValue`]: crate::FenceValue
1377    fn check_active(
1378        device: &ash::Device,
1379        mut last_completed: crate::FenceValue,
1380        active: &[(crate::FenceValue, vk::Fence)],
1381    ) -> Result<crate::FenceValue, crate::DeviceError> {
1382        for &(value, raw) in active.iter() {
1383            unsafe {
1384                if value > last_completed
1385                    && device
1386                        .get_fence_status(raw)
1387                        .map_err(map_host_device_oom_and_lost_err)?
1388                {
1389                    last_completed = value;
1390                }
1391            }
1392        }
1393        Ok(last_completed)
1394    }
1395
1396    /// Return the highest signalled [`FenceValue`] for `self`.
1397    ///
1398    /// [`FenceValue`]: crate::FenceValue
1399    fn get_latest(
1400        &self,
1401        device: &ash::Device,
1402        extension: Option<&ExtensionFn<khr::timeline_semaphore::Device>>,
1403    ) -> Result<crate::FenceValue, crate::DeviceError> {
1404        match *self {
1405            Self::TimelineSemaphore(raw) => unsafe {
1406                Ok(match *extension.unwrap() {
1407                    ExtensionFn::Extension(ref ext) => ext
1408                        .get_semaphore_counter_value(raw)
1409                        .map_err(map_host_device_oom_and_lost_err)?,
1410                    ExtensionFn::Promoted => device
1411                        .get_semaphore_counter_value(raw)
1412                        .map_err(map_host_device_oom_and_lost_err)?,
1413                })
1414            },
1415            Self::FencePool {
1416                last_completed,
1417                ref active,
1418                free: _,
1419            } => Self::check_active(device, last_completed, active),
1420        }
1421    }
1422
1423    /// Trim the internal state of this [`Fence`].
1424    ///
1425    /// This function has no externally visible effect, but you should call it
1426    /// periodically to keep this fence's resource consumption under control.
1427    ///
1428    /// For fences using the [`FencePool`] implementation, this function
1429    /// recycles fences that have been signaled. If you don't call this,
1430    /// [`Queue::submit`] will just keep allocating a new Vulkan fence every
1431    /// time it's called.
1432    ///
1433    /// [`FencePool`]: Fence::FencePool
1434    /// [`Queue::submit`]: crate::Queue::submit
1435    fn maintain(&mut self, device: &ash::Device) -> Result<(), crate::DeviceError> {
1436        match *self {
1437            Self::TimelineSemaphore(_) => {}
1438            Self::FencePool {
1439                ref mut last_completed,
1440                ref mut active,
1441                ref mut free,
1442            } => {
1443                let latest = Self::check_active(device, *last_completed, active)?;
1444                let base_free = free.len();
1445                for &(value, raw) in active.iter() {
1446                    if value <= latest {
1447                        free.push(raw);
1448                    }
1449                }
1450                if free.len() != base_free {
1451                    active.retain(|&(value, _)| value > latest);
1452                    unsafe { device.reset_fences(&free[base_free..]) }
1453                        .map_err(map_device_oom_err)?
1454                }
1455                *last_completed = latest;
1456            }
1457        }
1458        Ok(())
1459    }
1460}

1465    unsafe fn submit(
1466        &self,
1467        command_buffers: &[&CommandBuffer],
1468        surface_textures: &[&SurfaceTexture],
1469        (signal_fence, signal_value): (&mut Fence, crate::FenceValue),
1470    ) -> Result<(), crate::DeviceError> {
1471        let mut fence_raw = vk::Fence::null();
1472
1473        let mut wait_stage_masks = Vec::new();
1474        let mut wait_semaphores = Vec::new();
1475        let mut signal_semaphores = SemaphoreList::default();
1476
1477        // Double check that the same swapchain image isn't being given to us multiple times,
1478        // as that will deadlock when we try to lock them all.
1479        debug_assert!(
1480            {
1481                let mut check = HashSet::with_capacity(surface_textures.len());
1482                // We compare the Arcs by pointer, as Eq isn't well defined for SurfaceSemaphores.
1483                for st in surface_textures {
1484                    check.insert(Arc::as_ptr(&st.acquire_semaphores) as usize);
1485                    check.insert(Arc::as_ptr(&st.present_semaphores) as usize);
1486                }
1487                check.len() == surface_textures.len() * 2
1488            },
1489            "More than one surface texture is being used from the same swapchain. This will cause a deadlock in release."
1490        );
1491
1492        let locked_swapchain_semaphores = surface_textures
1493            .iter()
1494            .map(|st| {
1495                let acquire = st
1496                    .acquire_semaphores
1497                    .try_lock()
1498                    .expect("Failed to lock surface acquire semaphore");
1499                let present = st
1500                    .present_semaphores
1501                    .try_lock()
1502                    .expect("Failed to lock surface present semaphore");
1503
1504                (acquire, present)
1505            })
1506            .collect::<Vec<_>>();
1507
1508        for (mut acquire_semaphore, mut present_semaphores) in locked_swapchain_semaphores {
1509            acquire_semaphore.set_used_fence_value(signal_value);
1510
1511            // If we're the first submission to operate on this image, wait on
1512            // its acquire semaphore, to make sure the presentation engine is
1513            // done with it.
1514            if let Some(sem) = acquire_semaphore.get_acquire_wait_semaphore() {
1515                wait_stage_masks.push(vk::PipelineStageFlags::TOP_OF_PIPE);
1516                wait_semaphores.push(sem);
1517            }
1518
1519            // Get a semaphore to signal when we're done writing to this surface
1520            // image. Presentation of this image will wait for this.
1521            let signal_semaphore = present_semaphores.get_submit_signal_semaphore(&self.device)?;
1522            signal_semaphores.push_binary(signal_semaphore);
1523        }
1524
1525        let mut guard = self.signal_semaphores.lock();
1526        if !guard.is_empty() {
1527            signal_semaphores.append(&mut guard);
1528        }
1529
1530        // In order for submissions to be strictly ordered, we encode a dependency between each submission
1531        // using a pair of semaphores. This adds a wait if it is needed, and signals the next semaphore.
1532        let semaphore_state = self.relay_semaphores.lock().advance(&self.device)?;
1533
1534        if let Some(sem) = semaphore_state.wait {
1535            wait_stage_masks.push(vk::PipelineStageFlags::TOP_OF_PIPE);
1536            wait_semaphores.push(sem);
1537        }
1538
1539        signal_semaphores.push_binary(semaphore_state.signal);
1540
1541        // We need to signal our wgpu::Fence if we have one, this adds it to the signal list.
1542        signal_fence.maintain(&self.device.raw)?;
1543        match *signal_fence {
1544            Fence::TimelineSemaphore(raw) => {
1545                signal_semaphores.push_timeline(raw, signal_value);
1546            }
1547            Fence::FencePool {
1548                ref mut active,
1549                ref mut free,
1550                ..
1551            } => {
1552                fence_raw = match free.pop() {
1553                    Some(raw) => raw,
1554                    None => unsafe {
1555                        self.device
1556                            .raw
1557                            .create_fence(&vk::FenceCreateInfo::default(), None)
1558                            .map_err(map_host_device_oom_err)?
1559                    },
1560                };
1561                active.push((signal_value, fence_raw));
1562            }
1563        }
1564
1565        let vk_cmd_buffers = command_buffers
1566            .iter()
1567            .map(|cmd| cmd.raw)
1568            .collect::<Vec<_>>();
1569
1570        let mut vk_info = vk::SubmitInfo::default().command_buffers(&vk_cmd_buffers);
1571
1572        vk_info = vk_info
1573            .wait_semaphores(&wait_semaphores)
1574            .wait_dst_stage_mask(&wait_stage_masks);
1575
1576        let mut vk_timeline_info = mem::MaybeUninit::uninit();
1577        vk_info = signal_semaphores.add_to_submit(vk_info, &mut vk_timeline_info);
1578
1579        profiling::scope!("vkQueueSubmit");
1580        unsafe {
1581            self.device
1582                .raw
1583                .queue_submit(self.raw, &[vk_info], fence_raw)
1584                .map_err(map_host_device_oom_and_lost_err)?
1585        };
1586        Ok(())
1587    }
1588
1589    unsafe fn present(
1590        &self,
1591        surface: &Surface,
1592        texture: SurfaceTexture,
1593    ) -> Result<(), crate::SurfaceError> {
1594        let mut swapchain = surface.swapchain.write();
1595        let ssc = swapchain.as_mut().unwrap();
1596        let mut acquire_semaphore = texture.acquire_semaphores.lock();
1597        let mut present_semaphores = texture.present_semaphores.lock();
1598
1599        let wait_semaphores = present_semaphores.get_present_wait_semaphores();
1600
1601        // Reset the acquire and present semaphores internal state
1602        // to be ready for the next frame.
1603        //
1604        // We do this before the actual call to present to ensure that
1605        // even if this method errors and early outs, we have reset
1606        // the state for next frame.
1607        acquire_semaphore.end_semaphore_usage();
1608        present_semaphores.end_semaphore_usage();
1609
1610        drop(acquire_semaphore);
1611
1612        let swapchains = [ssc.raw];
1613        let image_indices = [texture.index];
1614        let vk_info = vk::PresentInfoKHR::default()
1615            .swapchains(&swapchains)
1616            .image_indices(&image_indices)
1617            .wait_semaphores(&wait_semaphores);
1618
1619        let mut display_timing;
1620        let present_times;
1621        let vk_info = if let Some(present_time) = ssc.next_present_time.take() {
1622            debug_assert!(
1623                ssc.device
1624                    .features
1625                    .contains(wgt::Features::VULKAN_GOOGLE_DISPLAY_TIMING),
1626                "`next_present_time` should only be set if `VULKAN_GOOGLE_DISPLAY_TIMING` is enabled"
1627            );
1628            present_times = [present_time];
1629            display_timing = vk::PresentTimesInfoGOOGLE::default().times(&present_times);
1630            // SAFETY: We know that VK_GOOGLE_display_timing is present because of the safety contract on `next_present_time`.
1631            vk_info.push_next(&mut display_timing)
1632        } else {
1633            vk_info
1634        };
1635
1636        let suboptimal = {
1637            profiling::scope!("vkQueuePresentKHR");
1638            unsafe { self.swapchain_fn.queue_present(self.raw, &vk_info) }.map_err(|error| {
1639                match error {
1640                    vk::Result::ERROR_OUT_OF_DATE_KHR => crate::SurfaceError::Outdated,
1641                    vk::Result::ERROR_SURFACE_LOST_KHR => crate::SurfaceError::Lost,
1642                    // We don't use VK_EXT_full_screen_exclusive
1643                    // VK_ERROR_FULL_SCREEN_EXCLUSIVE_MODE_LOST_EXT
1644                    _ => map_host_device_oom_and_lost_err(error).into(),
1645                }
1646            })?
1647        };
1648        if suboptimal {
1649            // We treat `VK_SUBOPTIMAL_KHR` as `VK_SUCCESS` on Android.
1650            // On Android 10+, libvulkan's `vkQueuePresentKHR` implementation returns `VK_SUBOPTIMAL_KHR` if not doing pre-rotation
1651            // (i.e `VkSwapchainCreateInfoKHR::preTransform` not being equal to the current device orientation).
1652            // This is always the case when the device orientation is anything other than the identity one, as we unconditionally use `VK_SURFACE_TRANSFORM_IDENTITY_BIT_KHR`.
1653            #[cfg(not(target_os = "android"))]
1654            log::warn!("Suboptimal present of frame {}", texture.index);
1655        }
1656        Ok(())
1657    }
