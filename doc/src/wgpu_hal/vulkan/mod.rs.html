1/*!
2# Vulkan API internals.
3
4## Stack memory
5
6Ash expects slices, which we don't generally have available.
7We cope with this requirement by the combination of the following ways:
8  - temporarily allocating `Vec` on heap, where overhead is permitted
9  - growing temporary local storage
10
11## Framebuffers and Render passes
12
13Render passes are cached on the device and kept forever.
14
15Framebuffers are also cached on the device, but they are removed when
16any of the image views (they have) gets removed.
17If Vulkan supports image-less framebuffers,
18then the actual views are excluded from the framebuffer key.
19
20## Fences
21
22If timeline semaphores are available, they are used 1:1 with wgpu-hal fences.
23Otherwise, we manage a pool of `VkFence` objects behind each `hal::Fence`.
24
25!*/

515/// Semaphores for forcing queue submissions to run in order.
516///
517/// The [`wgpu_hal::Queue`] trait promises that if two calls to [`submit`] are
518/// ordered, then the first submission will finish on the GPU before the second
519/// submission begins. To get this behavior on Vulkan we need to pass semaphores
520/// to [`vkQueueSubmit`] for the commands to wait on before beginning execution,
521/// and to signal when their execution is done.
522///
523/// Normally this can be done with a single semaphore, waited on and then
524/// signalled for each submission. At any given time there's exactly one
525/// submission that would signal the semaphore, and exactly one waiting on it,
526/// as Vulkan requires.
527///
528/// However, as of Oct 2021, bug [#5508] in the Mesa ANV drivers caused them to
529/// hang if we use a single semaphore. The workaround is to alternate between
530/// two semaphores. The bug has been fixed in Mesa, but we should probably keep
531/// the workaround until, say, Oct 2026.
532///
533/// [`wgpu_hal::Queue`]: crate::Queue
534/// [`submit`]: crate::Queue::submit
535/// [`vkQueueSubmit`]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#vkQueueSubmit
536/// [#5508]: https://gitlab.freedesktop.org/mesa/mesa/-/issues/5508
537#[derive(Clone)]
538struct RelaySemaphores {
539    /// The semaphore the next submission should wait on before beginning
540    /// execution on the GPU. This is `None` for the first submission, which
541    /// should not wait on anything at all.
542    wait: Option<vk::Semaphore>,
543
544    /// The semaphore the next submission should signal when it has finished
545    /// execution on the GPU.
546    signal: vk::Semaphore,
547}
548
549impl RelaySemaphores {
550    fn new(device: &DeviceShared) -> Result<Self, crate::DeviceError> {
551        Ok(Self {
552            wait: None,
553            signal: device.new_binary_semaphore("RelaySemaphores: 1")?,
554        })
555    }
556
557    /// Advances the semaphores, returning the semaphores that should be used for a submission.
558    fn advance(&mut self, device: &DeviceShared) -> Result<Self, crate::DeviceError> {
559        let old = self.clone();
560
561        // Build the state for the next submission.
562        match self.wait {
563            None => {
564                // The `old` values describe the first submission to this queue.
565                // The second submission should wait on `old.signal`, and then
566                // signal a new semaphore which we'll create now.
567                self.wait = Some(old.signal);
568                self.signal = device.new_binary_semaphore("RelaySemaphores: 2")?;
569            }
570            Some(ref mut wait) => {
571                // What this submission signals, the next should wait.
572                mem::swap(wait, &mut self.signal);
573            }
574        };
575
576        Ok(old)
577    }
578
579    /// Destroys the semaphores.
580    unsafe fn destroy(&self, device: &ash::Device) {
581        unsafe {
582            if let Some(wait) = self.wait {
583                device.destroy_semaphore(wait, None);
584            }
585            device.destroy_semaphore(self.signal, None);
586        }
587    }
588}

820/// Generates unique IDs for each resource of type `T`.
821///
822/// Because vk handles are not permanently unique, this
823/// provides a way to generate unique IDs for each resource.
824struct ResourceIdentityFactory<T> {
825    #[cfg(not(target_has_atomic = "64"))]
826    next_id: Mutex<u64>,
827    #[cfg(target_has_atomic = "64")]
828    next_id: core::sync::atomic::AtomicU64,
829    _phantom: PhantomData<T>,
830}
831
832impl<T> ResourceIdentityFactory<T> {
833    fn new() -> Self {
834        Self {
835            #[cfg(not(target_has_atomic = "64"))]
836            next_id: Mutex::new(0),
837            #[cfg(target_has_atomic = "64")]
838            next_id: core::sync::atomic::AtomicU64::new(0),
839            _phantom: PhantomData,
840        }
841    }
842
843    /// Returns a new unique ID for a resource of type `T`.
844    fn next(&self) -> ResourceIdentity<T> {
845        #[cfg(not(target_has_atomic = "64"))]
846        {
847            let mut next_id = self.next_id.lock();
848            let id = *next_id;
849            *next_id += 1;
850            ResourceIdentity {
851                id,
852                _phantom: PhantomData,
853            }
854        }
855
856        #[cfg(target_has_atomic = "64")]
857        ResourceIdentity {
858            id: self
859                .next_id
860                .fetch_add(1, core::sync::atomic::Ordering::Relaxed),
861            _phantom: PhantomData,
862        }
863    }
864}

958impl Drop for CommandEncoder {
959    fn drop(&mut self) {
960        // SAFETY:
961        //
962        // VUID-vkDestroyCommandPool-commandPool-00041: wgpu_hal requires that a
963        // `CommandBuffer` must live until its execution is complete, and that a
964        // `CommandBuffer` must not outlive the `CommandEncoder` that built it.
965        // Thus, we know that none of our `CommandBuffers` are in the "pending"
966        // state.
967        //
968        // The other VUIDs are pretty obvious.
969        unsafe {
970            // `vkDestroyCommandPool` also frees any command buffers allocated
971            // from that pool, so there's no need to explicitly call
972            // `vkFreeCommandBuffers` on `cmd_encoder`'s `free` and `discarded`
973            // fields.
974            self.device.raw.destroy_command_pool(self.raw, None);
975        }
976
977        for (_, fb) in self.framebuffers.drain() {
978            unsafe { self.device.raw.destroy_framebuffer(fb, None) };
979        }
980
981        for (_, view) in self.temp_texture_views.drain() {
982            unsafe { self.device.raw.destroy_image_view(view.raw, None) };
983        }
984
985        self.counters.command_encoders.sub(1);
986    }

1053/// The [`Api::Fence`] type for [`vulkan::Api`].
1054///
1055/// This is an `enum` because there are two possible implementations of
1056/// `wgpu-hal` fences on Vulkan: Vulkan fences, which work on any version of
1057/// Vulkan, and Vulkan timeline semaphores, which are easier and cheaper but
1058/// require non-1.0 features.
1059///
1060/// [`Device::create_fence`] returns a [`TimelineSemaphore`] if
1061/// [`VK_KHR_timeline_semaphore`] is available and enabled, and a [`FencePool`]
1062/// otherwise.
1063///
1064/// [`Api::Fence`]: crate::Api::Fence
1065/// [`vulkan::Api`]: Api
1066/// [`Device::create_fence`]: crate::Device::create_fence
1067/// [`TimelineSemaphore`]: Fence::TimelineSemaphore
1068/// [`VK_KHR_timeline_semaphore`]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#VK_KHR_timeline_semaphore
1069/// [`FencePool`]: Fence::FencePool
1070#[derive(Debug)]
1071pub enum Fence {
1072    /// A Vulkan [timeline semaphore].
1073    ///
1074    /// These are simpler to use than Vulkan fences, since timeline semaphores
1075    /// work exactly the way [`wpgu_hal::Api::Fence`] is specified to work.
1076    ///
1077    /// [timeline semaphore]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#synchronization-semaphores
1078    /// [`wpgu_hal::Api::Fence`]: crate::Api::Fence
1079    TimelineSemaphore(vk::Semaphore),
1080
1081    /// A collection of Vulkan [fence]s, each associated with a [`FenceValue`].
1082    ///
1083    /// The effective [`FenceValue`] of this variant is the greater of
1084    /// `last_completed` and the maximum value associated with a signalled fence
1085    /// in `active`.
1086    ///
1087    /// Fences are available in all versions of Vulkan, but since they only have
1088    /// two states, "signaled" and "unsignaled", we need to use a separate fence
1089    /// for each queue submission we might want to wait for, and remember which
1090    /// [`FenceValue`] each one represents.
1091    ///
1092    /// [fence]: https://registry.khronos.org/vulkan/specs/1.3-extensions/html/vkspec.html#synchronization-fences
1093    /// [`FenceValue`]: crate::FenceValue
1094    FencePool {
1095        last_completed: crate::FenceValue,
1096        /// The pending fence values have to be ascending.
1097        active: Vec<(crate::FenceValue, vk::Fence)>,
1098        free: Vec<vk::Fence>,
1099    },
1100}
1101
1102impl crate::DynFence for Fence {}
1103
1104impl Fence {
1105    /// Return the highest [`FenceValue`] among the signalled fences in `active`.
1106    ///
1107    /// As an optimization, assume that we already know that the fence has
1108    /// reached `last_completed`, and don't bother checking fences whose values
1109    /// are less than that: those fences remain in the `active` array only
1110    /// because we haven't called `maintain` yet to clean them up.
1111    ///
1112    /// [`FenceValue`]: crate::FenceValue
1113    fn check_active(
1114        device: &ash::Device,
1115        mut last_completed: crate::FenceValue,
1116        active: &[(crate::FenceValue, vk::Fence)],
1117    ) -> Result<crate::FenceValue, crate::DeviceError> {
1118        for &(value, raw) in active.iter() {
1119            unsafe {
1120                if value > last_completed
1121                    && device
1122                        .get_fence_status(raw)
1123                        .map_err(map_host_device_oom_and_lost_err)?
1124                {
1125                    last_completed = value;
1126                }
1127            }
1128        }
1129        Ok(last_completed)
1130    }
1131
1132    /// Return the highest signalled [`FenceValue`] for `self`.
1133    ///
1134    /// [`FenceValue`]: crate::FenceValue
1135    fn get_latest(
1136        &self,
1137        device: &ash::Device,
1138        extension: Option<&ExtensionFn<khr::timeline_semaphore::Device>>,
1139    ) -> Result<crate::FenceValue, crate::DeviceError> {
1140        match *self {
1141            Self::TimelineSemaphore(raw) => unsafe {
1142                Ok(match *extension.unwrap() {
1143                    ExtensionFn::Extension(ref ext) => ext
1144                        .get_semaphore_counter_value(raw)
1145                        .map_err(map_host_device_oom_and_lost_err)?,
1146                    ExtensionFn::Promoted => device
1147                        .get_semaphore_counter_value(raw)
1148                        .map_err(map_host_device_oom_and_lost_err)?,
1149                })
1150            },
1151            Self::FencePool {
1152                last_completed,
1153                ref active,
1154                free: _,
1155            } => Self::check_active(device, last_completed, active),
1156        }
1157    }
1158
1159    /// Trim the internal state of this [`Fence`].
1160    ///
1161    /// This function has no externally visible effect, but you should call it
1162    /// periodically to keep this fence's resource consumption under control.
1163    ///
1164    /// For fences using the [`FencePool`] implementation, this function
1165    /// recycles fences that have been signaled. If you don't call this,
1166    /// [`Queue::submit`] will just keep allocating a new Vulkan fence every
1167    /// time it's called.
1168    ///
1169    /// [`FencePool`]: Fence::FencePool
1170    /// [`Queue::submit`]: crate::Queue::submit
1171    fn maintain(&mut self, device: &ash::Device) -> Result<(), crate::DeviceError> {
1172        match *self {
1173            Self::TimelineSemaphore(_) => {}
1174            Self::FencePool {
1175                ref mut last_completed,
1176                ref mut active,
1177                ref mut free,
1178            } => {
1179                let latest = Self::check_active(device, *last_completed, active)?;
1180                let base_free = free.len();
1181                for &(value, raw) in active.iter() {
1182                    if value <= latest {
1183                        free.push(raw);
1184                    }
1185                }
1186                if free.len() != base_free {
1187                    active.retain(|&(value, _)| value > latest);
1188                    unsafe { device.reset_fences(&free[base_free..]) }
1189                        .map_err(map_device_oom_err)?
1190                }
1191                *last_completed = latest;
1192            }
1193        }
1194        Ok(())
1195    }
1196}

1198impl crate::Queue for Queue {
1199    type A = Api;
1200
1201    unsafe fn submit(
1202        &self,
1203        command_buffers: &[&CommandBuffer],
1204        surface_textures: &[&SurfaceTexture],
1205        (signal_fence, signal_value): (&mut Fence, crate::FenceValue),
1206    ) -> Result<(), crate::DeviceError> {
1207        let mut fence_raw = vk::Fence::null();
1208
1209        let mut wait_semaphores = SemaphoreList::new(SemaphoreListMode::Wait);
1210        let mut signal_semaphores = SemaphoreList::new(SemaphoreListMode::Signal);
1211
1212        // Double check that the same swapchain image isn't being given to us multiple times,
1213        // as that will deadlock when we try to lock them all.
1214        debug_assert!(
1215            {
1216                let mut check = HashSet::with_capacity(surface_textures.len());
1217                // We compare the Box by pointer, as Eq isn't well defined for SurfaceSemaphores.
1218                for st in surface_textures {
1219                    let ptr: *const () = <*const _>::cast(&*st.metadata);
1220                    check.insert(ptr as usize);
1221                }
1222                check.len() == surface_textures.len()
1223            },
1224            "More than one surface texture is being used from the same swapchain. This will cause a deadlock in release."
1225        );
1226
1227        let locked_swapchain_semaphores = surface_textures
1228            .iter()
1229            .map(|st| st.metadata.get_semaphore_guard())
1230            .collect::<Vec<_>>();
1231
1232        for mut semaphores in locked_swapchain_semaphores {
1233            semaphores.set_used_fence_value(signal_value);
1234
1235            // If we're the first submission to operate on this image, wait on
1236            // its acquire semaphore, to make sure the presentation engine is
1237            // done with it.
1238            if let Some(sem) = semaphores.get_acquire_wait_semaphore() {
1239                wait_semaphores.push_wait(sem, vk::PipelineStageFlags::TOP_OF_PIPE);
1240            }
1241
1242            // Get a semaphore to signal when we're done writing to this surface
1243            // image. Presentation of this image will wait for this.
1244            let signal_semaphore = semaphores.get_submit_signal_semaphore(&self.device)?;
1245            signal_semaphores.push_signal(signal_semaphore);
1246        }
1247
1248        let mut guard = self.signal_semaphores.lock();
1249        if !guard.is_empty() {
1250            signal_semaphores.append(&mut guard);
1251        }
1252
1253        // In order for submissions to be strictly ordered, we encode a dependency between each submission
1254        // using a pair of semaphores. This adds a wait if it is needed, and signals the next semaphore.
1255        let semaphore_state = self.relay_semaphores.lock().advance(&self.device)?;
1256
1257        if let Some(sem) = semaphore_state.wait {
1258            wait_semaphores.push_wait(
1259                SemaphoreType::Binary(sem),
1260                vk::PipelineStageFlags::TOP_OF_PIPE,
1261            );
1262        }
1263
1264        signal_semaphores.push_signal(SemaphoreType::Binary(semaphore_state.signal));
1265
1266        // We need to signal our wgpu::Fence if we have one, this adds it to the signal list.
1267        signal_fence.maintain(&self.device.raw)?;
1268        match *signal_fence {
1269            Fence::TimelineSemaphore(raw) => {
1270                signal_semaphores.push_signal(SemaphoreType::Timeline(raw, signal_value));
1271            }
1272            Fence::FencePool {
1273                ref mut active,
1274                ref mut free,
1275                ..
1276            } => {
1277                fence_raw = match free.pop() {
1278                    Some(raw) => raw,
1279                    None => unsafe {
1280                        self.device
1281                            .raw
1282                            .create_fence(&vk::FenceCreateInfo::default(), None)
1283                            .map_err(map_host_device_oom_err)?
1284                    },
1285                };
1286                active.push((signal_value, fence_raw));
1287            }
1288        }
1289
1290        let vk_cmd_buffers = command_buffers
1291            .iter()
1292            .map(|cmd| cmd.raw)
1293            .collect::<Vec<_>>();
1294
1295        let mut vk_info = vk::SubmitInfo::default().command_buffers(&vk_cmd_buffers);
1296        let mut vk_timeline_info = mem::MaybeUninit::uninit();
1297        vk_info = SemaphoreList::add_to_submit(
1298            &mut wait_semaphores,
1299            &mut signal_semaphores,
1300            vk_info,
1301            &mut vk_timeline_info,
1302        );
1303
1304        profiling::scope!("vkQueueSubmit");
1305        unsafe {
1306            self.device
1307                .raw
1308                .queue_submit(self.raw, &[vk_info], fence_raw)
1309                .map_err(map_host_device_oom_and_lost_err)?
1310        };
1311        Ok(())
1312    }

1329impl Queue {

1334    pub fn add_signal_semaphore(&self, semaphore: vk::Semaphore, semaphore_value: Option<u64>) {
1335        let mut guard = self.signal_semaphores.lock();
1336        if let Some(value) = semaphore_value {
1337            guard.push_signal(SemaphoreType::Timeline(semaphore, value));
1338        } else {
1339            guard.push_signal(SemaphoreType::Binary(semaphore));
1340        }
1341    }
