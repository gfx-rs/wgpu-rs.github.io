22/// Features supported by a [`vk::PhysicalDevice`] and its extensions.
23///
24/// This is used in two phases:
25///
26/// - When enumerating adapters, this represents the features offered by the
27///   adapter. [`Instance::expose_adapter`] calls `vkGetPhysicalDeviceFeatures2`
28///   (or `vkGetPhysicalDeviceFeatures` if that is not available) to collect
29///   this information about the `VkPhysicalDevice` represented by the
30///   `wgpu_hal::ExposedAdapter`.
31///
32/// - When opening a device, this represents the features we would like to
33///   enable. At `wgpu_hal::Device` construction time,
34///   [`PhysicalDeviceFeatures::from_extensions_and_requested_features`]
35///   constructs an value of this type indicating which Vulkan features to
36///   enable, based on the `wgpu_types::Features` requested.
37///
38/// [`Instance::expose_adapter`]: super::Instance::expose_adapter

538    /// built from, and a `caps` also built from those, determine which wgpu
539    /// features and downlevel flags the device can support.
540    ///
541    /// [`Features`]: wgt::Features
542    /// [`DownlevelFlags`]: wgt::DownlevelFlags
543    fn to_wgpu(
544        &self,
545        instance: &ash::Instance,
546        phd: vk::PhysicalDevice,
547        caps: &PhysicalDeviceProperties,
548    ) -> (wgt::Features, wgt::DownlevelFlags) {
549        use wgt::{DownlevelFlags as Df, Features as F};
550        let mut features = F::empty()
551            | F::MAPPABLE_PRIMARY_BUFFERS
552            | F::PUSH_CONSTANTS
553            | F::ADDRESS_MODE_CLAMP_TO_BORDER
554            | F::ADDRESS_MODE_CLAMP_TO_ZERO
555            | F::TIMESTAMP_QUERY
556            | F::TIMESTAMP_QUERY_INSIDE_ENCODERS
557            | F::TIMESTAMP_QUERY_INSIDE_PASSES
558            | F::TEXTURE_ADAPTER_SPECIFIC_FORMAT_FEATURES
559            | F::CLEAR_TEXTURE
560            | F::PIPELINE_CACHE
561            | F::SHADER_EARLY_DEPTH_TEST
562            | F::TEXTURE_ATOMIC
563            | F::EXPERIMENTAL_PASSTHROUGH_SHADERS;
564
565        let mut dl_flags = Df::COMPUTE_SHADERS
566            | Df::BASE_VERTEX
567            | Df::READ_ONLY_DEPTH_STENCIL
568            | Df::NON_POWER_OF_TWO_MIPMAPPED_TEXTURES
569            | Df::COMPARISON_SAMPLERS
570            | Df::VERTEX_STORAGE
571            | Df::FRAGMENT_STORAGE
572            | Df::DEPTH_TEXTURE_AND_BUFFER_COPIES
573            | Df::BUFFER_BINDINGS_NOT_16_BYTE_ALIGNED
574            | Df::UNRESTRICTED_INDEX_BUFFER
575            | Df::INDIRECT_EXECUTION
576            | Df::VIEW_FORMATS
577            | Df::UNRESTRICTED_EXTERNAL_TEXTURE_COPIES
578            | Df::NONBLOCKING_QUERY_RESOLVE;
579
580        dl_flags.set(
581            Df::SURFACE_VIEW_FORMATS,
582            caps.supports_extension(khr::swapchain_mutable_format::NAME),
583        );
584        dl_flags.set(Df::CUBE_ARRAY_TEXTURES, self.core.image_cube_array != 0);
585        dl_flags.set(Df::ANISOTROPIC_FILTERING, self.core.sampler_anisotropy != 0);
586        dl_flags.set(
587            Df::FRAGMENT_WRITABLE_STORAGE,
588            self.core.fragment_stores_and_atomics != 0,
589        );
590        dl_flags.set(Df::MULTISAMPLED_SHADING, self.core.sample_rate_shading != 0);
591        dl_flags.set(Df::INDEPENDENT_BLEND, self.core.independent_blend != 0);
592        dl_flags.set(
593            Df::FULL_DRAW_INDEX_UINT32,
594            self.core.full_draw_index_uint32 != 0,
595        );
596        dl_flags.set(Df::DEPTH_BIAS_CLAMP, self.core.depth_bias_clamp != 0);
597
598        features.set(
599            F::INDIRECT_FIRST_INSTANCE,
600            self.core.draw_indirect_first_instance != 0,
601        );
602        //if self.core.dual_src_blend != 0
603        features.set(F::POLYGON_MODE_LINE, self.core.fill_mode_non_solid != 0);
604        features.set(F::POLYGON_MODE_POINT, self.core.fill_mode_non_solid != 0);
605        //if self.core.depth_bounds != 0 {
606        //if self.core.alpha_to_one != 0 {
607        //if self.core.multi_viewport != 0 {
608        features.set(
609            F::TEXTURE_COMPRESSION_ETC2,
610            self.core.texture_compression_etc2 != 0,
611        );
612        features.set(
613            F::TEXTURE_COMPRESSION_ASTC,
614            self.core.texture_compression_astc_ldr != 0,
615        );
616        features.set(
617            F::TEXTURE_COMPRESSION_BC,
618            self.core.texture_compression_bc != 0,
619        );
620        features.set(
621            F::TEXTURE_COMPRESSION_BC_SLICED_3D,
622            self.core.texture_compression_bc != 0, // BC guarantees Sliced 3D
623        );
624        features.set(
625            F::PIPELINE_STATISTICS_QUERY,
626            self.core.pipeline_statistics_query != 0,
627        );
628        features.set(
629            F::VERTEX_WRITABLE_STORAGE,
630            self.core.vertex_pipeline_stores_and_atomics != 0,
631        );
632
633        features.set(F::SHADER_F64, self.core.shader_float64 != 0);
634        features.set(F::SHADER_INT64, self.core.shader_int64 != 0);
635        features.set(F::SHADER_I16, self.core.shader_int16 != 0);
636
637        features.set(F::SHADER_PRIMITIVE_INDEX, self.core.geometry_shader != 0);
638
639        if let Some(ref shader_atomic_int64) = self.shader_atomic_int64 {
640            features.set(
641                F::SHADER_INT64_ATOMIC_ALL_OPS | F::SHADER_INT64_ATOMIC_MIN_MAX,
642                shader_atomic_int64.shader_buffer_int64_atomics != 0
643                    && shader_atomic_int64.shader_shared_int64_atomics != 0,
644            );
645        }
646
647        if let Some(ref shader_image_atomic_int64) = self.shader_image_atomic_int64 {
648            features.set(
649                F::TEXTURE_INT64_ATOMIC,
650                shader_image_atomic_int64
651                    .shader_image_int64_atomics(true)
652                    .shader_image_int64_atomics
653                    != 0,
654            );
655        }
656
657        if let Some(ref shader_atomic_float) = self.shader_atomic_float {
658            features.set(
659                F::SHADER_FLOAT32_ATOMIC,
660                shader_atomic_float.shader_buffer_float32_atomics != 0
661                    && shader_atomic_float.shader_buffer_float32_atomic_add != 0,
662            );
663        }
664
665        //if caps.supports_extension(khr::sampler_mirror_clamp_to_edge::NAME) {
666        //if caps.supports_extension(ext::sampler_filter_minmax::NAME) {
667        features.set(
668            F::MULTI_DRAW_INDIRECT_COUNT,
669            caps.supports_extension(khr::draw_indirect_count::NAME),
670        );
671        features.set(
672            F::CONSERVATIVE_RASTERIZATION,
673            caps.supports_extension(ext::conservative_rasterization::NAME),
674        );
675        features.set(
676            F::EXPERIMENTAL_RAY_HIT_VERTEX_RETURN,
677            caps.supports_extension(khr::ray_tracing_position_fetch::NAME),
678        );
679
680        if let Some(ref descriptor_indexing) = self.descriptor_indexing {
681            // We use update-after-bind descriptors for all bind groups containing binding arrays.
682            //
683            // In those bind groups, we allow all binding types except uniform buffers to be present.
684            //
685            // As we can only switch between update-after-bind and not on a per bind group basis,
686            // all supported binding types need to be able to be marked update after bind.
687            //
688            // As such, we enable all features as a whole, rather individually.
689            let supports_descriptor_indexing =
690                // Sampled Images
691                descriptor_indexing.shader_sampled_image_array_non_uniform_indexing != 0
692                    && descriptor_indexing.descriptor_binding_sampled_image_update_after_bind != 0
693                    // Storage Images
694                    && descriptor_indexing.shader_storage_image_array_non_uniform_indexing != 0
695                    && descriptor_indexing.descriptor_binding_storage_image_update_after_bind != 0
696                    // Storage Buffers
697                    && descriptor_indexing.shader_storage_buffer_array_non_uniform_indexing != 0
698                    && descriptor_indexing.descriptor_binding_storage_buffer_update_after_bind != 0;
699
700            let descriptor_indexing_features = F::BUFFER_BINDING_ARRAY
701                | F::TEXTURE_BINDING_ARRAY
702                | F::STORAGE_RESOURCE_BINDING_ARRAY
703                | F::SAMPLED_TEXTURE_AND_STORAGE_BUFFER_ARRAY_NON_UNIFORM_INDEXING
704                | F::STORAGE_TEXTURE_ARRAY_NON_UNIFORM_INDEXING;
705
706            features.set(descriptor_indexing_features, supports_descriptor_indexing);
707
708            let supports_partially_bound =
709                descriptor_indexing.descriptor_binding_partially_bound != 0;
710
711            features.set(F::PARTIALLY_BOUND_BINDING_ARRAY, supports_partially_bound);
712        }
713
714        features.set(F::DEPTH_CLIP_CONTROL, self.core.depth_clamp != 0);
715        features.set(F::DUAL_SOURCE_BLENDING, self.core.dual_src_blend != 0);
716        features.set(F::CLIP_DISTANCES, self.core.shader_clip_distance != 0);
717
718        if let Some(ref multiview) = self.multiview {
719            features.set(F::MULTIVIEW, multiview.multiview != 0);
720        }
721
722        features.set(
723            F::TEXTURE_FORMAT_16BIT_NORM,
724            is_format_16bit_norm_supported(instance, phd),
725        );
726
727        if let Some(ref astc_hdr) = self.astc_hdr {
728            features.set(
729                F::TEXTURE_COMPRESSION_ASTC_HDR,
730                astc_hdr.texture_compression_astc_hdr != 0,
731            );
732        }
733
734        if self.core.texture_compression_astc_ldr != 0 {
735            features.set(
736                F::TEXTURE_COMPRESSION_ASTC_SLICED_3D,
737                supports_astc_3d(instance, phd),
738            );
739        }
740
741        if let (Some(ref f16_i8), Some(ref bit16)) = (self.shader_float16_int8, self._16bit_storage)
742        {
743            // Note `storage_input_output16` is not required, we polyfill `f16` I/O using `f32`
744            // types when this capability is not available
745            features.set(
746                F::SHADER_F16,
747                f16_i8.shader_float16 != 0
748                    && bit16.storage_buffer16_bit_access != 0
749                    && bit16.uniform_and_storage_buffer16_bit_access != 0,
750            );
751        }
752
753        if let Some(ref subgroup) = caps.subgroup {
754            if (caps.device_api_version >= vk::API_VERSION_1_3
755                || caps.supports_extension(ext::subgroup_size_control::NAME))
756                && subgroup.supported_operations.contains(
757                    vk::SubgroupFeatureFlags::BASIC
758                        | vk::SubgroupFeatureFlags::VOTE
759                        | vk::SubgroupFeatureFlags::ARITHMETIC
760                        | vk::SubgroupFeatureFlags::BALLOT
761                        | vk::SubgroupFeatureFlags::SHUFFLE
762                        | vk::SubgroupFeatureFlags::SHUFFLE_RELATIVE
763                        | vk::SubgroupFeatureFlags::QUAD,
764                )
765            {
766                features.set(
767                    F::SUBGROUP,
768                    subgroup
769                        .supported_stages
770                        .contains(vk::ShaderStageFlags::COMPUTE | vk::ShaderStageFlags::FRAGMENT),
771                );
772                features.set(
773                    F::SUBGROUP_VERTEX,
774                    subgroup
775                        .supported_stages
776                        .contains(vk::ShaderStageFlags::VERTEX),
777                );
778                features.insert(F::SUBGROUP_BARRIER);
779            }
780        }
781
782        let supports_depth_format = |format| {
783            supports_format(
784                instance,
785                phd,
786                format,
787                vk::ImageTiling::OPTIMAL,
788                depth_stencil_required_flags(),
789            )
790        };
791
792        let texture_s8 = supports_depth_format(vk::Format::S8_UINT);
793        let texture_d32 = supports_depth_format(vk::Format::D32_SFLOAT);
794        let texture_d24_s8 = supports_depth_format(vk::Format::D24_UNORM_S8_UINT);
795        let texture_d32_s8 = supports_depth_format(vk::Format::D32_SFLOAT_S8_UINT);
796
797        let stencil8 = texture_s8 || texture_d24_s8;
798        let depth24_plus_stencil8 = texture_d24_s8 || texture_d32_s8;
799
800        dl_flags.set(
801            Df::WEBGPU_TEXTURE_FORMAT_SUPPORT,
802            stencil8 && depth24_plus_stencil8 && texture_d32,
803        );
804
805        features.set(F::DEPTH32FLOAT_STENCIL8, texture_d32_s8);
806
807        let supports_acceleration_structures = caps
808            .supports_extension(khr::deferred_host_operations::NAME)
809            && caps.supports_extension(khr::acceleration_structure::NAME)
810            && caps.supports_extension(khr::buffer_device_address::NAME);
811
812        features.set(
813            F::EXPERIMENTAL_RAY_QUERY
814            // Although this doesn't really require ray queries, it does not make sense to be enabled if acceleration structures
815            // aren't enabled.
816                | F::EXTENDED_ACCELERATION_STRUCTURE_VERTEX_FORMATS,
817            supports_acceleration_structures && caps.supports_extension(khr::ray_query::NAME),
818        );
819
820        let rg11b10ufloat_renderable = supports_format(
821            instance,
822            phd,
823            vk::Format::B10G11R11_UFLOAT_PACK32,
824            vk::ImageTiling::OPTIMAL,
825            vk::FormatFeatureFlags::COLOR_ATTACHMENT
826                | vk::FormatFeatureFlags::COLOR_ATTACHMENT_BLEND,
827        );
828        features.set(F::RG11B10UFLOAT_RENDERABLE, rg11b10ufloat_renderable);
829
830        features.set(
831            F::BGRA8UNORM_STORAGE,
832            supports_bgra8unorm_storage(instance, phd, caps.device_api_version),
833        );
834
835        features.set(
836            F::FLOAT32_FILTERABLE,
837            is_float32_filterable_supported(instance, phd),
838        );
839
840        if let Some(ref _sampler_ycbcr_conversion) = self.sampler_ycbcr_conversion {
841            features.set(
842                F::TEXTURE_FORMAT_NV12,
843                supports_format(
844                    instance,
845                    phd,
846                    vk::Format::G8_B8R8_2PLANE_420_UNORM,
847                    vk::ImageTiling::OPTIMAL,
848                    vk::FormatFeatureFlags::SAMPLED_IMAGE
849                        | vk::FormatFeatureFlags::TRANSFER_SRC
850                        | vk::FormatFeatureFlags::TRANSFER_DST,
851                ) && !caps
852                    .driver
853                    .map(|driver| driver.driver_id == vk::DriverId::MOLTENVK)
854                    .unwrap_or_default(),
855            );
856        }
857
858        if let Some(ref _sampler_ycbcr_conversion) = self.sampler_ycbcr_conversion {
859            features.set(
860                F::TEXTURE_FORMAT_P010,
861                supports_format(
862                    instance,
863                    phd,
864                    vk::Format::G10X6_B10X6R10X6_2PLANE_420_UNORM_3PACK16,
865                    vk::ImageTiling::OPTIMAL,
866                    vk::FormatFeatureFlags::SAMPLED_IMAGE
867                        | vk::FormatFeatureFlags::TRANSFER_SRC
868                        | vk::FormatFeatureFlags::TRANSFER_DST,
869                ) && !caps
870                    .driver
871                    .map(|driver| driver.driver_id == vk::DriverId::MOLTENVK)
872                    .unwrap_or_default(),
873            );
874        }
875
876        features.set(
877            F::VULKAN_GOOGLE_DISPLAY_TIMING,
878            caps.supports_extension(google::display_timing::NAME),
879        );
880
881        features.set(
882            F::VULKAN_EXTERNAL_MEMORY_WIN32,
883            caps.supports_extension(khr::external_memory_win32::NAME),
884        );
885        features.set(
886            F::EXPERIMENTAL_MESH_SHADER,
887            caps.supports_extension(ext::mesh_shader::NAME),
888        );
889        if let Some(ref mesh_shader) = self.mesh_shader {
890            features.set(
891                F::EXPERIMENTAL_MESH_SHADER_MULTIVIEW,
892                mesh_shader.multiview_mesh_shader != 0,
893            );
894        }
895        (features, dl_flags)
896    }
897}

899/// Vulkan "properties" structures gathered about a physical device.
900///
901/// This structure holds the properties of a [`vk::PhysicalDevice`]:
902/// - the standard Vulkan device properties
903/// - the `VkExtensionProperties` structs for all available extensions, and
904/// - the per-extension properties structures for the available extensions that
905///   `wgpu` cares about.
906///
907/// Generally, if you get it from any of these functions, it's stored
908/// here:
909/// - `vkEnumerateDeviceExtensionProperties`
910/// - `vkGetPhysicalDeviceProperties`
911/// - `vkGetPhysicalDeviceProperties2`
912///
913/// This also includes a copy of the device API version, since we can
914/// use that as a shortcut for searching for an extension, if the
915/// extension has been promoted to core in the current version.
916///
917/// This does not include device features; for those, see
918/// [`PhysicalDeviceFeatures`].
919#[derive(Default, Debug)]
920pub struct PhysicalDeviceProperties {
921    /// Extensions supported by the `vk::PhysicalDevice`,
922    /// as returned by `vkEnumerateDeviceExtensionProperties`.
923    supported_extensions: Vec<vk::ExtensionProperties>,
924
925    /// Properties of the `vk::PhysicalDevice`, as returned by
926    /// `vkGetPhysicalDeviceProperties`.
927    properties: vk::PhysicalDeviceProperties,
928
929    /// Additional `vk::PhysicalDevice` properties from the
930    /// `VK_KHR_maintenance3` extension, promoted to Vulkan 1.1.
931    maintenance_3: Option<vk::PhysicalDeviceMaintenance3Properties<'static>>,
932
933    /// Additional `vk::PhysicalDevice` properties from the
934    /// `VK_EXT_descriptor_indexing` extension, promoted to Vulkan 1.2.
935    descriptor_indexing: Option<vk::PhysicalDeviceDescriptorIndexingPropertiesEXT<'static>>,
936
937    /// Additional `vk::PhysicalDevice` properties from the
938    /// `VK_KHR_acceleration_structure` extension.
939    acceleration_structure: Option<vk::PhysicalDeviceAccelerationStructurePropertiesKHR<'static>>,
940
941    /// Additional `vk::PhysicalDevice` properties from the
942    /// `VK_KHR_driver_properties` extension, promoted to Vulkan 1.2.
943    driver: Option<vk::PhysicalDeviceDriverPropertiesKHR<'static>>,
944
945    /// Additional `vk::PhysicalDevice` properties from Vulkan 1.1.
946    subgroup: Option<vk::PhysicalDeviceSubgroupProperties<'static>>,
947
948    /// Additional `vk::PhysicalDevice` properties from the
949    /// `VK_EXT_subgroup_size_control` extension, promoted to Vulkan 1.3.
950    subgroup_size_control: Option<vk::PhysicalDeviceSubgroupSizeControlProperties<'static>>,
951
952    /// Additional `vk::PhysicalDevice` properties from the
953    /// `VK_EXT_robustness2` extension.
954    robustness2: Option<vk::PhysicalDeviceRobustness2PropertiesEXT<'static>>,
955
956    /// Additional `vk::PhysicalDevice` properties from the
957    /// `VK_EXT_mesh_shader` extension.
958    mesh_shader: Option<vk::PhysicalDeviceMeshShaderPropertiesEXT<'static>>,
959
960    /// The device API version.
961    ///
962    /// Which is the version of Vulkan supported for device-level functionality.
963    ///
964    /// It is associated with a `VkPhysicalDevice` and its children.
965    device_api_version: u32,
966}
967
968impl PhysicalDeviceProperties {
969    pub fn properties(&self) -> vk::PhysicalDeviceProperties {
970        self.properties
971    }
972
973    pub fn supports_extension(&self, extension: &CStr) -> bool {
974        self.supported_extensions
975            .iter()
976            .any(|ep| ep.extension_name_as_c_str() == Ok(extension))
977    }
978
979    /// Map `requested_features` to the list of Vulkan extension strings required to create the logical device.
980    fn get_required_extensions(&self, requested_features: wgt::Features) -> Vec<&'static CStr> {
981        let mut extensions = Vec::new();
982
983        // Note that quite a few extensions depend on the `VK_KHR_get_physical_device_properties2` instance extension.
984        // We enable `VK_KHR_get_physical_device_properties2` unconditionally (if available).
985
986        // Require `VK_KHR_swapchain`
987        extensions.push(khr::swapchain::NAME);
988
989        if self.device_api_version < vk::API_VERSION_1_1 {
990            // Require `VK_KHR_maintenance1`
991            extensions.push(khr::maintenance1::NAME);
992
993            // Optional `VK_KHR_maintenance2`
994            if self.supports_extension(khr::maintenance2::NAME) {
995                extensions.push(khr::maintenance2::NAME);
996            }
997
998            // Optional `VK_KHR_maintenance3`
999            if self.supports_extension(khr::maintenance3::NAME) {
1000                extensions.push(khr::maintenance3::NAME);
1001            }
1002
1003            // Require `VK_KHR_storage_buffer_storage_class`
1004            extensions.push(khr::storage_buffer_storage_class::NAME);
1005
1006            // Require `VK_KHR_multiview` if the associated feature was requested
1007            if requested_features.contains(wgt::Features::MULTIVIEW) {
1008                extensions.push(khr::multiview::NAME);
1009            }
1010
1011            // Require `VK_KHR_sampler_ycbcr_conversion` if the associated feature was requested
1012            if requested_features.contains(wgt::Features::TEXTURE_FORMAT_NV12) {
1013                extensions.push(khr::sampler_ycbcr_conversion::NAME);
1014            }
1015
1016            // Require `VK_KHR_16bit_storage` if the feature `SHADER_F16` was requested
1017            if requested_features.contains(wgt::Features::SHADER_F16) {
1018                // - Feature `SHADER_F16` also requires `VK_KHR_shader_float16_int8`, but we always
1019                //   require that anyway (if it is available) below.
1020                // - `VK_KHR_16bit_storage` requires `VK_KHR_storage_buffer_storage_class`, however
1021                //   we require that one already.
1022                extensions.push(khr::_16bit_storage::NAME);
1023            }
1024        }
1025
1026        if self.device_api_version < vk::API_VERSION_1_2 {
1027            // Optional `VK_KHR_image_format_list`
1028            if self.supports_extension(khr::image_format_list::NAME) {
1029                extensions.push(khr::image_format_list::NAME);
1030            }
1031
1032            // Optional `VK_KHR_driver_properties`
1033            if self.supports_extension(khr::driver_properties::NAME) {
1034                extensions.push(khr::driver_properties::NAME);
1035            }
1036
1037            // Optional `VK_KHR_timeline_semaphore`
1038            if self.supports_extension(khr::timeline_semaphore::NAME) {
1039                extensions.push(khr::timeline_semaphore::NAME);
1040            }
1041
1042            // Require `VK_EXT_descriptor_indexing` if one of the associated features was requested
1043            if requested_features.intersects(INDEXING_FEATURES) {
1044                extensions.push(ext::descriptor_indexing::NAME);
1045            }
1046
1047            // Always require `VK_KHR_shader_float16_int8` if available as it enables
1048            // Int8 optimizations. Also require it even if it's not available but
1049            // requested so that we get a corresponding error message.
1050            if requested_features.contains(wgt::Features::SHADER_F16)
1051                || self.supports_extension(khr::shader_float16_int8::NAME)
1052            {
1053                extensions.push(khr::shader_float16_int8::NAME);
1054            }
1055
1056            if requested_features.intersects(wgt::Features::EXPERIMENTAL_MESH_SHADER) {
1057                extensions.push(khr::spirv_1_4::NAME);
1058            }
1059
1060            //extensions.push(khr::sampler_mirror_clamp_to_edge::NAME);
1061            //extensions.push(ext::sampler_filter_minmax::NAME);
1062        }
1063
1064        if self.device_api_version < vk::API_VERSION_1_3 {
1065            // Optional `VK_EXT_image_robustness`
1066            if self.supports_extension(ext::image_robustness::NAME) {
1067                extensions.push(ext::image_robustness::NAME);
1068            }
1069
1070            // Require `VK_EXT_subgroup_size_control` if the associated feature was requested
1071            if requested_features.contains(wgt::Features::SUBGROUP) {
1072                extensions.push(ext::subgroup_size_control::NAME);
1073            }
1074
1075            if requested_features.intersects(wgt::Features::EXPERIMENTAL_MESH_SHADER) {
1076                extensions.push(khr::maintenance4::NAME);
1077            }
1078
1079            // Optional `VK_KHR_shader_integer_dot_product`
1080            if self.supports_extension(khr::shader_integer_dot_product::NAME) {
1081                extensions.push(khr::shader_integer_dot_product::NAME);
1082            }
1083        }
1084
1085        // Optional `VK_KHR_swapchain_mutable_format`
1086        if self.supports_extension(khr::swapchain_mutable_format::NAME) {
1087            extensions.push(khr::swapchain_mutable_format::NAME);
1088        }
1089
1090        // Optional `VK_EXT_robustness2`
1091        if self.supports_extension(ext::robustness2::NAME) {

1603
1604            if capabilities.supports_extension(ext::mesh_shader::NAME) {
1605                let next = features
1606                    .mesh_shader
1607                    .insert(vk::PhysicalDeviceMeshShaderFeaturesEXT::default());

1609            }

1612            if capabilities.device_api_version >= vk::API_VERSION_1_3
1613                || capabilities.supports_extension(khr::shader_integer_dot_product::NAME)
1614            {
1615                let next = features
1616                    .shader_integer_dot_product
1617                    .insert(vk::PhysicalDeviceShaderIntegerDotProductFeatures::default());

1619            }

1621            unsafe { get_device_properties.get_physical_device_features2(phd, &mut features2) };

1623        } else {
1624            unsafe { self.raw.get_physical_device_features(phd) }
1625        };

1627        (capabilities, features)
1628    }

2819        let features3 = properties3.optimal_tiling_features;
2820
2821        features2.contains(vk::FormatFeatureFlags::STORAGE_IMAGE)
2822            && features3.contains(vk::FormatFeatureFlags2::STORAGE_WRITE_WITHOUT_FORMAT)
2823    }

2826// For https://github.com/gfx-rs/wgpu/issues/4599
2827// Intel iGPUs with outdated drivers can break rendering if `VK_EXT_robustness2` is used.
2828// Driver version 31.0.101.2115 works, but there's probably an earlier functional version.
2829fn is_intel_igpu_outdated_for_robustness2(
2830    props: vk::PhysicalDeviceProperties,
2831    driver: Option<vk::PhysicalDeviceDriverPropertiesKHR>,
2832) -> bool {
2833    const DRIVER_VERSION_WORKING: u32 = (101 << 14) | 2115; // X.X.101.2115
2834
2835    let is_outdated = props.vendor_id == crate::auxil::db::intel::VENDOR
2836        && props.device_type == vk::PhysicalDeviceType::INTEGRATED_GPU
2837        && props.driver_version < DRIVER_VERSION_WORKING
2838        && driver
2839            .map(|driver| driver.driver_id == vk::DriverId::INTEL_PROPRIETARY_WINDOWS)
2840            .unwrap_or_default();
2841
2842    if is_outdated {
2843        log::warn!(
2844            "Disabling robustBufferAccess2 and robustImageAccess2: IntegratedGpu Intel Driver is outdated. Found with version 0x{:X}, less than the known good version 0x{:X} (31.0.101.2115)",
2845            props.driver_version,
2846            DRIVER_VERSION_WORKING
2847        );
2848    }
2849    is_outdated
