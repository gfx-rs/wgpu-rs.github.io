2fn compile_wgsl(device: &wgpu::Device) -> wgpu::ShaderModule {
3    device.create_shader_module(wgpu::ShaderModuleDescriptor {
4        label: None,
5        source: wgpu::ShaderSource::Wgsl(include_str!("shader.wgsl").into()),
6    })
7}
8fn compile_hlsl(device: &wgpu::Device, entry: &str, stage_str: &str) -> wgpu::ShaderModule {
9    let out_path = format!(
10        "{}/src/mesh_shader/shader.{stage_str}.cso",
11        env!("CARGO_MANIFEST_DIR")
12    );
13    let cmd = std::process::Command::new("dxc")
14        .args([
15            "-T",
16            &format!("{stage_str}_6_5"),
17            "-E",
18            entry,
19            &format!("{}/src/mesh_shader/shader.hlsl", env!("CARGO_MANIFEST_DIR")),
20            "-Fo",
21            &out_path,
22        ])
23        .output()
24        .unwrap();
25    if !cmd.status.success() {
26        panic!("DXC failed:\n{}", String::from_utf8(cmd.stderr).unwrap());
27    }
28    let file = std::fs::read(&out_path).unwrap();
29    std::fs::remove_file(out_path).unwrap();
30    unsafe {
31        device.create_shader_module_passthrough(wgpu::ShaderModuleDescriptorPassthrough {
32            entry_point: entry.to_owned(),
33            label: None,
34            num_workgroups: (1, 1, 1),
35            dxil: Some(std::borrow::Cow::Owned(file)),
36            ..Default::default()
37        })
38    }
39}

41fn compile_msl(device: &wgpu::Device, entry: &str) -> wgpu::ShaderModule {

45            label: None,

47            num_workgroups: (1, 1, 1),
48            ..Default::default()

53pub struct Example {
54    pipeline: wgpu::RenderPipeline,
55}
56impl crate::framework::Example for Example {
57    fn init(
58        config: &wgpu::SurfaceConfiguration,
59        adapter: &wgpu::Adapter,
60        device: &wgpu::Device,
61        _queue: &wgpu::Queue,
62    ) -> Self {
63        let (ts, ms, fs, ts_name, ms_name, fs_name) = match adapter.get_info().backend {
64            wgpu::Backend::Vulkan => (
65                compile_wgsl(device),
66                compile_wgsl(device),
67                compile_wgsl(device),
68                "ts_main",
69                "ms_main",
70                "fs_main",
71            ),
72            wgpu::Backend::Dx12 => (
73                compile_hlsl(device, "Task", "as"),
74                compile_hlsl(device, "Mesh", "ms"),
75                compile_hlsl(device, "Frag", "ps"),
76                "main",
77                "main",
78                "main",
79            ),
80            wgpu::Backend::Metal => (
81                compile_msl(device, "taskShader"),
82                compile_msl(device, "meshShader"),
83                compile_msl(device, "fragShader"),
84                "main",
85                "main",
86                "main",
87            ),
88            _ => panic!("Example can currently only run on vulkan, dx12 or metal"),
89        };
90        let pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
91            label: None,
92            bind_group_layouts: &[],
93            immediate_size: 0,
94        });
95        let pipeline = device.create_mesh_pipeline(&wgpu::MeshPipelineDescriptor {
96            label: None,
97            layout: Some(&pipeline_layout),
98            task: Some(wgpu::TaskState {
99                module: &ts,
100                entry_point: Some(ts_name),
101                compilation_options: Default::default(),
102            }),
103            mesh: wgpu::MeshState {
104                module: &ms,
105                entry_point: Some(ms_name),
106                compilation_options: Default::default(),
107            },
108            fragment: Some(wgpu::FragmentState {
109                module: &fs,
110                entry_point: Some(fs_name),
111                compilation_options: Default::default(),
112                targets: &[Some(config.view_formats[0].into())],
113            }),
114            primitive: wgpu::PrimitiveState {
115                cull_mode: Some(wgpu::Face::Back),
116                ..Default::default()
117            },
118            depth_stencil: None,
119            multisample: Default::default(),
120            multiview: None,
121            cache: None,
122        });
123        Self { pipeline }
124    }
125    fn render(&mut self, view: &wgpu::TextureView, device: &wgpu::Device, queue: &wgpu::Queue) {
126        let mut encoder =
127            device.create_command_encoder(&wgpu::CommandEncoderDescriptor { label: None });
128        {
129            let mut rpass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
130                label: None,
131                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
132                    view,
133                    resolve_target: None,
134                    ops: wgpu::Operations {
135                        load: wgpu::LoadOp::Clear(wgpu::Color {
136                            r: 0.1,
137                            g: 0.2,
138                            b: 0.3,
139                            a: 1.0,
140                        }),
141                        store: wgpu::StoreOp::Store,
142                    },
143                    depth_slice: None,
144                })],
145                depth_stencil_attachment: None,
146                timestamp_writes: None,
147                occlusion_query_set: None,
148                multiview_mask: None,
149            });
150            rpass.push_debug_group("Prepare data for draw.");
151            rpass.set_pipeline(&self.pipeline);
152            rpass.pop_debug_group();
153            rpass.insert_debug_marker("Draw!");
154            rpass.draw_mesh_tasks(1, 1, 1);
155        }
156        queue.submit(Some(encoder.finish()));
157    }
158    fn required_downlevel_capabilities() -> wgpu::DownlevelCapabilities {
159        Default::default()
160    }
161    fn required_features() -> wgpu::Features {
162        wgpu::Features::EXPERIMENTAL_MESH_SHADER | wgpu::Features::EXPERIMENTAL_PASSTHROUGH_SHADERS
163    }
164    fn required_limits() -> wgpu::Limits {
165        wgpu::Limits::defaults().using_recommended_minimum_mesh_shader_values()
166    }
167    fn resize(
168        &mut self,
169        _config: &wgpu::SurfaceConfiguration,
170        _device: &wgpu::Device,
171        _queue: &wgpu::Queue,
172    ) {
173        // empty
174    }
175    fn update(&mut self, _event: winit::event::WindowEvent) {
176        // empty
177    }
178}
