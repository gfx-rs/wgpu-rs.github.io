54// Initialize logging in platform dependant ways.
55fn init_logger() {
56    cfg_if::cfg_if! {
57        if #[cfg(target_arch = "wasm32")] {
58            // As we don't have an environment to pull logging level from, we use the query string.
59            let query_string = web_sys::window().unwrap().location().search().unwrap();
60            let query_level: Option<log::LevelFilter> = parse_url_query_string(&query_string, "RUST_LOG")
61                .and_then(|x| x.parse().ok());
62
63            // We keep wgpu at Error level, as it's very noisy.
64            let base_level = query_level.unwrap_or(log::LevelFilter::Info);
65            let wgpu_level = query_level.unwrap_or(log::LevelFilter::Error);
66
67            // On web, we use fern, as console_log doesn't have filtering on a per-module level.
68            fern::Dispatch::new()
69                .level(base_level)
70                .level_for("wgpu_core", wgpu_level)
71                .level_for("wgpu_hal", wgpu_level)
72                .level_for("naga", wgpu_level)
73                .chain(fern::Output::call(console_log::log))
74                .apply()
75                .unwrap();
76            std::panic::set_hook(Box::new(console_error_panic_hook::hook));
77        } else {
78            // parse_default_env will read the RUST_LOG environment variable and apply it on top
79            // of these default filters.
80            env_logger::builder()
81                .filter_level(log::LevelFilter::Info)
82                // We keep wgpu at Error level, as it's very noisy.
83                .filter_module("wgpu_core", log::LevelFilter::Info)
84                .filter_module("wgpu_hal", log::LevelFilter::Error)
85                .filter_module("naga", log::LevelFilter::Error)
86                .parse_default_env()
87                .init();
88        }
89    }
90}

97impl EventLoopWrapper {
98    pub fn new(title: &str) -> Self {
99        let event_loop = EventLoop::new().unwrap();
100        let mut builder = winit::window::WindowBuilder::new();
101        #[cfg(target_arch = "wasm32")]
102        {
103            use wasm_bindgen::JsCast;
104            use winit::platform::web::WindowBuilderExtWebSys;
105            let canvas = web_sys::window()
106                .unwrap()
107                .document()
108                .unwrap()
109                .get_element_by_id("canvas")
110                .unwrap()
111                .dyn_into::<web_sys::HtmlCanvasElement>()
112                .unwrap();
113            builder = builder.with_canvas(Some(canvas));
114        }
115        builder = builder.with_title(title);
116        let window = Arc::new(builder.build(&event_loop).unwrap());
117
118        Self { event_loop, window }
119    }

122/// Wrapper type which manages the surface and surface configuration.
123///
124/// As surface usage varies per platform, wrapping this up cleans up the event loop code.
125struct SurfaceWrapper {
126    surface: Option<wgpu::Surface<'static>>,
127    config: Option<wgpu::SurfaceConfiguration>,
128}
129
130impl SurfaceWrapper {
131    /// Create a new surface wrapper with no surface or configuration.
132    fn new() -> Self {
133        Self {
134            surface: None,
135            config: None,
136        }
137    }
138
139    /// Called after the instance is created, but before we request an adapter.
140    ///
141    /// On wasm, we need to create the surface here, as the WebGL backend needs
142    /// a surface (and hence a canvas) to be present to create the adapter.
143    ///
144    /// We cannot unconditionally create a surface here, as Android requires
145    /// us to wait until we receive the `Resumed` event to do so.
146    fn pre_adapter(&mut self, instance: &Instance, window: Arc<Window>) {
147        if cfg!(target_arch = "wasm32") {
148            self.surface = Some(instance.create_surface(window).unwrap());
149        }
150    }
151
152    /// Check if the event is the start condition for the surface.
153    fn start_condition(e: &Event<()>) -> bool {
154        match e {
155            // On all other platforms, we can create the surface immediately.
156            Event::NewEvents(StartCause::Init) => !cfg!(target_os = "android"),
157            // On android we need to wait for a resumed event to create the surface.
158            Event::Resumed => cfg!(target_os = "android"),
159            _ => false,
160        }
161    }
162
163    /// Called when an event which matches [`Self::start_condition`] is received.
164    ///
165    /// On all native platforms, this is where we create the surface.
166    ///
167    /// Additionally, we configure the surface based on the (now valid) window size.
168    fn resume(&mut self, context: &ExampleContext, window: Arc<Window>, srgb: bool) {
169        // Window size is only actually valid after we enter the event loop.
170        let window_size = window.inner_size();
171        let width = window_size.width.max(1);
172        let height = window_size.height.max(1);
173
174        log::info!("Surface resume {window_size:?}");
175
176        // We didn't create the surface in pre_adapter, so we need to do so now.
177        if !cfg!(target_arch = "wasm32") {
178            self.surface = Some(context.instance.create_surface(window).unwrap());
179        }
180
181        // From here on, self.surface should be Some.
182
183        let surface = self.surface.as_ref().unwrap();
184
185        // Get the default configuration,
186        let mut config = surface
187            .get_default_config(&context.adapter, width, height)
188            .expect("Surface isn't supported by the adapter.");
189        if srgb {
190            // Not all platforms (WebGPU) support sRGB swapchains, so we need to use view formats
191            let view_format = config.format.add_srgb_suffix();
192            config.view_formats.push(view_format);
193        } else {
194            // All platforms support non-sRGB swapchains, so we can just use the format directly.
195            let format = config.format.remove_srgb_suffix();
196            config.format = format;
197            config.view_formats.push(format);
198        };
199        config.present_mode = wgpu::PresentMode::Immediate;
200        config.desired_maximum_frame_latency = 3;
201
202        surface.configure(&context.device, &config);
203        self.config = Some(config);
204    }
205
206    /// Resize the surface, making sure to not resize to zero.
207    fn resize(&mut self, context: &ExampleContext, size: PhysicalSize<u32>) {
208        log::info!("Surface resize {size:?}");
209
210        let config = self.config.as_mut().unwrap();
211        config.width = size.width.max(1);
212        config.height = size.height.max(1);
213        let surface = self.surface.as_ref().unwrap();
214        surface.configure(&context.device, config);
215    }
216
217    /// Acquire the next surface texture.
218    fn acquire(&mut self, context: &ExampleContext) -> wgpu::SurfaceTexture {
219        let surface = self.surface.as_ref().unwrap();
220
221        match surface.get_current_texture() {
222            Ok(frame) => frame,
223            // If we timed out, just try again
224            Err(wgpu::SurfaceError::Timeout) => surface
225                .get_current_texture()
226                .expect("Failed to acquire next surface texture!"),
227            Err(
228                // If the surface is outdated, or was lost, reconfigure it.
229                wgpu::SurfaceError::Outdated
230                | wgpu::SurfaceError::Lost
231                | wgpu::SurfaceError::Other
232                // If OutOfMemory happens, reconfiguring may not help, but we might as well try
233                | wgpu::SurfaceError::OutOfMemory,
234            ) => {
235                surface.configure(&context.device, self.config());
236                surface
237                    .get_current_texture()
238                    .expect("Failed to acquire next surface texture!")
239            }
240        }
241    }
242
243    /// On suspend on android, we drop the surface, as it's no longer valid.
244    ///
245    /// A suspend event is always followed by at least one resume event.
246    fn suspend(&mut self) {
247        if cfg!(target_os = "android") {
248            self.surface = None;
249        }
250    }
251
252    fn get(&self) -> Option<&'_ Surface<'static>> {
253        self.surface.as_ref()
254    }
255
256    fn config(&self) -> &wgpu::SurfaceConfiguration {
257        self.config.as_ref().unwrap()
258    }
259}

261/// Context containing global wgpu resources.
262struct ExampleContext {
263    instance: wgpu::Instance,
264    adapter: wgpu::Adapter,
265    device: wgpu::Device,
266    queue: wgpu::Queue,
267}
268impl ExampleContext {
269    /// Initializes the example context.
270    async fn init_async<E: Example>(surface: &mut SurfaceWrapper, window: Arc<Window>) -> Self {
271        log::info!("Initializing wgpu...");
272
273        let instance_descriptor = wgpu::InstanceDescriptor::from_env_or_default();
274        let instance = wgpu::Instance::new(&instance_descriptor);
275        surface.pre_adapter(&instance, window);
276        let adapter = get_adapter_with_capabilities_or_from_env(
277            &instance,
278            &E::required_features(),
279            &E::required_downlevel_capabilities(),
280            &surface.get(),
281        )
282        .await;
283        // Make sure we use the texture resolution limits from the adapter, so we can support images the size of the surface.
284        let needed_limits = E::required_limits().using_resolution(adapter.limits());
285
286        let (device, queue) = adapter
287            .request_device(&wgpu::DeviceDescriptor {
288                label: None,
289                required_features: (E::optional_features() & adapter.features())
290                    | E::required_features(),
291                required_limits: needed_limits,
292                experimental_features: unsafe { wgpu::ExperimentalFeatures::enabled() },
293                memory_hints: wgpu::MemoryHints::MemoryUsage,
294                trace: match std::env::var_os("WGPU_TRACE") {
295                    Some(path) => wgpu::Trace::Directory(path.into()),
296                    None => wgpu::Trace::Off,
297                },
298            })
299            .await
300            .expect("Unable to find a suitable GPU adapter!");
301
302        Self {
303            instance,
304            adapter,
305            device,
306            queue,
307        }
308    }
309}

311struct FrameCounter {
312    // Instant of the last time we printed the frame time.
313    last_printed_instant: web_time::Instant,
314    // Number of frames since the last time we printed the frame time.
315    frame_count: u32,
316}
317
318impl FrameCounter {
319    fn new() -> Self {
320        Self {
321            last_printed_instant: web_time::Instant::now(),
322            frame_count: 0,
323        }
324    }
325
326    fn update(&mut self) {
327        self.frame_count += 1;
328        let new_instant = web_time::Instant::now();
329        let elapsed_secs = (new_instant - self.last_printed_instant).as_secs_f32();
330        if elapsed_secs > 1.0 {
331            let elapsed_ms = elapsed_secs * 1000.0;
332            let frame_time = elapsed_ms / self.frame_count as f32;
333            let fps = self.frame_count as f32 / elapsed_secs;
334            log::info!("Frame time {frame_time:.2}ms ({fps:.1} FPS)");
335
336            self.last_printed_instant = new_instant;
337            self.frame_count = 0;
338        }
339    }
340}

342async fn start<E: Example>(title: &str) {
343    init_logger();
344
345    log::debug!(
346        "Enabled backends: {:?}",
347        wgpu::Instance::enabled_backend_features()
348    );
349
350    let window_loop = EventLoopWrapper::new(title);
351    let mut surface = SurfaceWrapper::new();
352    let context = ExampleContext::init_async::<E>(&mut surface, window_loop.window.clone()).await;
353    let mut frame_counter = FrameCounter::new();
354
355    // We wait to create the example until we have a valid surface.
356    let mut example = None;
357
358    cfg_if::cfg_if! {
359        if #[cfg(target_arch = "wasm32")] {
360            use winit::platform::web::EventLoopExtWebSys;
361            let event_loop_function = EventLoop::spawn;
362        } else {
363            let event_loop_function = EventLoop::run;
364        }
365    }
366
367    log::info!("Entering event loop...");
368    #[cfg_attr(target_arch = "wasm32", expect(clippy::let_unit_value))]
369    let _ = (event_loop_function)(
370        window_loop.event_loop,
371        move |event: Event<()>, target: &EventLoopWindowTarget<()>| {
372            match event {
373                ref e if SurfaceWrapper::start_condition(e) => {
374                    surface.resume(&context, window_loop.window.clone(), E::SRGB);
375
376                    // If we haven't created the example yet, do so now.
377                    if example.is_none() {
378                        example = Some(E::init(
379                            surface.config(),
380                            &context.adapter,
381                            &context.device,
382                            &context.queue,
383                        ));
384                    }
385                }
386                Event::Suspended => {
387                    surface.suspend();
388                }
389                Event::WindowEvent { event, .. } => match event {
390                    WindowEvent::Resized(size) => {
391                        surface.resize(&context, size);
392                        example.as_mut().unwrap().resize(
393                            surface.config(),
394                            &context.device,
395                            &context.queue,
396                        );
397
398                        window_loop.window.request_redraw();
399                    }
400                    WindowEvent::KeyboardInput {
401                        event:
402                            KeyEvent {
403                                logical_key: Key::Named(NamedKey::Escape),
404                                ..
405                            },
406                        ..
407                    }
408                    | WindowEvent::CloseRequested => {
409                        target.exit();
410                    }
411                    #[cfg(not(target_arch = "wasm32"))]
412                    WindowEvent::KeyboardInput {
413                        event:
414                            KeyEvent {
415                                logical_key: Key::Character(s),
416                                ..
417                            },
418                        ..
419                    } if s == "r" => {
420                        println!("{:#?}", context.instance.generate_report());
421                    }
422                    WindowEvent::RedrawRequested => {
423                        // On MacOS, currently redraw requested comes in _before_ Init does.
424                        // If this happens, just drop the requested redraw on the floor.
425                        //
426                        // See https://github.com/rust-windowing/winit/issues/3235 for some discussion
427                        if example.is_none() {
428                            return;
429                        }
430
431                        frame_counter.update();
432
433                        let frame = surface.acquire(&context);
434                        let view = frame.texture.create_view(&wgpu::TextureViewDescriptor {
435                            format: Some(surface.config().view_formats[0]),
436                            ..wgpu::TextureViewDescriptor::default()
437                        });
438
439                        example
440                            .as_mut()
441                            .unwrap()
442                            .render(&view, &context.device, &context.queue);
443
444                        window_loop.window.pre_present_notify();
445                        frame.present();
446
447                        window_loop.window.request_redraw();
448                    }
449                    _ => example.as_mut().unwrap().update(event),
450                },
451                _ => {}
452            }
453        },
454    );
455}

457pub fn run<E: Example>(title: &'static str) {
458    cfg_if::cfg_if! {
459        if #[cfg(target_arch = "wasm32")] {
460            wasm_bindgen_futures::spawn_local(async move { start::<E>(title).await })
461        } else {
462            pollster::block_on(start::<E>(title));
463        }
464    }

468/// Parse the query string as returned by `web_sys::window()?.location().search()?` and get a
469/// specific key out of it.
470pub fn parse_url_query_string<'a>(query: &'a str, search_key: &str) -> Option<&'a str> {
471    let query_string = query.strip_prefix('?')?;
472
473    for pair in query_string.split('&') {
474        let mut pair = pair.split('=');
475        let key = pair.next()?;
476        let value = pair.next()?;
477
478        if key == search_key {
479            return Some(value);
480        }
481    }
482
483    None
484}

523            })
524            .run_async(move |ctx| async move {
525                let format = if E::SRGB {
526                    wgpu::TextureFormat::Rgba8UnormSrgb
527                } else {
528                    wgpu::TextureFormat::Rgba8Unorm
529                };
530                let dst_texture = ctx.device.create_texture(&wgpu::TextureDescriptor {
531                    label: Some("destination"),
532                    size: wgpu::Extent3d {
533                        width: params.width,
534                        height: params.height,
535                        depth_or_array_layers: 1,
536                    },
537                    mip_level_count: 1,
538                    sample_count: 1,
539                    dimension: wgpu::TextureDimension::D2,
540                    format,
541                    usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::COPY_SRC,
542                    view_formats: &[],
543                });
544
545                let dst_view = dst_texture.create_view(&wgpu::TextureViewDescriptor::default());
546
547                let dst_buffer = ctx.device.create_buffer(&wgpu::BufferDescriptor {
548                    label: Some("image map buffer"),
549                    size: params.width as u64 * params.height as u64 * 4,
550                    usage: wgpu::BufferUsages::COPY_DST | wgpu::BufferUsages::MAP_READ,
551                    mapped_at_creation: false,
552                });
553
554                let mut example = E::init(
555                    &wgpu::SurfaceConfiguration {
556                        usage: wgpu::TextureUsages::RENDER_ATTACHMENT,
557                        format,
558                        width: params.width,
559                        height: params.height,
560                        desired_maximum_frame_latency: 2,
561                        present_mode: wgpu::PresentMode::Fifo,
562                        alpha_mode: wgpu::CompositeAlphaMode::Auto,
563                        view_formats: vec![format],
564                    },
565                    &ctx.adapter,
566                    &ctx.device,
567                    &ctx.queue,
568                );
569
570                example.render(&dst_view, &ctx.device, &ctx.queue);
571
572                let mut cmd_buf = ctx
573                    .device
574                    .create_command_encoder(&wgpu::CommandEncoderDescriptor::default());
575
576                cmd_buf.copy_texture_to_buffer(
577                    wgpu::TexelCopyTextureInfo {
578                        texture: &dst_texture,
579                        mip_level: 0,
580                        origin: wgpu::Origin3d::ZERO,
581                        aspect: wgpu::TextureAspect::All,
582                    },
583                    wgpu::TexelCopyBufferInfo {
584                        buffer: &dst_buffer,
585                        layout: wgpu::TexelCopyBufferLayout {
586                            offset: 0,
587                            bytes_per_row: Some(params.width * 4),
588                            rows_per_image: None,
589                        },
590                    },
591                    wgpu::Extent3d {
592                        width: params.width,
593                        height: params.height,
594                        depth_or_array_layers: 1,
595                    },
596                );
597
598                ctx.queue.submit(Some(cmd_buf.finish()));
599
600                let dst_buffer_slice = dst_buffer.slice(..);
601                dst_buffer_slice.map_async(wgpu::MapMode::Read, |_| ());
602                ctx.async_poll(wgpu::PollType::wait_indefinitely())
603                    .await
604                    .unwrap();
605                let bytes = dst_buffer_slice.get_mapped_range().to_vec();
606
607                wgpu_test::image::compare_image_output(
608                    dbg!(env!("CARGO_MANIFEST_DIR").to_string() + "/../../" + params.image_path),
609                    &ctx.adapter_info,
610                    params.width,
611                    params.height,
612                    &bytes,
613                    params.comparisons,
614                )
615                .await;
616            })
617    }
